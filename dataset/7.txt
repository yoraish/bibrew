פָּרָשַׁת וָאֵרָא

קְרִיאַת הַתּוֹרָה וְהַהַפְטָרָה · מִקְרָא וְתַרְגּוּם · שְׁנַֽיִם מִקְרָא וְאֶחָד תַּרְגּוּם

מִקְרָא וְתַרְגּוּם וְרַשִׁ"י (בהכנה) · מִקְרָא מְבוֹאָר





ראשון


ס פָּרָשַׁת לְשׁוֹנוֹת הַגְּאֻלָּה





* * *



ו ב וַיְדַבֵּ֥ר אֱלֹהִ֖ים אֶל־מֹשֶׁ֑ה וַיֹּ֥אמֶר אֵלָ֖יו אֲנִ֥י יְהֹוָֽה׃ אֱלֹהִים דִּבֵּר עִם מֹשֶׁה וְאָמַר לוֹ: אֲנִי ה'.





* * *



ג וָאֵרָ֗א אֶל־אַבְרָהָ֛ם אֶל־יִצְחָ֥ק וְאֶֽל־יַעֲקֹ֖ב בְּאֵ֣ל שַׁדָּ֑י וּשְׁמִ֣י יְהֹוָ֔ה לֹ֥א נוֹדַ֖עְתִּי לָהֶֽם׃ אֲנִי הִתְגַּלֵיתִי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב בַּשֵּׁם אֵל-שַׁדַּי כְּלוֹמַר בַּעַל הַכֹּחַ לְקַיֵּם אֶת הַבְטָחוֹתָיו, אֲבָל לֹא הוֹדַעְתִּי לָהֶם אֶת שְׁמִי הָעִקָּרִי: יהוה – מְהַוֶּה אֶת הָעוֹלָם וּמְשַׁנֶּה אוֹתוֹ.





* * *



ד וְגַ֨ם הֲקִמֹ֤תִי אֶת־בְּרִיתִי֙ אִתָּ֔ם לָתֵ֥ת לָהֶ֖ם אֶת־אֶ֣רֶץ כְּנָ֑עַן אֵ֛ת אֶ֥רֶץ מְגֻרֵיהֶ֖ם אֲשֶׁר־גָּ֥רוּ בָֽהּ׃ וְגַם עָשִׂיתִי אִתָּם הֶסְכֵּם שֶׁאֶתֵּן לָהֶם אֶת אֶרֶץ כְּנַעַן – אֶת הָאָרֶץ שֶׁגָּרוּ בָּהּ כְּתוֹשָׁבִים זְמַנִּיִּים.





* * *



ה וְגַ֣ם ׀ אֲנִ֣י שָׁמַ֗עְתִּי אֶֽת־נַאֲקַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֥ר מִצְרַ֖יִם מַעֲבִדִ֣ים אֹתָ֑ם וָאֶזְכֹּ֖ר אֶת־בְּרִיתִֽי׃ וְגַם שָׁמַעְתִּי אֶת הַזְּעָקוֹת שֶׁל בְּנֵי יִשְׂרָאֵל שֶׁמִּצְרַיִם מַעֲבִידִים אוֹתָם, וְזָכַרְתִּי אֶת הַהֶסְכֵּם.





* * *



ו לָכֵ֞ן אֱמֹ֥ר לִבְנֵֽי־יִשְׂרָאֵל֮ אֲנִ֣י יְהֹוָה֒ וְהוֹצֵאתִ֣י אֶתְכֶ֗ם מִתַּ֙חַת֙ סִבְלֹ֣ת מִצְרַ֔יִם וְהִצַּלְתִּ֥י אֶתְכֶ֖ם מֵעֲבֹדָתָ֑ם וְגָאַלְתִּ֤י אֶתְכֶם֙ בִּזְר֣וֹעַ נְטוּיָ֔ה וּבִשְׁפָטִ֖ים גְּדֹלִֽים׃ לָכֵן אֱמוֹר לִבְנֵי יִשְׂרָאֵל: אֲנִי ה', אוֹצִיא אֶתְכֶם מֵהַסֵּבֶל שֶׁל מִצְרַיִם, וְאַצִּיל אֶתְכֶם מִלַּעֲבוֹד אוֹתָם, וְאֶגְאַל אֶתְכֶם בִּזְרוֹעַ מוּנֶפֶת לְהַכּוֹת, וּבְעוֹנָשִׁים קָשִׁים.





* * *



ז וְלָקַחְתִּ֨י אֶתְכֶ֥ם לִי֙ לְעָ֔ם וְהָיִ֥יתִי לָכֶ֖ם לֵֽאלֹהִ֑ים וִֽידַעְתֶּ֗ם כִּ֣י אֲנִ֤י יְהֹוָה֙ אֱלֹ֣הֵיכֶ֔ם הַמּוֹצִ֣יא אֶתְכֶ֔ם מִתַּ֖חַת סִבְל֥וֹת מִצְרָֽיִם׃ אֶקַּח אֶתְכֶם לִי לְעָם וְאֶהְיֶה לָכֶם לֵאלֹהִים שֶׁתַּעַבְדוּ אוֹתִי, וְתֵדְעוּ שֶׁאֲנִי ה' אֱלֹהֵיכֶם שֶׁמּוֹצִיא אֶתְכֶם מֵהַסֵּבֶל שֶׁל מִצְרַיִם.





* * *



ח וְהֵבֵאתִ֤י אֶתְכֶם֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֙אתִי֙ אֶת־יָדִ֔י לָתֵ֣ת אֹתָ֔הּ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹ֑ב וְנָתַתִּ֨י אֹתָ֥הּ לָכֶ֛ם מוֹרָשָׁ֖ה אֲנִ֥י יְהֹוָֽה׃ וְאָבִיא אֶתְכֶם לָאָרֶץ שֶׁנִּשְׁבַּעְתִּי לָתֵת לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב, וְאֶתֵּן אוֹתָהּ לָכֶם לִירוּשָׁה – אֲנִי ה' אֲקַיֵּם זֹאת!





* * *



ט וַיְדַבֵּ֥ר מֹשֶׁ֛ה כֵּ֖ן אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל וְלֹ֤א שָֽׁמְעוּ֙ אֶל־מֹשֶׁ֔ה מִקֹּ֣צֶר ר֔וּחַ וּמֵעֲבֹדָ֖ה קָשָֽׁה׃ מֹשֶׁה סִפֵּר דְּבָרִים אֵלֶּה לִבְנֵי יִשְׂרָאֵל, וְהֵם לֹא הִקְשִׁיבוּ לְמֹשֶׁה בִּגְלַל קוֹצֶר רוּחַ וַעֲבוֹדָה קָשָׁה.





* * *



פ





* * *



י וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:





* * *



יא בֹּ֣א דַבֵּ֔ר אֶל־פַּרְעֹ֖ה מֶ֣לֶךְ מִצְרָ֑יִם וִֽישַׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאַרְצֽוֹ׃ בֹּא דַּבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם שֶׁיְּשַׁחְרֵר אֶת בְּנֵי יִשְׂרָאֵל מֵאַרְצוֹ.





* * *



יב וַיְדַבֵּ֣ר מֹשֶׁ֔ה לִפְנֵ֥י יְהֹוָ֖ה לֵאמֹ֑ר הֵ֤ן בְּנֵֽי־יִשְׂרָאֵל֙ לֹֽא־שָׁמְע֣וּ אֵלַ֔י וְאֵיךְ֙ יִשְׁמָעֵ֣נִי פַרְעֹ֔ה וַאֲנִ֖י עֲרַ֥ל שְׂפָתָֽיִם׃ מֹשֶׁה דִּבֵּר לִפְנֵי ה' וְאָמַר: הֲרֵי אֲפִלּוּ בְּנֵי יִשְׂרָאֵל לֹא שָׁמְעוּ אֵלַי, וְאֵיךְ פַּרְעֹה יַקְשִׁיב לִי, וַהֲרֵי אֲנִי קְשֵׁה דִּבּוּר!





* * *



פ





* * *



יג וַיְדַבֵּ֣ר יְהֹוָה֮ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹן֒ וַיְצַוֵּם֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאֶל־פַּרְעֹ֖ה מֶ֣לֶךְ מִצְרָ֑יִם לְהוֹצִ֥יא אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאֶ֥רֶץ מִצְרָֽיִם׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְצִוָּה אוֹתָם לְדַבֵּר אֶל בְּנֵי יִשְׂרָאֵל וְאֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם, כְּדֵי לְהוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם.





* * *





שני


ס פָּרָשַׁת רָאשֵׁי הָאָבוֹת





* * *



יד אֵ֖לֶּה רָאשֵׁ֣י בֵית־אֲבֹתָ֑ם בְּנֵ֨י רְאוּבֵ֜ן בְּכֹ֣ר יִשְׂרָאֵ֗ל חֲנ֤וֹךְ וּפַלּוּא֙ חֶצְרֹ֣ן וְכַרְמִ֔י אֵ֖לֶּה מִשְׁפְּחֹ֥ת רְאוּבֵֽן׃ אֵלֶּה רָאשֵׁי מִשְׁפְּחוֹת בְּנֵי יִשְׂרָאֵל: בְּנֵי רְאוּבֵן בְּכוֹר יִשְׂרָאֵל: חֲנוֹךְ, פַּלּוּא, חֶצְרוֹן, וְכַרְמִי, אֵלֶּה מִשְׁפְּחוֹת רְאוּבֵן.





* * *



טו וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֨ל וְיָמִ֤ין וְאֹ֙הַד֙ וְיָכִ֣ין וְצֹ֔חַר וְשָׁא֖וּל בֶּן־הַֽכְּנַעֲנִ֑ית אֵ֖לֶּה מִשְׁפְּחֹ֥ת שִׁמְעֽוֹן׃ וּבְנֵי שִׁמְעוֹן: יְמוּאֵל, יָמִין, אֹהַד, יָכִין, צוֹחַר, וְשָׁאוּל בֶּן הַכְּנַעֲנִית, אֵלֶּה מִשְׁפְּחוֹת שִׁמְעוֹן.





* * *



טז וְאֵ֨לֶּה שְׁמ֤וֹת בְּנֵֽי־לֵוִי֙ לְתֹ֣לְדֹתָ֔ם גֵּרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִ֑י וּשְׁנֵי֙ חַיֵּ֣י לֵוִ֔י שֶׁ֧בַע וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ וְאֵלֶּה שְׁמוֹת בְּנֵי לֵוִי כְּסֵדֶר לֵידָתָם: גֵּרְשׁוֹן, קְהָת, וּמְרָרִי, וּמִסְפַּר שְׁנוֹת חַיֵּי לֵוִי – מֵאָה שְׁלוֹשִׁים וְשֶׁבַע שָׁנִים.





* * *



יז בְּנֵ֥י גֵרְשׁ֛וֹן לִבְנִ֥י וְשִׁמְעִ֖י לְמִשְׁפְּחֹתָֽם׃ מִשְׁפְּחוֹת בְּנֵי גֵּרְשׁוֹן: לִבְנִי וְשִׁמְעִי.





* * *



יח וּבְנֵ֣י קְהָ֔ת עַמְרָ֣ם וְיִצְהָ֔ר וְחֶבְר֖וֹן וְעֻזִּיאֵ֑ל וּשְׁנֵי֙ חַיֵּ֣י קְהָ֔ת שָׁלֹ֧שׁ וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ וּבְנֵי קְהָת: עַמְרָם, יִצְהָר, חֶבְרוֹן, וְעֻזִּיאֵל, וּמִסְפַּר שְׁנוֹת חַיֵּי קְהָת – מֵאָה שְׁלוֹשִׁים וְשָׁלוֹשׁ שָׁנִים.





* * *



יט וּבְנֵ֥י מְרָרִ֖י מַחְלִ֣י וּמוּשִׁ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְתֹלְדֹתָֽם׃ וּבְנֵי מְרָרִי: מַחְלִי וּמוּשִׁי, אֵלֶּה מִשְׁפְּחוֹת הַלֵּוִי כְּסֵדֶר לֵידָתָם.





* * *



כ וַיִּקַּ֨ח עַמְרָ֜ם אֶת־יוֹכֶ֤בֶד דֹּֽדָתוֹ֙ ל֣וֹ לְאִשָּׁ֔ה וַתֵּ֣לֶד ל֔וֹ אֶֽת־אַהֲרֹ֖ן וְאֶת־מֹשֶׁ֑ה וּשְׁנֵי֙ חַיֵּ֣י עַמְרָ֔ם שֶׁ֧בַע וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ עַמְרָם נָשָׂא לְאִשָּׁה אֶת יוֹכֶבֶד דּוֹדָתוֹ, וְהִיא יָלְדָה לוֹ אֶת אַהֲרֹן וְאֶת מֹשֶׁה, וּמִסְפַּר שְׁנוֹת חַיֵּי עַמְרָם – מֵאָה וּשְׁלוֹשִׁים וְשֶׁבַע שָׁנִים.





* * *



כא וּבְנֵ֖י יִצְהָ֑ר קֹ֥רַח וָנֶ֖פֶג וְזִכְרִֽי׃ וּבְנֵי יִצְהָר: קוֹרַח, נֶפֶג, וְזִכְרִי.





* * *



כב וּבְנֵ֖י עֻזִּיאֵ֑ל מִֽישָׁאֵ֥ל וְאֶלְצָפָ֖ן וְסִתְרִֽי׃ וּבְנֵי עֻזִּיאֵל: מִישָׁאֵל, אֶלְצָפָן, וְסִתְרִי.





* * *



כג וַיִּקַּ֨ח אַהֲרֹ֜ן אֶת־אֱלִישֶׁ֧בַע בַּת־עַמִּינָדָ֛ב אֲח֥וֹת נַחְשׁ֖וֹן ל֣וֹ לְאִשָּׁ֑ה וַתֵּ֣לֶד ל֗וֹ אֶת־נָדָב֙ וְאֶת־אֲבִיה֔וּא אֶת־אֶלְעָזָ֖ר וְאֶת־אִֽיתָמָֽר׃ אַהֲרֹן נָשָׂא לְאִשָּׁה אֶת אֱלִישֶׁבַע בַּת עַמִּינָדָב אֲחוֹתוֹ שֶׁל נַחְשׁוֹן, וְהִיא יָלְדָה לוֹ אֶת נָדָב אֲבִיהוּא אֶלְעָזָר וְאִיתָמָר.





* * *



כד וּבְנֵ֣י קֹ֔רַח אַסִּ֥יר וְאֶלְקָנָ֖ה וַאֲבִיאָסָ֑ף אֵ֖לֶּה מִשְׁפְּחֹ֥ת הַקׇּרְחִֽי׃ וּבְנֵי קוֹרַח: אַסִּיר אֶלְקָנָה וַאֲבִיאָסָף, אֵלֶּה מִשְׁפְּחוֹת קוֹרַח.





* * *



כה וְאֶלְעָזָ֨ר בֶּֽן־אַהֲרֹ֜ן לָקַֽח־ל֨וֹ מִבְּנ֤וֹת פּֽוּטִיאֵל֙ ל֣וֹ לְאִשָּׁ֔ה וַתֵּ֥לֶד ל֖וֹ אֶת־פִּֽינְחָ֑ס אֵ֗לֶּה רָאשֵׁ֛י אֲב֥וֹת הַלְוִיִּ֖ם לְמִשְׁפְּחֹתָֽם׃ וְאֶלְעָזָר בֶּן אַהֲרֹן נָשָׂא לוֹ לְאִשָּׁה אֶת אַחַת מִבְּנוֹת פּוּטִיאֵל וְהִיא יָלְדָה לוֹ אֶת פִּינְחָס, אֵלֶּה רָאשֵׁי אֲבוֹת מִשְׁפְּחוֹת הַלְּוִיִים.





* * *



כו ה֥וּא אַהֲרֹ֖ן וּמֹשֶׁ֑ה אֲשֶׁ֨ר אָמַ֤ר יְהֹוָה֙ לָהֶ֔ם הוֹצִ֜יאוּ אֶת־בְּנֵ֧י יִשְׂרָאֵ֛ל מֵאֶ֥רֶץ מִצְרַ֖יִם עַל־צִבְאֹתָֽם׃ הֵם אַהֲרֹן וּמֹשֶׁה שֶׁה' אָמַר לָהֶם הוֹצִיאוּ מֵאֶרֶץ מִצְרַיִם אֶת בְּנֵי יִשְׂרָאֵל וְכָל קְהִלּוֹתֵיהֶם.





* * *



כז הֵ֗ם הַֽמְדַבְּרִים֙ אֶל־פַּרְעֹ֣ה מֶֽלֶךְ־מִצְרַ֔יִם לְהוֹצִ֥יא אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מִמִּצְרָ֑יִם ה֥וּא מֹשֶׁ֖ה וְאַהֲרֹֽן׃ הֵם אֵלֶּה שֶׁמְּדַבְּרִים אֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם לְהוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרַיִם, הֵם מֹשֶׁה וְאַהֲרֹן.





* * *



כח וַיְהִ֗י בְּי֨וֹם דִּבֶּ֧ר יְהֹוָ֛ה אֶל־מֹשֶׁ֖ה בְּאֶ֥רֶץ מִצְרָֽיִם׃ וְכָךְ הָיָה בַּיּוֹם שֶׁה' דִּבֵּר אֶל מֹשֶׁה בְּאֶרֶץ מִצְרַיִם:





* * *





שלישי


ס





* * *



כט וַיְדַבֵּ֧ר יְהֹוָ֛ה אֶל־מֹשֶׁ֥ה לֵּאמֹ֖ר אֲנִ֣י יְהֹוָ֑ה דַּבֵּ֗ר אֶל־פַּרְעֹה֙ מֶ֣לֶךְ מִצְרַ֔יִם אֵ֛ת כׇּל־אֲשֶׁ֥ר אֲנִ֖י דֹּבֵ֥ר אֵלֶֽיךָ׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר: אֲנִי ה', דַּבֵּר אֶל פַּרְעֹה מֶלֶךְ מִצְרַיִם אֶת כָּל מַה שֶּׁאֲנִי מְדַבֵּר אֵלֶיךָ.





* * *



ל וַיֹּ֥אמֶר מֹשֶׁ֖ה לִפְנֵ֣י יְהֹוָ֑ה הֵ֤ן אֲנִי֙ עֲרַ֣ל שְׂפָתַ֔יִם וְאֵ֕יךְ יִשְׁמַ֥ע אֵלַ֖י פַּרְעֹֽה׃ מֹשֶׁה אָמַר לִפְנֵי ה': הֲרֵי אֲנִי מִתְקַשֶּׁה בְּדִבּוּר, וְאֵיךְ יִשְׁמַע אֵלַי פַּרְעֹה?!





* * *



פ פָּרָשַׁת שְׁלִיחַת מֹשֶׁה וְאַהֲרֹן





* * *



ז א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה רְאֵ֛ה נְתַתִּ֥יךָ אֱלֹהִ֖ים לְפַרְעֹ֑ה וְאַהֲרֹ֥ן אָחִ֖יךָ יִהְיֶ֥ה נְבִיאֶֽךָ׃ ה' אָמַר לְמֹשֶׁה: רְאֵה מִנִּיתִי אוֹתְךָ שֶׁתִּהְיֶה שׁוֹפֵט מַעֲנִישׁ לְפַרְעֹה, וְאַהֲרֹן אָחִיךָ יִהְיֶה הַדּוֹבֵר שֶׁלְּךָ.





* * *



ב אַתָּ֣ה תְדַבֵּ֔ר אֵ֖ת כׇּל־אֲשֶׁ֣ר אֲצַוֶּ֑ךָּ וְאַהֲרֹ֤ן אָחִ֙יךָ֙ יְדַבֵּ֣ר אֶל־פַּרְעֹ֔ה וְשִׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאַרְצֽוֹ׃ אַתָּה תְּדַבֵּר אֶת כָּל מַה שֶּׁאֲצַוֶּה אוֹתְךָ, וְאַהֲרֹן אָחִיךָ יְדַבֵּר אֶל פַּרְעֹה, וְהוּא יְשַׁחְרֵר אֶת בְּנֵי יִשְׂרָאֵל מֵאַרְצוֹ.





* * *



ג וַאֲנִ֥י אַקְשֶׁ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְהִרְבֵּיתִ֧י אֶת־אֹתֹתַ֛י וְאֶת־מוֹפְתַ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ וַאֲנִי אַקְשִׁיחַ אֶת לֵב פַּרְעֹה, וְאֶעֱשֶׂה סִימָנִים רַבִּים לְכוֹחִי וְנִסִּים רַבִּים בְּאֶרֶץ מִצְרַיִם.





* * *



ד וְלֹֽא־יִשְׁמַ֤ע אֲלֵכֶם֙ פַּרְעֹ֔ה וְנָתַתִּ֥י אֶת־יָדִ֖י בְּמִצְרָ֑יִם וְהוֹצֵאתִ֨י אֶת־צִבְאֹתַ֜י אֶת־עַמִּ֤י בְנֵֽי־יִשְׂרָאֵל֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בִּשְׁפָטִ֖ים גְּדֹלִֽים׃ וּפַרְעֹה לֹא יִשְׁמַע לָכֶם, וְאַכֶּה אֶת מִצְרַיִם, וְאוֹצִיא אֶת קְהִלּוֹתַי אֶת עַמִּי בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם עַל יְדֵי עוֹנָשִׁים גְּדוֹלִים לַמִּצְרִים.





* * *



ה וְיָדְע֤וּ מִצְרַ֙יִם֙ כִּֽי־אֲנִ֣י יְהֹוָ֔ה בִּנְטֹתִ֥י אֶת־יָדִ֖י עַל־מִצְרָ֑יִם וְהוֹצֵאתִ֥י אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מִתּוֹכָֽם׃ מִצְרַיִם יֵדְעוּ שֶׁאֲנִי ה' כְּשֶׁאָרִים אֶת יָדִי עַל מִצְרַיִם לְהַכּוֹת אוֹתָם, וְאוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל מִתּוֹכָם.





* * *



ו וַיַּ֥עַשׂ מֹשֶׁ֖ה וְאַהֲרֹ֑ן כַּאֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֹתָ֖ם כֵּ֥ן עָשֽׂוּ׃ מֹשֶׁה וְאַהֲרֹן בִּצְּעוּ – כְּמוֹ שֶׁה' צִוָּה אוֹתָם כָּךְ הֵם עָשׂוּ.





* * *



ז וּמֹשֶׁה֙ בֶּן־שְׁמֹנִ֣ים שָׁנָ֔ה וְאַֽהֲרֹ֔ן בֶּן־שָׁלֹ֥שׁ וּשְׁמֹנִ֖ים שָׁנָ֑ה בְּדַבְּרָ֖ם אֶל־פַּרְעֹֽה׃ מֹשֶׁה הָיָה בֶּן שְׁמוֹנִים שָׁנָה וְאַהֲרֹן בֶּן שְׁמוֹנִים וְשָׁלוֹשׁ שָׁנִים כְּשֶׁהֵם דִּבְּרוּ אֶל פַּרְעֹה.





* * *





רביעי


פ





* * *



ח וַיֹּ֣אמֶר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' אָמַר לְמֹשֶׁה וּלְאַהֲרֹן:





* * *



ט כִּי֩ יְדַבֵּ֨ר אֲלֵכֶ֤ם פַּרְעֹה֙ לֵאמֹ֔ר תְּנ֥וּ לָכֶ֖ם מוֹפֵ֑ת וְאָמַרְתָּ֣ אֶֽל־אַהֲרֹ֗ן קַ֧ח אֶֽת־מַטְּךָ֛ וְהַשְׁלֵ֥ךְ לִפְנֵֽי־פַרְעֹ֖ה יְהִ֥י לְתַנִּֽין׃ כְּשֶׁפַּרְעֹה יְדַבֵּר אֲלֵכֶם וְיֹאמַר: עֲשׂוּ נֵס לְסִימָן שֶׁאַתֶּם שְׁלִיחֵי ה', תֹּאמַר לְאַהֲרֹן: קַח אֶת הַמַקֵּל שֶׁלְּךָ וּזְרוֹק לִפְנֵי פַּרְעֹה וְהוּא יֵהָפֵךְ לְתַנִּין.





* * *



י וַיָּבֹ֨א מֹשֶׁ֤ה וְאַהֲרֹן֙ אֶל־פַּרְעֹ֔ה וַיַּ֣עֲשׂוּ כֵ֔ן כַּאֲשֶׁ֖ר צִוָּ֣ה יְהֹוָ֑ה וַיַּשְׁלֵ֨ךְ אַהֲרֹ֜ן אֶת־מַטֵּ֗הוּ לִפְנֵ֥י פַרְעֹ֛ה וְלִפְנֵ֥י עֲבָדָ֖יו וַיְהִ֥י לְתַנִּֽין׃ מֹשֶׁה וְאַהֲרֹן בָּאוּ לְפַרְעֹה וְעָשׂוּ כָּךְ – כְּמוֹ שֶׁה' צִוָּה, אַהֲרֹן זָרַק אֶת מַקְלוֹ לִפְנֵי פַּרְעֹה וְלִפְנֵי עֲבָדָיו וְהוּא נֶהְפַּךְ לְתַנִּין.





* * *



יא וַיִּקְרָא֙ גַּם־פַּרְעֹ֔ה לַֽחֲכָמִ֖ים וְלַֽמְכַשְּׁפִ֑ים וַיַּֽעֲשׂ֨וּ גַם־הֵ֜ם חַרְטֻמֵּ֥י מִצְרַ֛יִם בְּלַהֲטֵיהֶ֖ם כֵּֽן׃ גַּם פַּרְעֹה קָרָא לַחֲכָמִים וְלַמְכַשְּׁפִים, וְגַם הֵם – מְכַשְּׁפֵי מִצְרַיִם, עָשׂוּ כָּךְ בִּכְשָׁפֵיהֶם.





* * *



יב וַיַּשְׁלִ֙יכוּ֙ אִ֣ישׁ מַטֵּ֔הוּ וַיִּהְי֖וּ לְתַנִּינִ֑ם וַיִּבְלַ֥ע מַטֵּֽה־אַהֲרֹ֖ן אֶת־מַטֹּתָֽם׃ הֵם הִשְׁלִיכוּ כָּל אֶחָד אֶת מַקְלוֹ וְהֵם הָפְכוּ לְתַנִּינִים, וּמַקְלוֹ שֶׁל אַהֲרֹן בָּלַע אֶת מַקְּלוֹתֵיהֶם.





* * *



יג וַיֶּחֱזַק֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ פַּרְעֹה הִתְעַקֵּשׁ וְלֹא שָׁמַע לָהֶם, כְּמוֹ שֶׁה' דִּבֵּר.





* * *



ס פָּרָשַׁת מַכַּת הַדָּם





* * *



יד וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה כָּבֵ֖ד לֵ֣ב פַּרְעֹ֑ה מֵאֵ֖ן לְשַׁלַּ֥ח הָעָֽם׃ ה' אָמַר לְמֹשֶׁה: פַּרְעֹה עַקְשָׁן וּמְסָרֵב לְשַׁחְרֵר אֶת הָעָם.





* * *



טו לֵ֣ךְ אֶל־פַּרְעֹ֞ה בַּבֹּ֗קֶר הִנֵּה֙ יֹצֵ֣א הַמַּ֔יְמָה וְנִצַּבְתָּ֥ לִקְרָאת֖וֹ עַל־שְׂפַ֣ת הַיְאֹ֑ר וְהַמַּטֶּ֛ה אֲשֶׁר־נֶהְפַּ֥ךְ לְנָחָ֖שׁ תִּקַּ֥ח בְּיָדֶֽךָ׃ לֵךְ לְפַרְעֹה בַּבּוֹקֶר הִנֵּה הוּא יוֹצֵא לַמַּיִם, תַּעֲמוֹד לִקְרָאתוֹ עַל שְׂפַת הַיְּאוֹר וְתִקַּח בְּיָדְךָ אֶת הַמַּקֵּל שֶׁנֶּהְפַּךְ לְנָחָשׁ.





* * *



טז וְאָמַרְתָּ֣ אֵלָ֗יו יְהֹוָ֞ה אֱלֹהֵ֤י הָעִבְרִים֙ שְׁלָחַ֤נִי אֵלֶ֙יךָ֙ לֵאמֹ֔ר שַׁלַּח֙ אֶת־עַמִּ֔י וְיַֽעַבְדֻ֖נִי בַּמִּדְבָּ֑ר וְהִנֵּ֥ה לֹא־שָׁמַ֖עְתָּ עַד־כֹּֽה׃ וֶאֱמוֹר לוֹ: ה' אֱלֹהֵי הָעִבְרִים שָׁלַח אוֹתִי אֵלֶיךָ לוֹמַר לְךָ: שַׁחְרֵר אֶת עַמִּי וְיַעַבְדוּ לְפָנַי בַּמִּדְבָּר, וְלֹא שָׁמַעְתָּ בְּקוֹלִי עַד עַכְשָׁיו.





* * *



יז כֹּ֚ה אָמַ֣ר יְהֹוָ֔ה בְּזֹ֣את תֵּדַ֔ע כִּ֖י אֲנִ֣י יְהֹוָ֑ה הִנֵּ֨ה אָנֹכִ֜י מַכֶּ֣ה ׀ בַּמַּטֶּ֣ה אֲשֶׁר־בְּיָדִ֗י עַל־הַמַּ֛יִם אֲשֶׁ֥ר בַּיְאֹ֖ר וְנֶהֶפְכ֥וּ לְדָֽם׃ כָּךְ אָמַר ה': כָּךְ תֵּדַע שֶׁאֲנִי ה' – הִנֵּה אֲנִי מַכֶּה בַּמַּקֵּל שֶׁבְּיָדִי עַל הַמַּיִם שֶׁבַּיְאוֹר וְהֵם יֵהָפְכוּ לְדָם.





* * *



יח וְהַדָּגָ֧ה אֲשֶׁר־בַּיְאֹ֛ר תָּמ֖וּת וּבָאַ֣שׁ הַיְאֹ֑ר וְנִלְא֣וּ מִצְרַ֔יִם לִשְׁתּ֥וֹת מַ֖יִם מִן־הַיְאֹֽר׃ וְהַדָּגִים שֶׁבַּיְאוֹר יָמוּתוּ וְהַיְּאוֹר יַסְרִיחַ, וּמִצְרַיִם יִתְעַיְּפוּ מִנִּסְיוֹנוֹתֵיהֶם לִשְׁתּוֹת מַיִם מֵהַיְּאוֹר.





* * *



ס





* * *



יט וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה אֱמֹ֣ר אֶֽל־אַהֲרֹ֡ן קַ֣ח מַטְּךָ֣ וּנְטֵֽה־יָדְךָ֩ עַל־מֵימֵ֨י מִצְרַ֜יִם עַֽל־נַהֲרֹתָ֣ם ׀ עַל־יְאֹרֵיהֶ֣ם וְעַל־אַגְמֵיהֶ֗ם וְעַ֛ל כׇּל־מִקְוֵ֥ה מֵימֵיהֶ֖ם וְיִֽהְיוּ־דָ֑ם וְהָ֤יָה דָם֙ בְּכׇל־אֶ֣רֶץ מִצְרַ֔יִם וּבָעֵצִ֖ים וּבָאֲבָנִֽים׃ ה' אָמַר לְמֹשֶׁה: אֱמוֹר לְאַהֲרֹן – קַח אֶת הַמַקֵּל שֶׁלְּךָ וְהָנֵף אֶת יָדְךָ עַל מֵימֵי מִצְרַיִם, עַל נַהֲרוֹתֵיהֶם עַל עֲרוּצֵי הַיְּאוֹר וְעַל אַגְמֵיהֶם וְעַל כָּל מַאֲגָרֵי הַמַּיִם שֶׁלָּהֶם, וְהֵם יֵהָפְכוּ לְדָם, וְיִהְיֶה דָּם בְּכָל אֶרֶץ מִצְרַיִם וּבִכְלֵי הָעֵץ וְהָאֶבֶן שֶׁמְּלֵאִים מַיִם.





* * *



כ וַיַּֽעֲשׂוּ־כֵן֩ מֹשֶׁ֨ה וְאַהֲרֹ֜ן כַּאֲשֶׁ֣ר ׀ צִוָּ֣ה יְהֹוָ֗ה וַיָּ֤רֶם בַּמַּטֶּה֙ וַיַּ֤ךְ אֶת־הַמַּ֙יִם֙ אֲשֶׁ֣ר בַּיְאֹ֔ר לְעֵינֵ֣י פַרְעֹ֔ה וּלְעֵינֵ֖י עֲבָדָ֑יו וַיֵּהָ֥פְכ֛וּ כׇּל־הַמַּ֥יִם אֲשֶׁר־בַּיְאֹ֖ר לְדָֽם׃ מֹשֶׁה וְאַהֲרֹן עָשׂוּ כָּךְ כְּמוֹ שֶׁה' צִוָּה, אַהֲרֹן הֵנִיף אֶת יָדוֹ עִם הַמַּקֵּל, וְהִכָּה בַּמַּיִם שֶׁבַּיְאוֹר לְעֵינֵי פַּרְעֹה וּלְעֵינֵי עֲבָדָיו, וְכָל הַמַּיִם שֶׁבַּיְאוֹר נֶהֶפְכוּ לְדָם.





* * *



כא וְהַדָּגָ֨ה אֲשֶׁר־בַּיְאֹ֥ר מֵ֙תָה֙ וַיִּבְאַ֣שׁ הַיְאֹ֔ר וְלֹא־יָכְל֣וּ מִצְרַ֔יִם לִשְׁתּ֥וֹת מַ֖יִם מִן־הַיְאֹ֑ר וַיְהִ֥י הַדָּ֖ם בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ וְהַדָּגִים שֶׁבַּיְאוֹר מֵתוּ, וְהַיְּאוֹר הִסְרִיחַ וְהַמִּצְרַיִם לֹא יָכְלוּ לִשְׁתּוֹת מַיִם מֵהַיְּאוֹר, הַדָּם הָיָה בְּכָל אֶרֶץ מִצְרַיִם.





* * *



כב וַיַּֽעֲשׂוּ־כֵ֛ן חַרְטֻמֵּ֥י מִצְרַ֖יִם בְּלָטֵיהֶ֑ם וַיֶּחֱזַ֤ק לֵב־פַּרְעֹה֙ וְלֹא־שָׁמַ֣ע אֲלֵהֶ֔ם כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ גַּם מְכַשְּׁפֵי מִצְרַיִם עָשׂוּ כָּךְ בִּכְשָׁפֵיהֶם, וּפַרְעֹה הִתְעַקֵּשׁ וְלֹא שָׁמַע לָהֶם, כְּמוֹ שֶׁה' דִּבֵּר.





* * *



כג וַיִּ֣פֶן פַּרְעֹ֔ה וַיָּבֹ֖א אֶל־בֵּית֑וֹ וְלֹא־שָׁ֥ת לִבּ֖וֹ גַּם־לָזֹֽאת׃ פַּרְעֹה הִתְעַלֵּם וּבָא לְבֵיתוֹ וְלֹא הִתְיַחֵס גַּם לְזֶה.





* * *



כד וַיַּחְפְּר֧וּ כׇל־מִצְרַ֛יִם סְבִיבֹ֥ת הַיְאֹ֖ר מַ֣יִם לִשְׁתּ֑וֹת כִּ֣י לֹ֤א יָֽכְלוּ֙ לִשְׁתֹּ֔ת מִמֵּימֵ֖י הַיְאֹֽר׃ כָּל מִצְרַיִם חָפְרוּ בִּסְבִיבוֹת הַיְּאוֹר כְּדֵי לִמְצֹא מַיִם לִשְׁתּוֹת, כִּי הֵם לֹא יָכְלוּ לִשְׁתּוֹת מִמֵּימֵי הַיְּאוֹר.





* * *



כה וַיִּמָּלֵ֖א שִׁבְעַ֣ת יָמִ֑ים אַחֲרֵ֥י הַכּוֹת־יְהֹוָ֖ה אֶת־הַיְאֹֽר׃ עָבְרוּ שִׁבְעָה יָמִים אַחֲרֵי שֶׁה' הִכָּה אֶת הַיְּאוֹר.





* * *



פ פָּרָשַׁת מַכַּת צְפַרְדֵּעַ





* * *



כו וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה וְאָמַרְתָּ֣ אֵלָ֗יו כֹּ֚ה אָמַ֣ר יְהֹוָ֔ה שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וַה' אָמַר לְמֹשֶׁה: בֹּא לְפַרְעֹה וֶאֱמוֹר לוֹ – כָּךְ אָמַר ה': שַׁחְרֵר אֶת עַמִּי וְהֵם יַעַבְדוּ לְפָנַי.





* * *



כז וְאִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֑חַ הִנֵּ֣ה אָנֹכִ֗י נֹגֵ֛ף אֶת־כׇּל־גְּבוּלְךָ֖ בַּֽצְפַרְדְּעִֽים׃ וְאִם אַתָּה מְסָרֵב לְשַׁחְרֵר, הִנֵּה אֲנִי מַכֶּה אֶת כָּל הַשֶּׁטַח שֶׁלְּךָ בִּצְפַרְדְּעִים.





* * *



כח וְשָׁרַ֣ץ הַיְאֹר֮ צְפַרְדְּעִים֒ וְעָלוּ֙ וּבָ֣אוּ בְּבֵיתֶ֔ךָ וּבַחֲדַ֥ר מִשְׁכָּבְךָ֖ וְעַל־מִטָּתֶ֑ךָ וּבְבֵ֤ית עֲבָדֶ֙יךָ֙ וּבְעַמֶּ֔ךָ וּבְתַנּוּרֶ֖יךָ וּבְמִשְׁאֲרוֹתֶֽיךָ׃ הַיְּאוֹר יִשְׁרוֹץ צְפַרְדְּעִים, וְהֵם יַעֲלוּ וְיָבוֹאוּ בְּבֵיתְךָ וּבַחֲדַר הַשֵּׁנָה שֶׁלְּךָ וְעַל מִטָּתְךָ וּבְבֵית עֲבָדֶיךָ וּבְעַמְּךָ וּבְתַנּוּרֶיךָ וּבַבָּצֵק שֶׁלְּךָ.





* * *



כט וּבְכָ֥ה וּֽבְעַמְּךָ֖ וּבְכׇל־עֲבָדֶ֑יךָ יַעֲל֖וּ הַֽצְפַרְדְּעִֽים׃ וּבְךָ וּבְעַמְּךָ וּבְכָל עֲבָדֶיךָ יַעֲלוּ הַצְפַרְדְּעִים.





* * *



ח א וַיֹּ֣אמֶר יְהֹוָה֮ אֶל־מֹשֶׁה֒ אֱמֹ֣ר אֶֽל־אַהֲרֹ֗ן נְטֵ֤ה אֶת־יָדְךָ֙ בְּמַטֶּ֔ךָ עַ֨ל־הַנְּהָרֹ֔ת עַל־הַיְאֹרִ֖ים וְעַל־הָאֲגַמִּ֑ים וְהַ֥עַל אֶת־הַֽצְפַרְדְּעִ֖ים עַל־אֶ֥רֶץ מִצְרָֽיִם׃ ה' אָמַר לְמֹשֶׁה: אֱמוֹר לְאַהֲרֹן הָנֵף אֶת יָדְךָ עִם הַמַּקֵּל שֶׁלְּךָ עַל הַנְּהָרוֹת עַל עֲרוּצֵי הַיְּאוֹר וְעַל הָאֲגַמִּים, וְהַעֲלֵה אֶת הַצְפַרְדְּעִים עַל אֶרֶץ מִצְרַיִם.





* * *



ב וַיֵּ֤ט אַהֲרֹן֙ אֶת־יָד֔וֹ עַ֖ל מֵימֵ֣י מִצְרָ֑יִם וַתַּ֙עַל֙ הַצְּפַרְדֵּ֔עַ וַתְּכַ֖ס אֶת־אֶ֥רֶץ מִצְרָֽיִם׃ אַהֲרֹן הֵנִיף אֶת יָדוֹ עַל מֵימֵי מִצְרַיִם, וְהַצְּפַרְדֵּעִים עָלוּ וְכִסּוּ אֶת אֶרֶץ מִצְרַיִם.





* * *



ג וַיַּֽעֲשׂוּ־כֵ֥ן הַֽחַרְטֻמִּ֖ים בְּלָטֵיהֶ֑ם וַיַּעֲל֥וּ אֶת־הַֽצְפַרְדְּעִ֖ים עַל־אֶ֥רֶץ מִצְרָֽיִם׃ כָּךְ עָשׂוּ גַּם הַמְּכַשְּׁפִים בִּכְשָׁפֵיהֶם, וְהֵם הֶעֱלוּ צְפַרְדְּעִים עַל אֶרֶץ מִצְרַיִם.





* * *



ד וַיִּקְרָ֨א פַרְעֹ֜ה לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֗ן וַיֹּ֙אמֶר֙ הַעְתִּ֣ירוּ אֶל־יְהֹוָ֔ה וְיָסֵר֙ הַֽצְפַרְדְּעִ֔ים מִמֶּ֖נִּי וּמֵֽעַמִּ֑י וַאֲשַׁלְּחָה֙ אֶת־הָעָ֔ם וְיִזְבְּח֖וּ לַיהֹוָֽה׃ פַּרְעֹה קָרָא לְמֹשֶׁה וּלְאַהֲרֹן וְאָמַר: הִתְפַּלְּלוּ אֶל ה' וְיָסִיר אֶת הַצְפַרְדְּעִים מִמֶּנִּי וּמֵעַמִּי וַאֲשַׁחְרֵר אֶת הָעָם וְהֵם יִשְׁחֲטוּ קָרְבָּנוֹת לַה'.





* * *



ה וַיֹּ֨אמֶר מֹשֶׁ֣ה לְפַרְעֹה֮ הִתְפָּאֵ֣ר עָלַי֒ לְמָתַ֣י ׀ אַעְתִּ֣יר לְךָ֗ וְלַעֲבָדֶ֙יךָ֙ וּֽלְעַמְּךָ֔ לְהַכְרִית֙ הַֽצְפַרְדְּעִ֔ים מִמְּךָ֖ וּמִבָּתֶּ֑יךָ רַ֥ק בַּיְאֹ֖ר תִּשָּׁאַֽרְנָה׃ מֹשֶׁה אָמַר לְפַרְעֹה: הִתְכַּבֵּד, וּבַקֵּשׁ מִמֶּנִּי – לְמָתַי אֶתְפַּלֵּל בִּשְׁבִילְךָ וּבִשְׁבִיל עֲבָדֶיךָ וְעַמְּךָ לְהַשְׁמִיד אֶת הַצְפַרְדְּעִים מִמְּךָ וּמִבָּתֶּיךָ, שֶׁיִּשָּׁאֲרוּ רַק בַּיְאוֹר?





* * *



ו וַיֹּ֖אמֶר לְמָחָ֑ר וַיֹּ֙אמֶר֙ כִּדְבָ֣רְךָ֔ לְמַ֣עַן תֵּדַ֔ע כִּי־אֵ֖ין כַּיהֹוָ֥ה אֱלֹהֵֽינוּ׃ אָמַר: לְמָחָר, אָמַר: כִּדְבָרְךָ! כְּדֵי שֶׁתֵּדַע שֶׁאֵין כַּה' אֱלֹהֵינוּ.





* * *





חמישי


ז וְסָר֣וּ הַֽצְפַרְדְּעִ֗ים מִמְּךָ֙ וּמִבָּ֣תֶּ֔יךָ וּמֵעֲבָדֶ֖יךָ וּמֵעַמֶּ֑ךָ רַ֥ק בַּיְאֹ֖ר תִּשָּׁאַֽרְנָה׃ הַצְפַרְדְּעִים יָסוּרוּ מִמְּךָ וּמִבָּתֶּיךָ וּמֵעֲבָדֶיךָ וּמֵעַמְּךָ, וְיִשָּׁאֲרוּ רַק בַּיְאוֹר.





* * *



ח וַיֵּצֵ֥א מֹשֶׁ֛ה וְאַהֲרֹ֖ן מֵעִ֣ם פַּרְעֹ֑ה וַיִּצְעַ֤ק מֹשֶׁה֙ אֶל־יְהֹוָ֔ה עַל־דְּבַ֥ר הַֽצְפַרְדְּעִ֖ים אֲשֶׁר־שָׂ֥ם לְפַרְעֹֽה׃ מֹשֶׁה וְאַהֲרֹן יָצְאוּ מֵאֵצֶל פַּרְעֹה, וּמֹשֶׁה זָעַק בִּתְפִלָּה אֶל ה' עַל עִנְיַן הַצְפַרְדְּעִים שֶׁהוּא שָׂם לְפַרְעֹה.





* * *



ט וַיַּ֥עַשׂ יְהֹוָ֖ה כִּדְבַ֣ר מֹשֶׁ֑ה וַיָּמֻ֙תוּ֙ הַֽצְפַרְדְּעִ֔ים מִן־הַבָּתִּ֥ים מִן־הַחֲצֵרֹ֖ת וּמִן־הַשָּׂדֹֽת׃ ה' עָשָׂה כִּדְבַר מֹשֶׁה, וְהַצְפַרְדְּעִים מֵתוּ מֵהַבָּתִּים מֵהַחֲצֵרוֹת וּמֵהַשָּׂדוֹת.





* * *



י וַיִּצְבְּר֥וּ אֹתָ֖ם חֳמָרִ֣ם חֳמָרִ֑ם וַתִּבְאַ֖שׁ הָאָֽרֶץ׃ אָסְפוּ אוֹתָם עֲרֵמוֹת עֲרֵמוֹת, וְהָאָרֶץ הִסְרִיחָה.





* * *



יא וַיַּ֣רְא פַּרְעֹ֗ה כִּ֤י הָֽיְתָה֙ הָֽרְוָחָ֔ה וְהַכְבֵּד֙ אֶת־לִבּ֔וֹ וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ פַּרְעֹה רָאָה שִֶׁהָיְתָה הֲקָלָה, וְהוּא הִתְעַקֵּשׁ וְלֹא שָׁמַע לָהֶם – כְּמוֹ שֶׁה' דִּבֵּר.





* * *



ס פָּרָשַׁת מַכַּת הַכִּנִּים





* * *



יב וַיֹּ֣אמֶר יְהֹוָה֮ אֶל־מֹשֶׁה֒ אֱמֹר֙ אֶֽל־אַהֲרֹ֔ן נְטֵ֣ה אֶֽת־מַטְּךָ֔ וְהַ֖ךְ אֶת־עֲפַ֣ר הָאָ֑רֶץ וְהָיָ֥ה לְכִנִּ֖ם בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ ה' אָמַר לְמֹשֶׁה: אֱמוֹר לְאַהֲרֹן – הָנֵף אֶת הַמַּקֵּל שֶׁלְּךָ וְהַכֵּה אֶת עֲפַר הָאָרֶץ, וְהוּא יֵהָפֵךְ לְכִנִּים בְּכָל אֶרֶץ מִצְרַיִם.





* * *



יג וַיַּֽעֲשׂוּ־כֵ֗ן וַיֵּט֩ אַהֲרֹ֨ן אֶת־יָד֤וֹ בְמַטֵּ֙הוּ֙ וַיַּךְ֙ אֶת־עֲפַ֣ר הָאָ֔רֶץ וַתְּהִי֙ הַכִּנָּ֔ם בָּאָדָ֖ם וּבַבְּהֵמָ֑ה כׇּל־עֲפַ֥ר הָאָ֛רֶץ הָיָ֥ה כִנִּ֖ים בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ הֵם עָשׂוּ כָּךְ, אַהֲרֹן הֵנִיף אֶת יָדוֹ עִם מַקְלוֹ, וְהִכָּה אֶת עֲפַר הָאָרֶץ, וְהָיְתָה מַכַּת הַכִּנִּים בָּאָדָם וּבַבְּהֵמָה, כָּל עֲפַר הָאָרֶץ הָפַךְ לְכִנִּים בְּכָל אֶרֶץ מִצְרַיִם.





* * *



יד וַיַּעֲשׂוּ־כֵ֨ן הַחַרְטֻמִּ֧ים בְּלָטֵיהֶ֛ם לְהוֹצִ֥יא אֶת־הַכִּנִּ֖ים וְלֹ֣א יָכֹ֑לוּ וַתְּהִי֙ הַכִּנָּ֔ם בָּאָדָ֖ם וּבַבְּהֵמָֽה׃ כָּךְ עָשׂוּ גַּם הַמְּכַשְּׁפִים בִּכְשָׁפֵיהֶם כְּדֵי לַעֲשׂוֹת כִּנִּים – וְלֹא הִצְלִיחוּ, מַכַּת הַכִּנִּים הָיְתָה בָּאָדָם וּבַבְּהֵמָה.





* * *



טו וַיֹּאמְר֤וּ הַֽחַרְטֻמִּם֙ אֶל־פַּרְעֹ֔ה אֶצְבַּ֥ע אֱלֹהִ֖ים הִ֑וא וַיֶּחֱזַ֤ק לֵב־פַּרְעֹה֙ וְלֹֽא־שָׁמַ֣ע אֲלֵהֶ֔ם כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ הַמְּכַשְּׁפִים אָמְרוּ לְפַרְעֹה: אֶצְבַּע אֱלֹהִים הִיא! פַּרְעֹה הִתְעַקֵּשׁ וְלֹא שָׁמַע בְּקוֹלָם – כְּמוֹ שֶׁה' דִּבֵּר.





* * *



ס פָּרָשַׁת מַכַּת הֶעָרוֹב





* * *



טז וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה הַשְׁכֵּ֤ם בַּבֹּ֙קֶר֙ וְהִתְיַצֵּב֙ לִפְנֵ֣י פַרְעֹ֔ה הִנֵּ֖ה יוֹצֵ֣א הַמָּ֑יְמָה וְאָמַרְתָּ֣ אֵלָ֗יו כֹּ֚ה אָמַ֣ר יְהֹוָ֔ה שַׁלַּ֥ח עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ ה' אָמַר לְמֹשֶׁה: הַשְׁכֵּם בַּבּוֹקֶר וְהִתְיַצֵּב לִפְנֵי פַּרְעֹה, הִנֵּה הוּא יוֹצֵא לַמַּיִם, וֶאֱמוֹר לוֹ: כָּךְ אָמַר ה': שַׁחְרֵר אֶת עַמִּי וְיַעַבְדוּ לְפָנַי.





* * *



יז כִּ֣י אִם־אֵינְךָ֮ מְשַׁלֵּ֣חַ אֶת־עַמִּי֒ הִנְנִי֩ מַשְׁלִ֨יחַ בְּךָ֜ וּבַעֲבָדֶ֧יךָ וּֽבְעַמְּךָ֛ וּבְבָתֶּ֖יךָ אֶת־הֶעָרֹ֑ב וּמָ֨לְא֜וּ בָּתֵּ֤י מִצְרַ֙יִם֙ אֶת־הֶ֣עָרֹ֔ב וְגַ֥ם הָאֲדָמָ֖ה אֲשֶׁר־הֵ֥ם עָלֶֽיהָ׃ כִּי אִם אֵינְךָ מְשַׁחְרֵר אֶת עַמִּי, הִנֵּה אֲנִי שׁוֹלֵחַ בְּךָ וּבַעֲבָדֶיךָ וּבְעַמְּךָ וּבְבָתֶּיךָ תַּעֲרוֹבֶת שֶׁל חַיּוֹת רָעוֹת, וּבָתֵּי מִצְרַיִם יִתְמַלְּאוּ בְּתַעֲרוֹבֶת הַחַיּוֹת, וְגַם הָאֲדָמָה שֶׁהֵם עָלֶיהָ תִּתְמַלֵּא.





* * *



יח וְהִפְלֵיתִי֩ בַיּ֨וֹם הַה֜וּא אֶת־אֶ֣רֶץ גֹּ֗שֶׁן אֲשֶׁ֤ר עַמִּי֙ עֹמֵ֣ד עָלֶ֔יהָ לְבִלְתִּ֥י הֱיֽוֹת־שָׁ֖ם עָרֹ֑ב לְמַ֣עַן תֵּדַ֔ע כִּ֛י אֲנִ֥י יְהֹוָ֖ה בְּקֶ֥רֶב הָאָֽרֶץ׃ בְּאוֹתוֹ יוֹם אַבְדִּיל אֶת אֶרֶץ גּוֹשֶׁן שֶׁעַמִּי נִמְצָא עָלֶיהָ שֶׁלֹּא תִּהְיֶה שָׁם תַּעֲרוֹבֶת הַחַיּוֹת, כְּדֵי שֶׁתֵּדַע שֶׁאֲנִי ה' שׁוֹלֵט בְּתוֹךְ הָאָרֶץ.





* * *





ששי


יט וְשַׂמְתִּ֣י פְדֻ֔ת בֵּ֥ין עַמִּ֖י וּבֵ֣ין עַמֶּ֑ךָ לְמָחָ֥ר יִהְיֶ֖ה הָאֹ֥ת הַזֶּֽה׃ וְאֶעֱשֶׂה רֶוַח וְהַצָּלָה בֵּין עַמִּי לְבֵין עַמְּךָ, מָחָר יִהְיֶה הַסִּימָן הַזֶּה.





* * *



כ וַיַּ֤עַשׂ יְהֹוָה֙ כֵּ֔ן וַיָּבֹא֙ עָרֹ֣ב כָּבֵ֔ד בֵּ֥יתָה פַרְעֹ֖ה וּבֵ֣ית עֲבָדָ֑יו וּבְכׇל־אֶ֧רֶץ מִצְרַ֛יִם תִּשָּׁחֵ֥ת הָאָ֖רֶץ מִפְּנֵ֥י הֶעָרֹֽב׃ ה' עָשָׂה כָּךְ, וְתַעֲרוֹבֶת גְּדוֹלָה שֶׁל חַיּוֹת הִגִּיעָה לְבֵית פַּרְעֹה וּלְבֵית עֲבָדָיו, וּבְכָל אֶרֶץ מִצְרַיִם הָאָרֶץ נֶהֶרְסָה בִּגְלַל תַּעֲרוֹבֶת הַחַיּוֹת.





* * *



כא וַיִּקְרָ֣א פַרְעֹ֔ה אֶל־מֹשֶׁ֖ה וּֽלְאַהֲרֹ֑ן וַיֹּ֗אמֶר לְכ֛וּ זִבְח֥וּ לֵאלֹֽהֵיכֶ֖ם בָּאָֽרֶץ׃ פַּרְעֹה קָרָא לְמֹשֶׁה וּלְאַהֲרֹן וְאָמַר: לְכוּ וְתִשְׁחֲטוּ לֵאלֹהֵיכֶם בָּאָרֶץ.





* * *



כב וַיֹּ֣אמֶר מֹשֶׁ֗ה לֹ֤א נָכוֹן֙ לַעֲשׂ֣וֹת כֵּ֔ן כִּ֚י תּוֹעֲבַ֣ת מִצְרַ֔יִם נִזְבַּ֖ח לַיהֹוָ֣ה אֱלֹהֵ֑ינוּ הֵ֣ן נִזְבַּ֞ח אֶת־תּוֹעֲבַ֥ת מִצְרַ֛יִם לְעֵינֵיהֶ֖ם וְלֹ֥א יִסְקְלֻֽנוּ׃ אָמַר מֹשֶׁה: לֹא נָכוֹן לַעֲשׂוֹת כָּךְ, כִּי אֲנַחְנוּ נִשְׁחַט לַה' אֱלֹהֵינוּ כְּבָשִׂים וְהֵם עֲבוֹדָה זָרָה לְמִצְרַיִם, הֲרֵי כְּשֶׁנִּשְׁחַט אֶת הָעֲבוֹדָה הַזָּרָה שֶׁל מִצְרַיִם מוּל עֵינֵיהֶם – וְכִי לֹא יִסְקְלוּ אוֹתָנוּ?!





* * *



כג דֶּ֚רֶךְ שְׁלֹ֣שֶׁת יָמִ֔ים נֵלֵ֖ךְ בַּמִּדְבָּ֑ר וְזָבַ֙חְנוּ֙ לַֽיהֹוָ֣ה אֱלֹהֵ֔ינוּ כַּאֲשֶׁ֖ר יֹאמַ֥ר אֵלֵֽינוּ׃ נֵלֵךְ בַּמִּדְבָּר דֶּרֶךְ שֶׁל שְׁלוֹשָׁה יָמִים, וְנִשְׁחַט לַה' אֱלֹהֵינוּ כְּמוֹ שֶׁיֹּאמַר לָנוּ.





* * *



כד וַיֹּ֣אמֶר פַּרְעֹ֗ה אָנֹכִ֞י אֲשַׁלַּ֤ח אֶתְכֶם֙ וּזְבַחְתֶּ֞ם לַיהֹוָ֤ה אֱלֹֽהֵיכֶם֙ בַּמִּדְבָּ֔ר רַ֛ק הַרְחֵ֥ק לֹא־תַרְחִ֖יקוּ לָלֶ֑כֶת הַעְתִּ֖ירוּ בַּעֲדִֽי׃ פַּרְעֹה אָמַר: אֲנִי אֲשַׁחְרֵר אֶתְכֶם וְתִשְׁחֲטוּ לַה' אֱלֹהֵיכֶם בַּמִּדְבָּר, רַק אַל תֵּלְכוּ רָחוֹק; הִתְפַּלְּלוּ עָלַי!





* * *



כה וַיֹּ֣אמֶר מֹשֶׁ֗ה הִנֵּ֨ה אָנֹכִ֜י יוֹצֵ֤א מֵֽעִמָּךְ֙ וְהַעְתַּרְתִּ֣י אֶל־יְהֹוָ֔ה וְסָ֣ר הֶעָרֹ֗ב מִפַּרְעֹ֛ה מֵעֲבָדָ֥יו וּמֵעַמּ֖וֹ מָחָ֑ר רַ֗ק אַל־יֹסֵ֤ף פַּרְעֹה֙ הָתֵ֔ל לְבִלְתִּי֙ שַׁלַּ֣ח אֶת־הָעָ֔ם לִזְבֹּ֖חַ לַֽיהֹוָֽה׃ מֹשֶׁה אָמַר: הִנֵּה אֲנִי יוֹצֵא מִמְּךָ, וְאֶתְפַּלֵּל אֶל ה', וְתַעֲרוֹבֶת הַחַיּוֹת תִּסְתַּלֵּק מָחָר מִפַּרְעֹה וּמֵעֲבָדָיו וּמֵעַמּוֹ; רַק שֶׁפַּרְעֹה לֹא יַמְשִׁיךְ לִצְחוֹק עָלֵינוּ – שֶׁלֹּא לְשַׁחְרֵר אֶת הָעָם לִשְׁחוֹט לַה'.





* * *



כו וַיֵּצֵ֥א מֹשֶׁ֖ה מֵעִ֣ם פַּרְעֹ֑ה וַיֶּעְתַּ֖ר אֶל־יְהֹוָֽה׃ מֹשֶׁה יָצָא מִלִּפְנֵי פַּרְעֹה וְהִתְפַּלֵּל אֶל ה'.





* * *



כז וַיַּ֤עַשׂ יְהֹוָה֙ כִּדְבַ֣ר מֹשֶׁ֔ה וַיָּ֙סַר֙ הֶעָרֹ֔ב מִפַּרְעֹ֖ה מֵעֲבָדָ֣יו וּמֵעַמּ֑וֹ לֹ֥א נִשְׁאַ֖ר אֶחָֽד׃ ה' עָשָׂה כְּדִבְרֵי מֹשֶׁה וְתַעֲרוֹבֶת הַחַיּוֹת הִסְתַּלְּקָה מִפַּרְעֹה וּמֵעֲבָדָיו וּמֵעַמּוֹ, לֹא נִשְׁאַר אַף אֶחָד.





* * *



כח וַיַּכְבֵּ֤ד פַּרְעֹה֙ אֶת־לִבּ֔וֹ גַּ֖ם בַּפַּ֣עַם הַזֹּ֑את וְלֹ֥א שִׁלַּ֖ח אֶת־הָעָֽם׃ פַּרְעֹה הִתְעַקֵּשׁ גַּם בַּפַּעַם הַזֹּאת וְלֹא שִׁחְרֵר אֶת הָעָם.





* * *



פ פָּרָשַׁת מַכַּת הַדֶּבֶר





* * *



ט א וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה וְדִבַּרְתָּ֣ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהֹוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ ה' אָמַר לְמֹשֶׁה בֹּא לְפַרְעֹה וְדַבֵּר אֵלָיו: כָּךְ אָמַר ה' אֱלֹהֵי הָעִבְרִים: שַׁחְרֵר אֶת עַמִּי וְהֵם יַעַבְדוּ לְפָנַי.





* * *



ב כִּ֛י אִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֑חַ וְעוֹדְךָ֖ מַחֲזִ֥יק בָּֽם׃ כִּי אִם אַתָּה מְסָרֵב לְשַׁחְרֵר וְאַתָּה עֲדַיִן מַחֲזִיק בָּהֶם.





* * *



ג הִנֵּ֨ה יַד־יְהֹוָ֜ה הוֹיָ֗ה בְּמִקְנְךָ֙ אֲשֶׁ֣ר בַּשָּׂדֶ֔ה בַּסּוּסִ֤ים בַּֽחֲמֹרִים֙ בַּגְּמַלִּ֔ים בַּבָּקָ֖ר וּבַצֹּ֑אן דֶּ֖בֶר כָּבֵ֥ד מְאֹֽד׃ הַמַּכָּה שֶׁל ה' תִּהְיֶה עַל בֶּהֱמוֹתֶיךָ שֶׁבַּשָּׂדֶה – הַסּוּסִים, הַחֲמוֹרִים, הַגְּמַלִּים, הַבָּקָר, וְהַצֹּאן – בְּדֶבֶר כָּבֵד מְאוֹד.





* * *



ד וְהִפְלָ֣ה יְהֹוָ֔ה בֵּ֚ין מִקְנֵ֣ה יִשְׂרָאֵ֔ל וּבֵ֖ין מִקְנֵ֣ה מִצְרָ֑יִם וְלֹ֥א יָמ֛וּת מִכׇּל־לִבְנֵ֥י יִשְׂרָאֵ֖ל דָּבָֽר׃ ה' יַבְדִּיל בֵּין בְּהֵמוֹת יִשְׂרָאֵל וּבֵין בְּהֵמוֹת מִצְרַיִם, וּמִכָּל הַשַּׁיָּךְ לִבְנֵי יִשְׂרָאֵל – שׁוּם דָּבָר לֹא יָמוּת.





* * *



ה וַיָּ֥שֶׂם יְהֹוָ֖ה מוֹעֵ֣ד לֵאמֹ֑ר מָחָ֗ר יַעֲשֶׂ֧ה יְהֹוָ֛ה הַדָּבָ֥ר הַזֶּ֖ה בָּאָֽרֶץ׃ ה' קָבַע זְמָן וְאָמַר: מָחָר ה' יַעֲשֶׂה בָּאָרֶץ אֶת הַדָּבָר הַזֶּה.





* * *



ו וַיַּ֨עַשׂ יְהֹוָ֜ה אֶת־הַדָּבָ֤ר הַזֶּה֙ מִֽמׇּחֳרָ֔ת וַיָּ֕מׇת כֹּ֖ל מִקְנֵ֣ה מִצְרָ֑יִם וּמִמִּקְנֵ֥ה בְנֵֽי־יִשְׂרָאֵ֖ל לֹא־מֵ֥ת אֶחָֽד׃ לְמָחֳרָת ה' עָשָׂה אֶת הַדָּבָר הַזֶּה וְכָל הַבְּהֵמוֹת שֶׁל מִצְרַיִם מֵתוּ, וּמֵהַבְּהֵמוֹת שֶׁל בְּנֵי יִשְׂרָאֵל לֹא מֵת אַף אֶחָד.





* * *



ז וַיִּשְׁלַ֣ח פַּרְעֹ֔ה וְהִנֵּ֗ה לֹא־מֵ֛ת מִמִּקְנֵ֥ה יִשְׂרָאֵ֖ל עַד־אֶחָ֑ד וַיִּכְבַּד֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שִׁלַּ֖ח אֶת־הָעָֽם׃ פַּרְעֹה שָׁלַח לִבְדּוֹק וְהִנֵּה לֹא מֵת מֵהַבְּהֵמוֹת שֶׁל יִשְׂרָאֵל אַף אֶחָד, פַּרְעֹה הִתְעַקֵּשׁ, וְלֹא שִׁחְרֵר אֶת הָעָם.





* * *



פ פָּרָשַׁת מַכַּת הַשְּׁחִין





* * *



ח וַיֹּ֣אמֶר יְהֹוָה֮ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹן֒ קְח֤וּ לָכֶם֙ מְלֹ֣א חׇפְנֵיכֶ֔ם פִּ֖יחַ כִּבְשָׁ֑ן וּזְרָק֥וֹ מֹשֶׁ֛ה הַשָּׁמַ֖יְמָה לְעֵינֵ֥י פַרְעֹֽה׃ ה' אָמַר לְמֹשֶׁה וּלְאַהֲרֹן: קְחוּ לָכֶם מְלֹא חָפְנֵיכֶם פִּיחַ כִּבְשָׁן וּמֹשֶׁה יִזְרוֹק אוֹתוֹ לַשָּׁמַיִם לְעֵינֵי פַּרְעֹה.





* * *



ט וְהָיָ֣ה לְאָבָ֔ק עַ֖ל כׇּל־אֶ֣רֶץ מִצְרָ֑יִם וְהָיָ֨ה עַל־הָאָדָ֜ם וְעַל־הַבְּהֵמָ֗ה לִשְׁחִ֥ין פֹּרֵ֛חַ אֲבַעְבֻּעֹ֖ת בְּכׇל־אֶ֥רֶץ מִצְרָֽיִם׃ הוּא יֵהָפֵךְ לְאָבָק וְיִתְפַּזֵּר עַל כָּל אֶרֶץ מִצְרַיִם, וְהוּא יֵהָפֵךְ לִפְרִיחָה שֶׁל שְׁחִין אֲבַעְבּוּעוֹת עַל הָאָדָם וְעַל הַבְּהֵמָה בְּכָל אֶרֶץ מִצְרַיִם.





* * *



י וַיִּקְח֞וּ אֶת־פִּ֣יחַ הַכִּבְשָׁ֗ן וַיַּֽעַמְדוּ֙ לִפְנֵ֣י פַרְעֹ֔ה וַיִּזְרֹ֥ק אֹת֛וֹ מֹשֶׁ֖ה הַשָּׁמָ֑יְמָה וַיְהִ֗י שְׁחִין֙ אֲבַעְבֻּעֹ֔ת פֹּרֵ֕חַ בָּאָדָ֖ם וּבַבְּהֵמָֽה׃ הֵם לָקְחוּ אֶת פִּיחַ הַכִּבְשָׁן וְעָמְדוּ לִפְנֵי פַּרְעֹה וּמֹשֶׁה זָרַק אוֹתוֹ לַשָּׁמַיִם, וְהָיְתָה פְּרִיחָה שֶׁל שְׁחִין אֲבַעְבּוּעוֹת בִּבְנֵי הָאָדָם וּבַבְּהֵמוֹת.





* * *



יא וְלֹֽא־יָכְל֣וּ הַֽחַרְטֻמִּ֗ים לַעֲמֹ֛ד לִפְנֵ֥י מֹשֶׁ֖ה מִפְּנֵ֣י הַשְּׁחִ֑ין כִּֽי־הָיָ֣ה הַשְּׁחִ֔ין בַּֽחַרְטֻמִּ֖ם וּבְכׇל־מִצְרָֽיִם׃ וְהַמְּכַשְּׁפִים לֹא יָכְלוּ לַעֲמוֹד לִפְנֵי מֹשֶׁה בִּגְלַל הַשְּׁחִין, כִּי הַשְּׁחִין הָיָה גַּם בַּמְּכַשְּׁפִים וּבְכָל הַמִּצְרִים.





* * *



יב וַיְחַזֵּ֤ק יְהֹוָה֙ אֶת־לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה אֶל־מֹשֶֽׁה׃ ה' הִקְשִׁיחַ אֶת לֵב פַּרְעֹה וְלֹא שָׁמַע אֲלֵהֶם, כְּמוֹ שֶׁה' דִּבֵּר אֶל מֹשֶׁה.





* * *



ס פָּרָשַׁת מַכַּת הַבָּרָד





* * *



יג וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה הַשְׁכֵּ֣ם בַּבֹּ֔קֶר וְהִתְיַצֵּ֖ב לִפְנֵ֣י פַרְעֹ֑ה וְאָמַרְתָּ֣ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהֹוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ ה' אָמַר לְמֹשֶׁה: הַשְׁכֵּם בַּבּוֹקֶר וַעֲמוֹד לִפְנֵי פַּרְעֹה וֶאֱמוֹר לוֹ: כָּךְ אָמַר ה' אֱלֹהֵי הָעִבְרִים – שַׁחְרֵר אֶת עַמִּי וְהֵם יַעַבְדוּ לְפָנַי.





* * *



יד כִּ֣י ׀ בַּפַּ֣עַם הַזֹּ֗את אֲנִ֨י שֹׁלֵ֜חַ אֶת־כׇּל־מַגֵּפֹתַי֙ אֶֽל־לִבְּךָ֔ וּבַעֲבָדֶ֖יךָ וּבְעַמֶּ֑ךָ בַּעֲב֣וּר תֵּדַ֔ע כִּ֛י אֵ֥ין כָּמֹ֖נִי בְּכׇל־הָאָֽרֶץ׃ כִּי בַּפַּעַם הַזֹּאת אֲנִי שׁוֹלֵחַ מַכָּה שֶׁכּוֹלֶלֶת אֶת כָּל סוּגֵי מַכּוֹתַי אֶל לִבְּךָ וּבַעֲבָדֶיךָ וּבְעַמְּךָ, כְּדֵי שֶׁתֵּדַע שֶׁאֵין כָּמוֹנִי בְּכָל הָאָרֶץ.





* * *



טו כִּ֤י עַתָּה֙ שָׁלַ֣חְתִּי אֶת־יָדִ֔י וָאַ֥ךְ אוֹתְךָ֛ וְאֶֽת־עַמְּךָ֖ בַּדָּ֑בֶר וַתִּכָּחֵ֖ד מִן־הָאָֽרֶץ׃ הֲרֵי עַכְשָׁיו אֲנִי יָכוֹל לְהוֹשִׁיט אֶת יָדִי וּלְהַכּוֹת אוֹתְךָ וְאֶת עַמְּךָ בַּדֶּבֶר וְתֵעָלֵם מֵהָאָרֶץ.





* * *



טז וְאוּלָ֗ם בַּעֲב֥וּר זֹאת֙ הֶעֱמַדְתִּ֔יךָ בַּעֲב֖וּר הַרְאֹתְךָ֣ אֶת־כֹּחִ֑י וּלְמַ֛עַן סַפֵּ֥ר שְׁמִ֖י בְּכׇל־הָאָֽרֶץ׃ וְאוּלָם בִּשְׁבִיל זֶה הִשְׁאַרְתִּי אוֹתְךָ בַּחַיִּים – כְּדֵי לְהַרְאוֹת לְךָ אֶת כּוֹחִי, וּכְדֵי לְסַפֵּר אֶת שְׁמִי בְּכָל הָאָרֶץ.





* * *





שביעי


יז עוֹדְךָ֖ מִסְתּוֹלֵ֣ל בְּעַמִּ֑י לְבִלְתִּ֖י שַׁלְּחָֽם׃ אַתָּה עֲדַיִן מְשַׁעְבֵּד אֶת עַמִּי וְאֵינְךָ מְשַׁחְרְרָם,





* * *



יח הִנְנִ֤י מַמְטִיר֙ כָּעֵ֣ת מָחָ֔ר בָּרָ֖ד כָּבֵ֣ד מְאֹ֑ד אֲשֶׁ֨ר לֹא־הָיָ֤ה כָמֹ֙הוּ֙ בְּמִצְרַ֔יִם לְמִן־הַיּ֥וֹם הִוָּסְדָ֖ה וְעַד־עָֽתָּה׃ הִנֵּה אֲנִי מַמְטִיר בְּשָׁעָה זוֹ מָחָר בָּרָד חָזָק מְאוֹד שֶׁלֹּא הָיָה כָּמוֹהוּ בְּמִצְרַיִם מֵהַיּוֹם שֶׁנּוֹסְדָה וְעַד עַכְשָׁיו.





* * *



יט וְעַתָּ֗ה שְׁלַ֤ח הָעֵז֙ אֶֽת־מִקְנְךָ֔ וְאֵ֛ת כׇּל־אֲשֶׁ֥ר לְךָ֖ בַּשָּׂדֶ֑ה כׇּל־הָאָדָ֨ם וְהַבְּהֵמָ֜ה אֲשֶֽׁר־יִמָּצֵ֣א בַשָּׂדֶ֗ה וְלֹ֤א יֵֽאָסֵף֙ הַבַּ֔יְתָה וְיָרַ֧ד עֲלֵהֶ֛ם הַבָּרָ֖ד וָמֵֽתוּ׃ וְעַכְשָׁיו שְׁלַח וְהַכְנֵס אֶת בֶּהֱמוֹתֶיךָ וְאֶת כָּל מַה שֶּׁשַּׁיָּךְ לְךָ שֶׁנִּמְצָא בַּשָּׂדֶה, כָּל הָאָדָם וְהַבְּהֵמָה שֶׁיִּהְיוּ בַּשָּׂדֶה וְלֹא יִכָּנְסוּ הַבַּיְתָה – הַבָּרָד יֵרֶד עֲלֵהֶם וְהֵם יָמוּתוּ.





* * *



כ הַיָּרֵא֙ אֶת־דְּבַ֣ר יְהֹוָ֔ה מֵֽעַבְדֵ֖י פַּרְעֹ֑ה הֵנִ֛יס אֶת־עֲבָדָ֥יו וְאֶת־מִקְנֵ֖הוּ אֶל־הַבָּתִּֽים׃ מִי שֶׁפָּחַד מִדְּבַר ה' מֵעַבְדֵי פַּרְעֹה, הִבְרִיחַ אֶת עֲבָדָיו וְאֶת בֶּהֱמוֹתָיו אֶל הַבָּתִּים.





* * *



כא וַאֲשֶׁ֥ר לֹא־שָׂ֛ם לִבּ֖וֹ אֶל־דְּבַ֣ר יְהֹוָ֑ה וַֽיַּעֲזֹ֛ב אֶת־עֲבָדָ֥יו וְאֶת־מִקְנֵ֖הוּ בַּשָּׂדֶֽה׃ וּמִי שֶׁלֹּא שָׂם לִבּוֹ אֶל דְּבַר ה' הִשְׁאִיר אֶת עֲבָדָיו וְאֶת בֶּהֱמוֹתָיו בַּשָּׂדֶה.





* * *



פ





* * *



כב וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֤ה אֶת־יָֽדְךָ֙ עַל־הַשָּׁמַ֔יִם וִיהִ֥י בָרָ֖ד בְּכׇל־אֶ֣רֶץ מִצְרָ֑יִם עַל־הָאָדָ֣ם וְעַל־הַבְּהֵמָ֗ה וְעַ֛ל כׇּל־עֵ֥שֶׂב הַשָּׂדֶ֖ה בְּאֶ֥רֶץ מִצְרָֽיִם׃ ה' אָמַר לְמֹשֶׁה: הָנֵף אֶת יָדְךָ אֶל הַשָּׁמַיִם וְיִהְיֶה בָּרָד בְּכָל אֶרֶץ מִצְרַיִם, עַל הָאָדָם וְעַל הַבְּהֵמָה וְעַל כָּל עֵשֶׂב הַשָּׂדֶה בְּאֶרֶץ מִצְרַיִם.





* * *



כג וַיֵּ֨ט מֹשֶׁ֣ה אֶת־מַטֵּ֘הוּ֮ עַל־הַשָּׁמַ֒יִם֒ וַֽיהֹוָ֗ה נָתַ֤ן קֹלֹת֙ וּבָרָ֔ד וַתִּ֥הֲלַךְ אֵ֖שׁ אָ֑רְצָה וַיַּמְטֵ֧ר יְהֹוָ֛ה בָּרָ֖ד עַל־אֶ֥רֶץ מִצְרָֽיִם׃ מֹשֶׁה הֵנִיף אֶת מַקְלוֹ אֶל הַשָּׁמַיִם, וַה' עָשָׂה קוֹלוֹת וּבָרָד, וְאֵשׁ הָלְכָה לָאָרֶץ, וַה' הִמְטִיר בָּרָד עַל אֶרֶץ מִצְרַיִם.





* * *



כד וַיְהִ֣י בָרָ֔ד וְאֵ֕שׁ מִתְלַקַּ֖חַת בְּת֣וֹךְ הַבָּרָ֑ד כָּבֵ֣ד מְאֹ֔ד אֲ֠שֶׁ֠ר לֹֽא־הָיָ֤ה כָמֹ֙הוּ֙ בְּכׇל־אֶ֣רֶץ מִצְרַ֔יִם מֵאָ֖ז הָיְתָ֥ה לְגֽוֹי׃ הָיָה בָּרָד, וְאֵשׁ נִדְלֶקֶת בְּתוֹךְ הַבָּרָד, בָּרָד כָּבֵד מְאוֹד שֶׁלֹּא הָיָה כָּמוֹהוּ בְּכָל אֶרֶץ מִצְרַיִם מֵאָז שֶׁהָיְתָה עָם.





* * *



כה וַיַּ֨ךְ הַבָּרָ֜ד בְּכׇל־אֶ֣רֶץ מִצְרַ֗יִם אֵ֚ת כׇּל־אֲשֶׁ֣ר בַּשָּׂדֶ֔ה מֵאָדָ֖ם וְעַד־בְּהֵמָ֑ה וְאֵ֨ת כׇּל־עֵ֤שֶׂב הַשָּׂדֶה֙ הִכָּ֣ה הַבָּרָ֔ד וְאֶת־כׇּל־עֵ֥ץ הַשָּׂדֶ֖ה שִׁבֵּֽר׃ הַבָּרָד הִכָּה בְּכָל אֶרֶץ מִצְרַיִם אֶת כָּל מַה שֶּׁבַּשָּׂדֶה מֵאָדָם וְעַד בְּהֵמָה, וְאֶת כָּל הָעֲשָׂבִים בַּשָּׂדֶה הַבָּרָד הִכָּה, וְשָׁבַר אֶת כָּל הָעֵצִים בַּשָּׂדֶה.





* * *



כו רַ֚ק בְּאֶ֣רֶץ גֹּ֔שֶׁן אֲשֶׁר־שָׁ֖ם בְּנֵ֣י יִשְׂרָאֵ֑ל לֹ֥א הָיָ֖ה בָּרָֽד׃ רַק בְּאֶרֶץ גּוֹשֶׁן שֶׁשָּׁם בְּנֵי יִשְׂרָאֵל לֹא הָיָה בָּרָד.





* * *



כז וַיִּשְׁלַ֣ח פַּרְעֹ֗ה וַיִּקְרָא֙ לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֔ן וַיֹּ֥אמֶר אֲלֵהֶ֖ם חָטָ֣אתִי הַפָּ֑עַם יְהֹוָה֙ הַצַּדִּ֔יק וַאֲנִ֥י וְעַמִּ֖י הָרְשָׁעִֽים׃ פַּרְעֹה שָׁלַח וְקָרָא לְמֹשֶׁה וּלְאַהֲרֹן וְאָמַר לָהֶם: הַפַּעַם אֲנִי מוֹדֶה שֶׁחָטָאתִי, ה' הוּא הַצַּדִּיק וַאֲנִי וְעַמִּי הָרְשָׁעִים.





* * *



כח הַעְתִּ֙ירוּ֙ אֶל־יְהֹוָ֔ה וְרַ֕ב מִֽהְיֹ֛ת קֹלֹ֥ת אֱלֹהִ֖ים וּבָרָ֑ד וַאֲשַׁלְּחָ֣ה אֶתְכֶ֔ם וְלֹ֥א תֹסִפ֖וּן לַעֲמֹֽד׃ הִתְפַּלְּלוּ אֶל ה' שֶׁהָיוּ מַסְפִּיק קוֹלוֹת אֱלֹהִים וּבָרָד, וַאֲשַׁחְרֵר אֶתְכֶם וְלֹא תַּמְשִׁיכוּ לְהִתְעַכֵּב.





* * *



כט וַיֹּ֤אמֶר אֵלָיו֙ מֹשֶׁ֔ה כְּצֵאתִי֙ אֶת־הָעִ֔יר אֶפְרֹ֥שׂ אֶת־כַּפַּ֖י אֶל־יְהֹוָ֑ה הַקֹּל֣וֹת יֶחְדָּל֗וּן וְהַבָּרָד֙ לֹ֣א יִֽהְיֶה־ע֔וֹד לְמַ֣עַן תֵּדַ֔ע כִּ֥י לַיהֹוָ֖ה הָאָֽרֶץ׃ מֹשֶׁה אָמַר אֵלָיו: כְּשֶׁאֵצֵא מֵהָעִיר אָרִים אֶת יָדַי בִּתְפִלָּה אֶל ה', הַקּוֹלוֹת יִפָּסְקוּ וְלֹא יִהְיֶה עוֹד הַבָּרָד כְּדֵי שֶׁתֵּדַע שֶׁהָאָרֶץ שַׁיֶּכֶת לַה'.





* * *



ל וְאַתָּ֖ה וַעֲבָדֶ֑יךָ יָדַ֕עְתִּי כִּ֚י טֶ֣רֶם תִּֽירְא֔וּן מִפְּנֵ֖י יְהֹוָ֥ה אֱלֹהִֽים׃ וַאֲנִי יוֹדֵעַ שֶׁאַתָּה וַעֲבָדֶיךָ עֲדַיִן אֵינְכֶם מְפַחֲדִים מִפְּנֵי ה' אֱלֹהִים.





* * *



לא וְהַפִּשְׁתָּ֥ה וְהַשְּׂעֹרָ֖ה נֻכָּ֑תָה כִּ֤י הַשְּׂעֹרָה֙ אָבִ֔יב וְהַפִּשְׁתָּ֖ה גִּבְעֹֽל׃ הַפִּשְׁתָּן וְהַשְּׂעוֹרָה נֶהֶרְסוּ בַּבָּרָד, כִּי הַשְּׂעוֹרָה כְּבָר הִתְחִילָה לִצְמוֹחַ וְלַפִּשְׁתָּן יֵשׁ גִּבְעוֹלִים קָשִׁים וּשְׁבִירִים.





* * *



לב וְהַחִטָּ֥ה וְהַכֻּסֶּ֖מֶת לֹ֣א נֻכּ֑וּ כִּ֥י אֲפִילֹ֖ת הֵֽנָּה׃ וְהַחִטָּה וְהַכֻּסֶּמֶת לֹא נֶהֶרְסוּ כִּי הֵן גְּדֵלוֹת מְאוּחָר.





* * *





מפטיר


לג וַיֵּצֵ֨א מֹשֶׁ֜ה מֵעִ֤ם פַּרְעֹה֙ אֶת־הָעִ֔יר וַיִּפְרֹ֥שׂ כַּפָּ֖יו אֶל־יְהֹוָ֑ה וַֽיַּחְדְּל֤וּ הַקֹּלוֹת֙ וְהַבָּרָ֔ד וּמָטָ֖ר לֹא־נִתַּ֥ךְ אָֽרְצָה׃ מֹשֶׁה יָצָא מִלִּפְנֵי פַּרְעֹה וּמֵהָעִיר וּפָרַשׂ אֶת יָדָיו אֶל ה' בִּתְפִלָּה, הַקּוֹלוֹת וְהַבָּרָד נִפְסְקוּ וְלֹא נִשְׁפַּךְ עוֹד גֶּשֶׁם לָאָרֶץ.





* * *



לד וַיַּ֣רְא פַּרְעֹ֗ה כִּֽי־חָדַ֨ל הַמָּטָ֧ר וְהַבָּרָ֛ד וְהַקֹּלֹ֖ת וַיֹּ֣סֶף לַחֲטֹ֑א וַיַּכְבֵּ֥ד לִבּ֖וֹ ה֥וּא וַעֲבָדָֽיו׃ פַּרְעֹה רָאָה שֶׁהַגֶּשֶׁם וְהַבָּרָד וְהַקּוֹלוֹת נִפְסְקוּ וְהוּא הוֹסִיף לַחֲטוֹא, וְהוּא וַעֲבָדָיו הִתְעַקְּשׁוּ.





* * *



לה וַֽיֶּחֱזַק֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שִׁלַּ֖ח אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר דִּבֶּ֥ר יְהֹוָ֖ה בְּיַד־מֹשֶֽׁה׃ פַּרְעֹה הִתְעַקֵּשׁ וְלֹא שִׁחְרֵר אֶת בְּנֵי יִשְׂרָאֵל, כְּמוֹ שֶׁה' אָמַר עַל יְדֵי מֹשֶׁה.
