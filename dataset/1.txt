קְרִיאַת הַתּוֹרָה וְהַהַפְטָרָה · מִקְרָא וְתַרְגּוּם · שְׁנַֽיִם מִקְרָא וְאֶחָד תַּרְגּוּם

מִקְרָא וְתַרְגּוּם וְרַשִׁ"י (בהכנה) · מִקְרָא מְבוֹאָר





ראשון


פָּרָשַׁת בְּרִיאַת הָעוֹלָם





* * *



א א בְּרֵאשִׁ֖ית בָּרָ֣א אֱלֹהִ֑ים אֵ֥ת הַשָּׁמַ֖יִם וְאֵ֥ת הָאָֽרֶץ׃ בַּתְּחִלָּה, כְּשֶׁאֱלֹהִים בָּרָא אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ.





* * *



ב וְהָאָ֗רֶץ הָיְתָ֥ה תֹ֙הוּ֙ וָבֹ֔הוּ וְחֹ֖שֶׁךְ עַל־פְּנֵ֣י תְה֑וֹם וְר֣וּחַ אֱלֹהִ֔ים מְרַחֶ֖פֶת עַל־פְּנֵ֥י הַמָּֽיִם׃ לִפְנֵי כֵן הָאָרֶץ הָיְתָה חַסְרַת מַמָּשׁוּת וְרֵיקָה, וְהָיָה חוֹשֶׁךְ מֵעַל הַתְּהוֹם, וְרוּחַ הָאֱלֹהִים נָשְׁבָה מֵעַל הַמַּיִם.





* * *



ג וַיֹּ֥אמֶר אֱלֹהִ֖ים יְהִ֣י א֑וֹר וַֽיְהִי־אֽוֹר׃ אֱלֹהִים אָמַר: יִהְיֶה אוֹר! וְנִהְיָה אוֹר.





* * *



ד וַיַּ֧רְא אֱלֹהִ֛ים אֶת־הָא֖וֹר כִּי־ט֑וֹב וַיַּבְדֵּ֣ל אֱלֹהִ֔ים בֵּ֥ין הָא֖וֹר וּבֵ֥ין הַחֹֽשֶׁךְ׃ אֱלֹהִים רָאָה אֶת הָאוֹר וְקָבַע שֶׁהוּא טוֹב, וֶאֱלֹהִים הִבְדִּיל בֵּין הָאוֹר וּבֵין הַחוֹשֶׁךְ.





* * *



ה וַיִּקְרָ֨א אֱלֹהִ֤ים ׀ לָאוֹר֙ י֔וֹם וְלַחֹ֖שֶׁךְ קָ֣רָא לָ֑יְלָה וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם אֶחָֽד׃ אֱלֹהִים קָרָא לָאוֹר יוֹם, וְלַחוֹשֶׁךְ הוּא קָרָא לַיְלָה, וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל יוֹם אֶחָד.





* * *



פ





* * *



ו וַיֹּ֣אמֶר אֱלֹהִ֔ים יְהִ֥י רָקִ֖יעַ בְּת֣וֹךְ הַמָּ֑יִם וִיהִ֣י מַבְדִּ֔יל בֵּ֥ין מַ֖יִם לָמָֽיִם׃ אֱלֹהִים אָמַר: יִהְיֶה רָקִיעַ בְּאֶמְצַע הַמַּיִם שֶׁיִּהְיֶה מַבְדִּיל בֵּין מַיִם לְמַיִם.





* * *



ז וַיַּ֣עַשׂ אֱלֹהִים֮ אֶת־הָרָקִ֒יעַ֒ וַיַּבְדֵּ֗ל בֵּ֤ין הַמַּ֙יִם֙ אֲשֶׁר֙ מִתַּ֣חַת לָרָקִ֔יעַ וּבֵ֣ין הַמַּ֔יִם אֲשֶׁ֖ר מֵעַ֣ל לָרָקִ֑יעַ וַֽיְהִי־כֵֽן׃ אֱלֹהִים עָשָׂה אֶת הָרָקִיעַ, וְהִבְדִּיל בֵּין הַמַּיִם שֶׁמִּתַּחַת לָרָקִיעַ וּבֵין הַמַּיִם שֶׁמֵּעַל לָרָקִיעַ, וְכָךְ נִהְיָה.





* * *



ח וַיִּקְרָ֧א אֱלֹהִ֛ים לָֽרָקִ֖יעַ שָׁמָ֑יִם וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם שֵׁנִֽי׃ אֱלֹהִים קָרָא לָרָקִיעַ שָׁמַיִם, וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל יוֹם שֵׁנִי.





* * *



פ





* * *



ט וַיֹּ֣אמֶר אֱלֹהִ֗ים יִקָּו֨וּ הַמַּ֜יִם מִתַּ֤חַת הַשָּׁמַ֙יִם֙ אֶל־מָק֣וֹם אֶחָ֔ד וְתֵרָאֶ֖ה הַיַּבָּשָׁ֑ה וַֽיְהִי־כֵֽן׃ אֱלֹהִים אָמַר: הַמַּיִם שֶׁמִּתַּחַת לַשָּׁמַיִם יִתְכַּנְּסוּ לְמָקוֹם אֶחָד, וְתֵרָאֶה הַיַּבָּשָׁה, וְכָךְ נִהְיָה.





* * *



י וַיִּקְרָ֨א אֱלֹהִ֤ים ׀ לַיַּבָּשָׁה֙ אֶ֔רֶץ וּלְמִקְוֵ֥ה הַמַּ֖יִם קָרָ֣א יַמִּ֑ים וַיַּ֥רְא אֱלֹהִ֖ים כִּי־טֽוֹב׃ אֱלֹהִים קָרָא לַיַּבָּשָׁה אֶרֶץ וְלִמְקוֹם כִּנּוּס הַמַּיִם קָרָא יַמִּים, אֱלֹהִים רָאָה וְקָבַע שֶׁזֶּה טוֹב.





* * *



יא וַיֹּ֣אמֶר אֱלֹהִ֗ים תַּֽדְשֵׁ֤א הָאָ֙רֶץ֙ דֶּ֗שֶׁא עֵ֚שֶׂב מַזְרִ֣יעַ זֶ֔רַע עֵ֣ץ פְּרִ֞י עֹ֤שֶׂה פְּרִי֙ לְמִינ֔וֹ אֲשֶׁ֥ר זַרְעוֹ־ב֖וֹ עַל־הָאָ֑רֶץ וַֽיְהִי־כֵֽן׃ אֱלֹהִים אָמַר: הָאָרֶץ תַּצְמִיחַ דֶּשֶׁא שֶׁל עֲשָׂבִים שֶׁמַּצְמִיחִים זְרָעִים, וְעֵץ פְּרִי שֶׁעוֹשֶׂה פְּרִי שֶׁמַּצְמִיחַ אֶת מִינוֹ – שֶׁיֵּשׁ בּוֹ זְרָעִים לִזְרוֹעַ בָּאָרֶץ, וְכָךְ נִהְיָה.





* * *



יב וַתּוֹצֵ֨א הָאָ֜רֶץ דֶּ֠שֶׁא עֵ֣שֶׂב מַזְרִ֤יעַ זֶ֙רַע֙ לְמִינֵ֔הוּ וְעֵ֧ץ עֹֽשֶׂה־פְּרִ֛י אֲשֶׁ֥ר זַרְעוֹ־ב֖וֹ לְמִינֵ֑הוּ וַיַּ֥רְא אֱלֹהִ֖ים כִּי־טֽוֹב׃ הָאָרֶץ הוֹצִיאָה דֶּשֶׁא שֶׁל עֲשָׂבִים שֶׁמַּצְמִיחִים זְרָעִים שֶׁמֵּהֶם גְּדֵלִים עֲשָׂבִים מִמִּין אוֹתוֹ הָעֵשֶׂב, וְעֵץ עוֹשֶׂה פְּרִי שֶׁבּוֹ זְרָעִים שֶׁמַּצְמִיחִים אֶת מִינוֹ, אֱלֹהִים רָאָה וְקָבַע שֶׁזֶּה טוֹב.





* * *



יג וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם שְׁלִישִֽׁי׃ וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל יוֹם שְׁלִישִׁי.





* * *



פ





* * *



יד וַיֹּ֣אמֶר אֱלֹהִ֗ים יְהִ֤י מְאֹרֹת֙ בִּרְקִ֣יעַ הַשָּׁמַ֔יִם לְהַבְדִּ֕יל בֵּ֥ין הַיּ֖וֹם וּבֵ֣ין הַלָּ֑יְלָה וְהָי֤וּ לְאֹתֹת֙ וּלְמ֣וֹעֲדִ֔ים וּלְיָמִ֖ים וְשָׁנִֽים׃ אֱלֹהִים אָמַר: יִהְיוּ מְאוֹרוֹת בָּרָקִיעַ שֶׁל הַשָּׁמַיִם לְהַבְדִּיל בֵּין הַיּוֹם וּבֵין הַלַּיְלָה, שֶׁיִּהְיוּ לְסִמָּנִים וּלְחִשּׁוּב זְמַנֵּי הַמּוֹעֲדִים וְהַיָּמִים וְהַשָּׁנִים.





* * *



טו וְהָי֤וּ לִמְאוֹרֹת֙ בִּרְקִ֣יעַ הַשָּׁמַ֔יִם לְהָאִ֖יר עַל־הָאָ֑רֶץ וַֽיְהִי־כֵֽן׃ וְיִהְיוּ לִמְאוֹרוֹת בָּרָקִיעַ שֶׁל הַשָּׁמַיִם לְהָאִיר עַל הָאָרֶץ, וְכָךְ נִהְיָה.





* * *



טז וַיַּ֣עַשׂ אֱלֹהִ֔ים אֶת־שְׁנֵ֥י הַמְּאֹרֹ֖ת הַגְּדֹלִ֑ים אֶת־הַמָּא֤וֹר הַגָּדֹל֙ לְמֶמְשֶׁ֣לֶת הַיּ֔וֹם וְאֶת־הַמָּא֤וֹר הַקָּטֹן֙ לְמֶמְשֶׁ֣לֶת הַלַּ֔יְלָה וְאֵ֖ת הַכּוֹכָבִֽים׃ אֱלֹהִים עָשָׂה אֶת שְׁנֵי הַמְּאוֹרוֹת הַגְּדוֹלִים – אֶת הַמָּאוֹר הַגָּדוֹל יוֹתֵר לִשְׁלוֹט בַּיּוֹם, וְאֶת הַמָּאוֹר הַקָּטָן יוֹתֵר לִשְׁלוֹט בַּלַּיְלָה, וְאֶת הַכּוֹכָבִים.





* * *



יז וַיִּתֵּ֥ן אֹתָ֛ם אֱלֹהִ֖ים בִּרְקִ֣יעַ הַשָּׁמָ֑יִם לְהָאִ֖יר עַל־הָאָֽרֶץ׃ אֱלֹהִים שָׂם אוֹתָם בָּרָקִיעַ שֶׁל הַשָּׁמַיִם לְהָאִיר עַל הָאָרֶץ.





* * *



יח וְלִמְשֹׁל֙ בַּיּ֣וֹם וּבַלַּ֔יְלָה וּֽלְהַבְדִּ֔יל בֵּ֥ין הָא֖וֹר וּבֵ֣ין הַחֹ֑שֶׁךְ וַיַּ֥רְא אֱלֹהִ֖ים כִּי־טֽוֹב׃ שֶׁיִּקְבְּעוּ אֶת זְמַנֵּי הַיּוֹם וְהַלַּיְלָה וְיַבְדִּילוּ בֵּין הָאוֹר וּבֵין הַחוֹשֶׁךְ, אֱלֹהִים רָאָה וְקָבַע שֶׁזֶּה טוֹב.





* * *



יט וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם רְבִיעִֽי׃ וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל יוֹם רְבִיעִי.





* * *



פ





* * *



כ וַיֹּ֣אמֶר אֱלֹהִ֔ים יִשְׁרְצ֣וּ הַמַּ֔יִם שֶׁ֖רֶץ נֶ֣פֶשׁ חַיָּ֑ה וְעוֹף֙ יְעוֹפֵ֣ף עַל־הָאָ֔רֶץ עַל־פְּנֵ֖י רְקִ֥יעַ הַשָּׁמָֽיִם׃ אֱלֹהִים אָמַר: יִוָּצְרוּ בַּמַּיִם שְׁרָצִים רַבִּים בַּעֲלֵי נֶפֶשׁ חַיָּה, וְעוֹף יְעוֹפֵף מֵעַל הָאָרֶץ לִפְנֵי רְקִיעַ הַשָּׁמַיִם.





* * *



כא וַיִּבְרָ֣א אֱלֹהִ֔ים אֶת־הַתַּנִּינִ֖ם הַגְּדֹלִ֑ים וְאֵ֣ת כׇּל־נֶ֣פֶשׁ הַֽחַיָּ֣ה ׀ הָֽרֹמֶ֡שֶׂת אֲשֶׁר֩ שָׁרְצ֨וּ הַמַּ֜יִם לְמִֽינֵהֶ֗ם וְאֵ֨ת כׇּל־ע֤וֹף כָּנָף֙ לְמִינֵ֔הוּ וַיַּ֥רְא אֱלֹהִ֖ים כִּי־טֽוֹב׃ אֱלֹהִים בָּרָא אֶת יְצוּרֵי הַיָּם הַגְּדוֹלִים, וְאֶת כָּל הַמִּינִים שֶׁל חַיּוֹת זוֹחֲלוֹת שֶׁנּוֹצְרוּ בַּמַּיִם, וְאֶת כָּל מִינֵי הָעוֹפוֹת בַּעֲלֵי הַכָּנָף, אֱלֹהִים רָאָה וְקָבַע שֶׁזֶּה טוֹב.





* * *



כב וַיְבָ֧רֶךְ אֹתָ֛ם אֱלֹהִ֖ים לֵאמֹ֑ר פְּר֣וּ וּרְב֗וּ וּמִלְא֤וּ אֶת־הַמַּ֙יִם֙ בַּיַּמִּ֔ים וְהָע֖וֹף יִ֥רֶב בָּאָֽרֶץ׃ אֱלֹהִים בֵּרֵךְ אוֹתָם וְאָמַר: הוֹלִידוּ וְלָדוֹת וְהִתְרַבּוּ, וּמַלְּאוּ אֶת הַמַּיִם שֶׁבַּיַּמִּים, וְהָעוֹף יִתְרַבֶּה בָּאָרֶץ.





* * *



כג וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם חֲמִישִֽׁי׃ וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל יוֹם חֲמִישִׁי.





* * *



פ





* * *



כד וַיֹּ֣אמֶר אֱלֹהִ֗ים תּוֹצֵ֨א הָאָ֜רֶץ נֶ֤פֶשׁ חַיָּה֙ לְמִינָ֔הּ בְּהֵמָ֥ה וָרֶ֛מֶשׂ וְחַֽיְתוֹ־אֶ֖רֶץ לְמִינָ֑הּ וַֽיְהִי־כֵֽן׃ אֱלֹהִים אָמַר: הָאָרֶץ תּוֹצִיא נְפָשׁוֹת שֶׁל חַיּוֹת לְמִינֵיהֶן הַשּׁוֹנִים, בְּהֵמוֹת וַחֲרָקִים וְחַיּוֹת הַבָּר לְמִינֵיהֶם, וְכָךְ נִהְיָה.





* * *



כה וַיַּ֣עַשׂ אֱלֹהִים֩ אֶת־חַיַּ֨ת הָאָ֜רֶץ לְמִינָ֗הּ וְאֶת־הַבְּהֵמָה֙ לְמִינָ֔הּ וְאֵ֛ת כׇּל־רֶ֥מֶשׂ הָֽאֲדָמָ֖ה לְמִינֵ֑הוּ וַיַּ֥רְא אֱלֹהִ֖ים כִּי־טֽוֹב׃ אֱלֹהִים עָשָׂה אֶת חַיּוֹת הַבָּר לְמִינֵיהֶן וְאֶת הַבְּהֵמוֹת לְמִינֵיהֶן וְאֶת כָּל הַחֲרָקִים שֶׁעַל הָאֲדָמָה לְמִינֵיהֶם, אֱלֹהִים רָאָה וְקָבַע שֶׁזֶּה טוֹב.





* * *



כו וַיֹּ֣אמֶר אֱלֹהִ֔ים נַֽעֲשֶׂ֥ה אָדָ֛ם בְּצַלְמֵ֖נוּ כִּדְמוּתֵ֑נוּ וְיִרְדּוּ֩ בִדְגַ֨ת הַיָּ֜ם וּבְע֣וֹף הַשָּׁמַ֗יִם וּבַבְּהֵמָה֙ וּבְכׇל־הָאָ֔רֶץ וּבְכׇל־הָרֶ֖מֶשׂ הָֽרֹמֵ֥שׂ עַל־הָאָֽרֶץ׃ אֱלֹהִים אָמַר: נַעֲשֶׂה אָדָם שֶׁיִּהְיֶה בְּצוּרָה אֱלֹהִית עֶלְיוֹנָה – דּוֹמֶה לֶאֱלֹהִים, וְיִשְׁלְטוּ בִּדְגֵי הַיָּם וּבְעוֹפוֹת הַשָּׁמַיִם וּבַבְּהֵמוֹת וּבְכָל הָאָרֶץ וּבְכָל הַחֲרָקִים שֶׁזּוֹחֲלִים עַל הָאָרֶץ.





* * *



כז וַיִּבְרָ֨א אֱלֹהִ֤ים ׀ אֶת־הָֽאָדָם֙ בְּצַלְמ֔וֹ בְּצֶ֥לֶם אֱלֹהִ֖ים בָּרָ֣א אֹת֑וֹ זָכָ֥ר וּנְקֵבָ֖ה בָּרָ֥א אֹתָֽם׃ אֱלֹהִים בָּרָא אֶת הָאָדָם בְּצוּרָתוֹ – הוּא בָּרָא אוֹתוֹ בְּצוּרָה אֱלֹהִית עֶלְיוֹנָה, וּבָרָא אוֹתָם זָכָר וּנְקֵבָה.





* * *



כח וַיְבָ֣רֶךְ אֹתָם֮ אֱלֹהִים֒ וַיֹּ֨אמֶר לָהֶ֜ם אֱלֹהִ֗ים פְּר֥וּ וּרְב֛וּ וּמִלְא֥וּ אֶת־הָאָ֖רֶץ וְכִבְשֻׁ֑הָ וּרְד֞וּ בִּדְגַ֤ת הַיָּם֙ וּבְע֣וֹף הַשָּׁמַ֔יִם וּבְכׇל־חַיָּ֖ה הָֽרֹמֶ֥שֶׂת עַל־הָאָֽרֶץ׃ אֱלֹהִים בֵּרֵךְ אוֹתָם, אֱלֹהִים אָמַר לָהֶם: הוֹלִידוּ יְלָדִים וְהִתְרַבּוּ, וּמַלְּאוּ אֶת הָאָרֶץ וְכִבְשׁוּ אוֹתָהּ, וְשִׁלְטוּ בִּדְגֵי הַיָּם וּבְעוֹפוֹת הַשָּׁמַיִם וּבְכָל הַחַיּוֹת שֶׁזּוֹחֲלוֹת עַל הָאָרֶץ.





* * *



כט וַיֹּ֣אמֶר אֱלֹהִ֗ים הִנֵּה֩ נָתַ֨תִּי לָכֶ֜ם אֶת־כׇּל־עֵ֣שֶׂב ׀ זֹרֵ֣עַ זֶ֗רַע אֲשֶׁר֙ עַל־פְּנֵ֣י כׇל־הָאָ֔רֶץ וְאֶת־כׇּל־הָעֵ֛ץ אֲשֶׁר־בּ֥וֹ פְרִי־עֵ֖ץ זֹרֵ֣עַ זָ֑רַע לָכֶ֥ם יִֽהְיֶ֖ה לְאׇכְלָֽה׃ אֱלֹהִים אָמַר: הִנֵּה נָתַתִּי לָכֶם אֶת כָּל הָעֲשָׂבִים מַצְמִיחֵי הַזֶּרַע שֶׁעַל פְּנֵי כָּל הָאָרֶץ וְאֶת כָּל עֲצֵי הַפְּרִי בַּעֲלֵי זְרָעִים לִזְרִיעָה, זֶה יִהְיֶה לָכֶם לַאֲכִילָה.





* * *



ל וּֽלְכׇל־חַיַּ֣ת הָ֠אָ֠רֶץ וּלְכׇל־ע֨וֹף הַשָּׁמַ֜יִם וּלְכֹ֣ל ׀ רוֹמֵ֣שׂ עַל־הָאָ֗רֶץ אֲשֶׁר־בּוֹ֙ נֶ֣פֶשׁ חַיָּ֔ה אֶת־כׇּל־יֶ֥רֶק עֵ֖שֶׂב לְאׇכְלָ֑ה וַֽיְהִי־כֵֽן׃ וּלְכָל חַיַּת הָאָרֶץ וּלְכָל עוֹפוֹת הַשָּׁמַיִם וּלְכָל הַזוֹחֲלִים עַל הָאָרֶץ שֶׁיֵּשׁ בָּהֶם נֶפֶשׁ שֶׁל חַיּוּת, נָתַתִּי רַק אֶת כָּל הָעֲשָׂבִים הַיְּרוּקִים לֶאֱכוֹל, וְכָךְ נִהְיָה.





* * *



לא וַיַּ֤רְא אֱלֹהִים֙ אֶת־כׇּל־אֲשֶׁ֣ר עָשָׂ֔ה וְהִנֵּה־ט֖וֹב מְאֹ֑ד וַֽיְהִי־עֶ֥רֶב וַֽיְהִי־בֹ֖קֶר י֥וֹם הַשִּׁשִּֽׁי׃ אֱלֹהִים רָאָה אֶת כָּל מַה שֶּׁעָשָׂה וְקָבַע שֶׁזֶּה טוֹב מְאוֹד, וְהָיָה עֶרֶב וּבוֹקֶר שֶׁל הַיּוֹם הַשִּׁשִּׁי.





* * *



פ





* * *



ב א וַיְכֻלּ֛וּ הַשָּׁמַ֥יִם וְהָאָ֖רֶץ וְכׇל־צְבָאָֽם׃ נִשְׁלְמָה בְּרִיאַת הַשָּׁמַיִם וְהָאָרֶץ וְכָל מַה שֶּׁבָּהֶם.





* * *



ב וַיְכַ֤ל אֱלֹהִים֙ בַּיּ֣וֹם הַשְּׁבִיעִ֔י מְלַאכְתּ֖וֹ אֲשֶׁ֣ר עָשָׂ֑ה וַיִּשְׁבֹּת֙ בַּיּ֣וֹם הַשְּׁבִיעִ֔י מִכׇּל־מְלַאכְתּ֖וֹ אֲשֶׁ֥ר עָשָֽׂה׃ אֱלֹהִים הִשְׁלִים בַּיּוֹם הַשְּׁבִיעִי אֶת הַמְּלָאכָה שֶׁעָשָׂה, וְשָׁבַת בַּיּוֹם הַשְּׁבִיעִי מִכָּל הַמְּלָאכָה שֶׁעָשָׂה.





* * *



ג וַיְבָ֤רֶךְ אֱלֹהִים֙ אֶת־י֣וֹם הַשְּׁבִיעִ֔י וַיְקַדֵּ֖שׁ אֹת֑וֹ כִּ֣י ב֤וֹ שָׁבַת֙ מִכׇּל־מְלַאכְתּ֔וֹ אֲשֶׁר־בָּרָ֥א אֱלֹהִ֖ים לַעֲשֽׂוֹת׃ אֱלֹהִים בֵּרֵךְ אֶת הַיּוֹם הַשְּׁבִיעִי וְקִדֵּשׁ אוֹתוֹ, כִּי בּוֹ הוּא שָׁבַת מִכָּל הַמְּלָאכָה שֶׁאֱלֹהִים בָּרָא וְעָשָׂה.





* * *





שני


פ פָּרָשַׁת גַּן עֵדֶן





* * *



ד אֵ֣לֶּה תוֹלְד֧וֹת הַשָּׁמַ֛יִם וְהָאָ֖רֶץ בְּהִבָּֽרְאָ֑ם בְּי֗וֹם עֲשׂ֛וֹת יְהֹוָ֥ה אֱלֹהִ֖ים אֶ֥רֶץ וְשָׁמָֽיִם׃ זֶה סִפּוּר תּוֹלְדוֹת הַשָּׁמַיִם וְהָאָרֶץ כְּשֶׁנִּבְרְאוּ, בַּיּוֹם שֶׁה' אֱלֹהִים עָשָׂה אֶרֶץ וְשָׁמַיִם.





* * *



ה וְכֹ֣ל ׀ שִׂ֣יחַ הַשָּׂדֶ֗ה טֶ֚רֶם יִֽהְיֶ֣ה בָאָ֔רֶץ וְכׇל־עֵ֥שֶׂב הַשָּׂדֶ֖ה טֶ֣רֶם יִצְמָ֑ח כִּי֩ לֹ֨א הִמְטִ֜יר יְהֹוָ֤ה אֱלֹהִים֙ עַל־הָאָ֔רֶץ וְאָדָ֣ם אַ֔יִן לַֽעֲבֹ֖ד אֶת־הָֽאֲדָמָֽה׃ עֲדַיִן לֹא הָיָה בָּאָרֶץ שׁוּם שִׂיחַ שָׂדֶה, וַעֲדַיִן לֹא צָמַח שׁוּם עֵשֶׂב שָׂדֶה, כִּי ה' אֱלֹהִים לֹא הִמְטִיר גֶּשֶׁם עַל הָאָרֶץ, וַעֲדַיִן לֹא הָיָה אָדָם שֶׁיַּעֲבוֹד אֶת הָאֲדָמָה.





* * *



ו וְאֵ֖ד יַֽעֲלֶ֣ה מִן־הָאָ֑רֶץ וְהִשְׁקָ֖ה אֶֽת־כׇּל־פְּנֵ֥י הָֽאֲדָמָֽה׃ וְעָנָן עָלָה מֵהָאָרֶץ וְהִשְׁקָה אֶת כָּל פְּנֵי הָאֲדָמָה.





* * *



ז וַיִּ֩יצֶר֩ יְהֹוָ֨ה אֱלֹהִ֜ים אֶת־הָֽאָדָ֗ם עָפָר֙ מִן־הָ֣אֲדָמָ֔ה וַיִּפַּ֥ח בְּאַפָּ֖יו נִשְׁמַ֣ת חַיִּ֑ים וַֽיְהִ֥י הָֽאָדָ֖ם לְנֶ֥פֶשׁ חַיָּֽה׃ ה' אֱלֹהִים יָצַר אֶת הָאָדָם עָפָר מֵהָאֲדָמָה, וְנָשַׁף בְּאַפּוֹ נִשְׁמַת חַיִּים וְהָאָדָם נִהְיָה נֶפֶשׁ חַיָּה.





* * *



ח וַיִּטַּ֞ע יְהֹוָ֧ה אֱלֹהִ֛ים גַּן־בְּעֵ֖דֶן מִקֶּ֑דֶם וַיָּ֣שֶׂם שָׁ֔ם אֶת־הָֽאָדָ֖ם אֲשֶׁ֥ר יָצָֽר׃ ה' אֱלֹהִים נָטַע גַּן בְּעֵדֶן בְּצַד מִזְרָח, וְשָׂם שָׁם אֶת הָאָדָם שֶׁיָּצַר.





* * *



ט וַיַּצְמַ֞ח יְהֹוָ֤ה אֱלֹהִים֙ מִן־הָ֣אֲדָמָ֔ה כׇּל־עֵ֛ץ נֶחְמָ֥ד לְמַרְאֶ֖ה וְט֣וֹב לְמַאֲכָ֑ל וְעֵ֤ץ הַֽחַיִּים֙ בְּת֣וֹךְ הַגָּ֔ן וְעֵ֕ץ הַדַּ֖עַת ט֥וֹב וָרָֽע׃ ה' אֱלֹהִים הִצְמִיחַ מֵהָאֲדָמָה כָּל מִינֵי עֵצִים נֶחֱמָדִים לְמַרְאֶה וְטוֹבִים לְמַאֲכָל, וְעֵץ הַחַיִּים הָיָה בְּתוֹךְ הַגָּן וְגַם עֵץ הַהַבְחָנָה בֵּין טוֹב לְרָע.





* * *



י וְנָהָר֙ יֹצֵ֣א מֵעֵ֔דֶן לְהַשְׁק֖וֹת אֶת־הַגָּ֑ן וּמִשָּׁם֙ יִפָּרֵ֔ד וְהָיָ֖ה לְאַרְבָּעָ֥ה רָאשִֽׁים׃ וְנָהָר יוֹצֵא מֵעֵדֶן לְהַשְׁקוֹת אֶת הַגָּן, וּמִשָּׁם נִפְרָד וְנִהְיָה לְאַרְבָּעָה רָאשֵׁי נְהָרוֹת.





* * *



יא שֵׁ֥ם הָֽאֶחָ֖ד פִּישׁ֑וֹן ה֣וּא הַסֹּבֵ֗ב אֵ֚ת כׇּל־אֶ֣רֶץ הַֽחֲוִילָ֔ה אֲשֶׁר־שָׁ֖ם הַזָּהָֽב׃ שֵׁם הָאֶחָד פִּישׁוֹן הוּא זֶה שֶׁמַּקִּיף אֶת כָּל אֶרֶץ הֹדּוּ שֶׁשָּׁם הַזָּהָב.





* * *



יב וּֽזְהַ֛ב הָאָ֥רֶץ הַהִ֖וא ט֑וֹב שָׁ֥ם הַבְּדֹ֖לַח וְאֶ֥בֶן הַשֹּֽׁהַם׃ וּזְהַב הָאָרֶץ הַהִיא טוֹב, שָׁם הַפְּנִינִים וְאֶבֶן הַשּׁוֹהַם.





* * *



יג וְשֵֽׁם־הַנָּהָ֥ר הַשֵּׁנִ֖י גִּיח֑וֹן ה֣וּא הַסּוֹבֵ֔ב אֵ֖ת כׇּל־אֶ֥רֶץ כּֽוּשׁ׃ וְשֵׁם הַנָּהָר הַשֵּׁנִי גִּיחוֹן, הוּא זֶה שֶׁמַּקִּיף אֶת כָּל אֶרֶץ כּוּשׁ.





* * *



יד וְשֵׁ֨ם הַנָּהָ֤ר הַשְּׁלִישִׁי֙ חִדֶּ֔קֶל ה֥וּא הַֽהֹלֵ֖ךְ קִדְמַ֣ת אַשּׁ֑וּר וְהַנָּהָ֥ר הָֽרְבִיעִ֖י ה֥וּא פְרָֽת׃ וְשֵׁם הַנָּהָר הַשְּׁלִישִׁי חִדֶּקֶל, הוּא זֶה שֶׁהוֹלֵךְ מִזְרָחִית לָעִיר אַשּׁוּר, וְהַנָּהָר הָרְבִיעִי הוּא פְּרָת.





* * *



טו וַיִּקַּ֛ח יְהֹוָ֥ה אֱלֹהִ֖ים אֶת־הָֽאָדָ֑ם וַיַּנִּחֵ֣הוּ בְגַן־עֵ֔דֶן לְעׇבְדָ֖הּ וּלְשׇׁמְרָֽהּ׃ ה' אֱלֹהִים לָקַח אֶת הָאָדָם, וְהִנִּיחַ אוֹתוֹ בְּגַן עֵדֶן לְעָבְדוֹ וּלְשָׁמְרוֹ.





* * *



טז וַיְצַו֙ יְהֹוָ֣ה אֱלֹהִ֔ים עַל־הָֽאָדָ֖ם לֵאמֹ֑ר מִכֹּ֥ל עֵֽץ־הַגָּ֖ן אָכֹ֥ל תֹּאכֵֽל׃ ה' אֱלֹהִים צִוָּה עַל הָאָדָם וְאָמַר: מִכָּל עֲצֵי הַגָּן מֻתָּר לְךָ לֶאֱכוֹל.





* * *



יז וּמֵעֵ֗ץ הַדַּ֙עַת֙ ט֣וֹב וָרָ֔ע לֹ֥א תֹאכַ֖ל מִמֶּ֑נּוּ כִּ֗י בְּי֛וֹם אֲכׇלְךָ֥ מִמֶּ֖נּוּ מ֥וֹת תָּמֽוּת׃ וּמֵעֵץ הַהַבְחָנָה בֵּין טוֹב לְרָע אַל תֹּאכַל מִמֶּנּוּ, כִּי בַּיּוֹם שֶׁתֹּאכַל מִמֶּנּוּ תָּמוּת.





* * *



יח וַיֹּ֙אמֶר֙ יְהֹוָ֣ה אֱלֹהִ֔ים לֹא־ט֛וֹב הֱי֥וֹת הָֽאָדָ֖ם לְבַדּ֑וֹ אֶֽעֱשֶׂה־לּ֥וֹ עֵ֖זֶר כְּנֶגְדּֽוֹ׃ ה' אֱלֹהִים אָמַר: לֹא טוֹב שֶׁהָאָדָם יִהְיֶה לְבַדּוֹ, אֶעֱשֶׂה לוֹ עוֹזֵר לְצִדּוֹ.





* * *



יט וַיִּ֩צֶר֩ יְהֹוָ֨ה אֱלֹהִ֜ים מִן־הָֽאֲדָמָ֗ה כׇּל־חַיַּ֤ת הַשָּׂדֶה֙ וְאֵת֙ כׇּל־ע֣וֹף הַשָּׁמַ֔יִם וַיָּבֵא֙ אֶל־הָ֣אָדָ֔ם לִרְא֖וֹת מַה־יִּקְרָא־ל֑וֹ וְכֹל֩ אֲשֶׁ֨ר יִקְרָא־ל֧וֹ הָֽאָדָ֛ם נֶ֥פֶשׁ חַיָּ֖ה ה֥וּא שְׁמֽוֹ׃ ה' אֱלֹהִים יָצַר מֵהָאֲדָמָה אֶת כָּל חַיּוֹת הַשָּׂדֶה, וְאֶת כָּל עוֹפוֹת הַשָּׁמַיִם, וְהֵבִיא אֶל הָאָדָם לִרְאוֹת מַה יִּקְרָא לָהֶם, וְכָל נֶפֶשׁ חַיָּה שֶׁהָאָדָם קָרָא לוֹ בְּשֵׁם – הוּא שְׁמוֹ.





* * *





שלישי


כ וַיִּקְרָ֨א הָֽאָדָ֜ם שֵׁמ֗וֹת לְכׇל־הַבְּהֵמָה֙ וּלְע֣וֹף הַשָּׁמַ֔יִם וּלְכֹ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה וּלְאָדָ֕ם לֹֽא־מָצָ֥א עֵ֖זֶר כְּנֶגְדּֽוֹ׃ הָאָדָם קָרָא שֵׁמוֹת לְכָל הַבְּהֵמוֹת וּלְעוֹפוֹת הַשָּׁמַיִם וּלְכָל חַיּוֹת הַשָּׂדֶה, וְרָאָה שֶׁלָּאָדָם אֵין עוֹזֵר לְצִדּוֹ כְּמוֹ לַחַיּוֹת.





* * *



כא וַיַּפֵּל֩ יְהֹוָ֨ה אֱלֹהִ֧ים ׀ תַּרְדֵּמָ֛ה עַל־הָאָדָ֖ם וַיִּישָׁ֑ן וַיִּקַּ֗ח אַחַת֙ מִצַּלְעֹתָ֔יו וַיִּסְגֹּ֥ר בָּשָׂ֖ר תַּחְתֶּֽנָּה׃ ה' אֱלֹהִים הִפִּיל תַּרְדֵּמָה עַל הָאָדָם וְהוּא נִרְדַּם, הוּא לָקַח אֶת אֶחָד מִצְּדָדָיו וְסָגַר בְּבָשָׂר בִּמְקוֹם הַחֲתָךְ.





* * *



כב וַיִּ֩בֶן֩ יְהֹוָ֨ה אֱלֹהִ֧ים ׀ אֶֽת־הַצֵּלָ֛ע אֲשֶׁר־לָקַ֥ח מִן־הָֽאָדָ֖ם לְאִשָּׁ֑ה וַיְבִאֶ֖הָ אֶל־הָֽאָדָֽם׃ ה' אֱלֹהִים בָּנָה אֶת הַחֵלֶק שֶׁלָּקַח מֵהָאָדָם, וְעָשָׂה מִמֶּנּוּ אִשָּׁה, וֶהֱבִיאָהּ לָאָדָם.





* * *



כג וַיֹּ֘אמֶר֮ הָֽאָדָם֒ זֹ֣את הַפַּ֗עַם עֶ֚צֶם מֵֽעֲצָמַ֔י וּבָשָׂ֖ר מִבְּשָׂרִ֑י לְזֹאת֙ יִקָּרֵ֣א אִשָּׁ֔ה כִּ֥י מֵאִ֖ישׁ לֻֽקְחָה־זֹּֽאת׃ הָאָדָם אָמַר: הַפַּעַם זֹאת עֶצֶם מֵעֲצָמַי וּבָשָׂר מִבְּשָׂרִי, זֹאת תִּקָּרֵא אִשָּׁה כִּי הִיא נִלְקְחָה מֵאִישׁ.





* * *



כד עַל־כֵּן֙ יַֽעֲזׇב־אִ֔ישׁ אֶת־אָבִ֖יו וְאֶת־אִמּ֑וֹ וְדָבַ֣ק בְּאִשְׁתּ֔וֹ וְהָי֖וּ לְבָשָׂ֥ר אֶחָֽד׃ לָכֵן עוֹזֵב אִישׁ אֶת בֵּית אָבִיו וְאִמּוֹ וְדָבֵק בְּאִשְׁתּוֹ, וְיַלְדֵיהֶם נַעֲשִׂים בָּשָׂר אֶחָד מִשְּׁנֵיהֶם.





* * *



פָּרָשַׁת הַחֵטְא





* * *



כה וַיִּֽהְי֤וּ שְׁנֵיהֶם֙ עֲרוּמִּ֔ים הָֽאָדָ֖ם וְאִשְׁתּ֑וֹ וְלֹ֖א יִתְבֹּשָֽׁשׁוּ׃ הָאָדָם וְאִשְׁתּוֹ שְׁנֵיהֶם הָיוּ עֲרוּמִּים וְלֹא הִתְבַּיְּשׁוּ.





* * *



ג א וְהַנָּחָשׁ֙ הָיָ֣ה עָר֔וּם מִכֹּל֙ חַיַּ֣ת הַשָּׂדֶ֔ה אֲשֶׁ֥ר עָשָׂ֖ה יְהֹוָ֣ה אֱלֹהִ֑ים וַיֹּ֙אמֶר֙ אֶל־הָ֣אִשָּׁ֔ה אַ֚ף כִּֽי־אָמַ֣ר אֱלֹהִ֔ים לֹ֣א תֹֽאכְל֔וּ מִכֹּ֖ל עֵ֥ץ הַגָּֽן׃ וְהַנָּחָשׁ הָיָה עַרְמוּמִי יוֹתֵר מִכָּל חַיּוֹת הַשָּׂדֶה שֶׁה' אֱלֹהִים עָשָׂה, וְהוּא אָמַר לָאִשָּׁה: הַאִם נָכוֹן שֶׁאֱלֹהִים אָמַר אַל תֹּאכְלוּ מִכָּל עֲצֵי הַגָּן?





* * *



ב וַתֹּ֥אמֶר הָֽאִשָּׁ֖ה אֶל־הַנָּחָ֑שׁ מִפְּרִ֥י עֵֽץ־הַגָּ֖ן נֹאכֵֽל׃ אָמְרָה הָאִשָּׁה לַנָּחָשׁ: מִפֵּרוֹת עֲצֵי הַגָּן נֹאכַל.





* * *



ג וּמִפְּרִ֣י הָעֵץ֮ אֲשֶׁ֣ר בְּתוֹךְ־הַגָּן֒ אָמַ֣ר אֱלֹהִ֗ים לֹ֤א תֹֽאכְלוּ֙ מִמֶּ֔נּוּ וְלֹ֥א תִגְּע֖וּ בּ֑וֹ פֶּן־תְּמֻתֽוּן׃ וּמִפֵּרוֹת הָעֵץ הַזֶּה שֶׁבְּתוֹךְ הַגָּן אֱלֹהִים אָמַר: אַל תֹּאכְלוּ מִמֶּנּוּ וְאַל תִּגְּעוּ בּוֹ – שֶׁלֹּא תָּמוּתוּ.





* * *



ד וַיֹּ֥אמֶר הַנָּחָ֖שׁ אֶל־הָֽאִשָּׁ֑ה לֹֽא־מ֖וֹת תְּמֻתֽוּן׃ הַנָּחָשׁ אָמַר לָאִשָּׁה: לֹא תָּמוּתוּ!





* * *



ה כִּ֚י יֹדֵ֣עַ אֱלֹהִ֔ים כִּ֗י בְּיוֹם֙ אֲכׇלְכֶ֣ם מִמֶּ֔נּוּ וְנִפְקְח֖וּ עֵֽינֵיכֶ֑ם וִהְיִיתֶם֙ כֵּֽאלֹהִ֔ים יֹדְעֵ֖י ט֥וֹב וָרָֽע׃ אֶלָּא אֱלֹהִים יוֹדֵעַ שֶׁבַּיּוֹם שֶׁתֹּאכְלוּ מִמֶּנּוּ יִפָּתַח שִׂכְלְכֶם, וְתִהְיוּ כֵּאלֹהִים – מַבְחִינִים בֵּין טוֹב לְרָע.





* * *



ו וַתֵּ֣רֶא הָֽאִשָּׁ֡ה כִּ֣י טוֹב֩ הָעֵ֨ץ לְמַאֲכָ֜ל וְכִ֧י תַֽאֲוָה־ה֣וּא לָעֵינַ֗יִם וְנֶחְמָ֤ד הָעֵץ֙ לְהַשְׂכִּ֔יל וַתִּקַּ֥ח מִפִּרְי֖וֹ וַתֹּאכַ֑ל וַתִּתֵּ֧ן גַּם־לְאִישָׁ֛הּ עִמָּ֖הּ וַיֹּאכַֽל׃ הָאִשָּׁה רָאֲתָה שֶׁהָעֵץ טוֹב לְמַאֲכָל וְשֶׁהוּא תַּאֲוָה לָעֵינַיִם וְהָעֵץ נֶחְמָד וְגוֹרֵם לְהַחְכִּים, הִיא לָקְחָה מִפִּרְיוֹ וְאָכְלָה וְנָתְנָה גַּם לְבַעֲלָהּ שֶׁיֹּאכַל אִתָּהּ, וְהוּא אָכַל.





* * *



ז וַתִּפָּקַ֙חְנָה֙ עֵינֵ֣י שְׁנֵיהֶ֔ם וַיֵּ֣דְע֔וּ כִּ֥י עֵֽירֻמִּ֖ם הֵ֑ם וַֽיִּתְפְּרוּ֙ עֲלֵ֣ה תְאֵנָ֔ה וַיַּעֲשׂ֥וּ לָהֶ֖ם חֲגֹרֹֽת׃ הַשֵּׂכֶל שֶׁל שְׁנֵיהֶם נִפְתַּח וְהֵם הֵבִינוּ שֶׁזֶּה רַע שֶׁהֵם עֲרֻמִּים, הֵם תָּפְרוּ עֲלֵי תְּאֵנָה וְעָשׂוּ לָהֶם חֲגוֹרוֹת.





* * *



ח וַֽיִּשְׁמְע֞וּ אֶת־ק֨וֹל יְהֹוָ֧ה אֱלֹהִ֛ים מִתְהַלֵּ֥ךְ בַּגָּ֖ן לְר֣וּחַ הַיּ֑וֹם וַיִּתְחַבֵּ֨א הָֽאָדָ֜ם וְאִשְׁתּ֗וֹ מִפְּנֵי֙ יְהֹוָ֣ה אֱלֹהִ֔ים בְּת֖וֹךְ עֵ֥ץ הַגָּֽן׃ הֵם שָׁמְעוּ אֶת קוֹל ה' אֱלֹהִים עוֹבֵר בַּגָּן מִכִּוּוּן הַשֶּׁמֶשׁ הַנּוֹטָה לִשְׁקוֹעַ, הָאָדָם וְאִשְׁתּוֹ הִתְחַבְּאוּ מִפְּנֵי ה' אֱלֹהִים בֵּין עֲצֵי הַגָּן.





* * *



ט וַיִּקְרָ֛א יְהֹוָ֥ה אֱלֹהִ֖ים אֶל־הָֽאָדָ֑ם וַיֹּ֥אמֶר ל֖וֹ אַיֶּֽכָּה׃ ה' אֱלֹהִים קָרָא לָאָדָם וְאָמַר לוֹ: אֵיפֹה אַתָּה?





* * *



י וַיֹּ֕אמֶר אֶת־קֹלְךָ֥ שָׁמַ֖עְתִּי בַּגָּ֑ן וָאִירָ֛א כִּֽי־עֵירֹ֥ם אָנֹ֖כִי וָאֵחָבֵֽא׃ אָמַר לוֹ: שָׁמַעְתִּי אֶת קוֹלְךָ בַּגָּן, וּפָחַדְתִּי כִּי אֲנִי עֵרוֹם, וְהִתְחַבֵּאתִי.





* * *



יא וַיֹּ֕אמֶר מִ֚י הִגִּ֣יד לְךָ֔ כִּ֥י עֵירֹ֖ם אָ֑תָּה הֲמִן־הָעֵ֗ץ אֲשֶׁ֧ר צִוִּיתִ֛יךָ לְבִלְתִּ֥י אֲכׇל־מִמֶּ֖נּוּ אָכָֽלְתָּ׃ אָמַר לוֹ: מִי סִפֵּר לְךָ שֶׁאַתָּה עֵרוֹם? הַאִם אָכַלְתָּ מֵהָעֵץ שֶׁצִּוִּיתִי אוֹתְךָ שֶׁלֹּא לֶאֱכוֹל מִמֶּנּוּ?





* * *



יב וַיֹּ֖אמֶר הָֽאָדָ֑ם הָֽאִשָּׁה֙ אֲשֶׁ֣ר נָתַ֣תָּה עִמָּדִ֔י הִ֛וא נָֽתְנָה־לִּ֥י מִן־הָעֵ֖ץ וָאֹכֵֽל׃ אָמַר הָאָדָם: הָאִשָּׁה שֶׁנָּתַתָּה לִי לִהְיוֹת אִתִּי הִיא נָתְנָה לִי מֵהָעֵץ וְאָכַלְתִּי.





* * *



יג וַיֹּ֨אמֶר יְהֹוָ֧ה אֱלֹהִ֛ים לָאִשָּׁ֖ה מַה־זֹּ֣את עָשִׂ֑ית וַתֹּ֙אמֶר֙ הָֽאִשָּׁ֔ה הַנָּחָ֥שׁ הִשִּׁיאַ֖נִי וָאֹכֵֽל׃ ה' אֱלֹהִים אָמַר לָאִשָּׁה: מֶה עָשִׂית?! אָמְרָה הָאִשָּׁה: הַנָּחָשׁ פִּתָּה אוֹתִי וְאָכַלְתִּי.





* * *



יד וַיֹּ֩אמֶר֩ יְהֹוָ֨ה אֱלֹהִ֥ים ׀ אֶֽל־הַנָּחָשׁ֮ כִּ֣י עָשִׂ֣יתָ זֹּאת֒ אָר֤וּר אַתָּה֙ מִכׇּל־הַבְּהֵמָ֔ה וּמִכֹּ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה עַל־גְּחֹנְךָ֣ תֵלֵ֔ךְ וְעָפָ֥ר תֹּאכַ֖ל כׇּל־יְמֵ֥י חַיֶּֽיךָ׃ ה' אֱלֹהִים אָמַר לַנָּחָשׁ: מִפְּנֵי שֶׁעָשִׂיתָ אֶת זֶה – אַתָּה אָרוּר יוֹתֵר מִכָּל הַבְּהֵמוֹת וּמִכָּל חַיּוֹת הַשָּׂדֶה, אַתָּה תִּזְחַל עַל בִּטְנְךָ וְתֹאכַל עָפָר כָּל יְמֵי חַיֶּיךָ.





* * *



טו וְאֵיבָ֣ה ׀ אָשִׁ֗ית בֵּֽינְךָ֙ וּבֵ֣ין הָֽאִשָּׁ֔ה וּבֵ֥ין זַרְעֲךָ֖ וּבֵ֣ין זַרְעָ֑הּ ה֚וּא יְשׁוּפְךָ֣ רֹ֔אשׁ וְאַתָּ֖ה תְּשׁוּפֶ֥נּוּ עָקֵֽב׃ וְאָשִׂים שִׂנְאָה בֵּינְךָ וּבֵין הָאִשָּׁה וּבֵין צֶאֱצָאֶיךָ וּבֵין צֶאֱצָאֶיהָ, הוּא יִמְחַץ אוֹתְךָ בְּרֹאשְׁךָ וְאַתָּה תַּכִּישׁ אוֹתוֹ בַּעֲקֵבוֹ.

יז וּלְאָדָ֣ם אָמַ֗ר כִּֽי־שָׁמַ֘עְתָּ֮ לְק֣וֹל אִשְׁתֶּ֒ךָ֒ וַתֹּ֙אכַל֙ מִן־הָעֵ֔ץ אֲשֶׁ֤ר צִוִּיתִ֙יךָ֙ לֵאמֹ֔ר לֹ֥א תֹאכַ֖ל מִמֶּ֑נּוּ אֲרוּרָ֤ה הָֽאֲדָמָה֙ בַּֽעֲבוּרֶ֔ךָ בְּעִצָּבוֹן֙ תֹּֽאכְלֶ֔נָּה כֹּ֖ל יְמֵ֥י חַיֶּֽיךָ׃ וּלְאָדָם אָמַר: בִּגְלַל שֶׁשָּׁמַעְתָּ לְקוֹל אִשְׁתְּךָ וְאָכַלְתָּ מֵהָעֵץ שֶׁצִּוִּיתִי אוֹתְךָ לֹא לֶאֱכוֹל מִמֶּנּוּ, הָאֲדָמָה תִּהְיֶה מְקֻלֶּלֶת בִּגְלַלְךָ, אַתָּה תֹּאכַל אֶת יְבוּלָהּ בְּעַצְבוּת וּבְמִַאֲמָץ בְּכָל יְמֵי חַיֶּיךָ.





* * *



יח וְק֥וֹץ וְדַרְדַּ֖ר תַּצְמִ֣יחַֽ לָ֑ךְ וְאָכַלְתָּ֖ אֶת־עֵ֥שֶׂב הַשָּׂדֶֽה׃ וְהִיא תַּצְמִיחַ לְךָ קוֹצִים וְדַרְדָּרִים, וְתֹאכַל אֶת עֵשֶׂב הַשָּׂדֶה.





* * *



יט בְּזֵעַ֤ת אַפֶּ֙יךָ֙ תֹּ֣אכַל לֶ֔חֶם עַ֤ד שֽׁוּבְךָ֙ אֶל־הָ֣אֲדָמָ֔ה כִּ֥י מִמֶּ֖נָּה לֻקָּ֑חְתָּ כִּֽי־עָפָ֣ר אַ֔תָּה וְאֶל־עָפָ֖ר תָּשֽׁוּב׃ בְּזֵעַת פָּנֶיךָ תֹּאכַל לֶחֶם, עַד שֶׁבְּמוֹתְךָ תַּחֲזוֹר לָאֲדָמָה כִּי מִמֶּנָּה נִבְרֵאתָ – כִּי אַתָּה עָשׂוּי מֵעָפָר וְאַתָּה תַּחֲזוֹר לְעָפָר.





* * *



כ וַיִּקְרָ֧א הָֽאָדָ֛ם שֵׁ֥ם אִשְׁתּ֖וֹ חַוָּ֑ה כִּ֛י הִ֥וא הָֽיְתָ֖ה אֵ֥ם כׇּל־חָֽי׃ הָאָדָם קָרָא אֶת שֵׁם אִשְׁתּוֹ חַוָּה מִלְּשׁוֹן חַיִּים, כִּי הִיא הָיְתָה הָאֵם שֶׁל כָּל חַי.





* * *



כא וַיַּ֩עַשׂ֩ יְהֹוָ֨ה אֱלֹהִ֜ים לְאָדָ֧ם וּלְאִשְׁתּ֛וֹ כׇּתְנ֥וֹת ע֖וֹר וַיַּלְבִּשֵֽׁם׃ ה' אֱלֹהִים עָשָׂה לְאָדָם וּלְאִשְׁתּוֹ כָּתְנוֹת עוֹר וְהִלְבִּישׁ אוֹתָם.

כב וַיֹּ֣אמֶר ׀ יְהֹוָ֣ה אֱלֹהִ֗ים הֵ֤ן הָֽאָדָם֙ הָיָה֙ כְּאַחַ֣ד מִמֶּ֔נּוּ לָדַ֖עַת ט֣וֹב וָרָ֑ע וְעַתָּ֣ה ׀ פֶּן־יִשְׁלַ֣ח יָד֗וֹ וְלָקַח֙ גַּ֚ם מֵעֵ֣ץ הַֽחַיִּ֔ים וְאָכַ֖ל וָחַ֥י לְעֹלָֽם׃ ה' אֱלֹהִים אָמַר: הֲרֵי הָאָדָם נִהְיָה דּוֹמֶה לֵאלֹהִים – שֶׁהוּא מַבְחִין בֵּין טוֹב לְרָע, וְעַכְשָׁיו צָרִיךְ לַחֲשׁוֹשׁ שֶׁלֹּא יוֹשִׁיט אֶת יָדוֹ וְיִקַּח גַּם מֵעֵץ הַחַיִּים וְיֹאכַל וְיִחְיֶה לְעוֹלָם.





* * *



כג וַֽיְשַׁלְּחֵ֛הוּ יְהֹוָ֥ה אֱלֹהִ֖ים מִגַּן־עֵ֑דֶן לַֽעֲבֹד֙ אֶת־הָ֣אֲדָמָ֔ה אֲשֶׁ֥ר לֻקַּ֖ח מִשָּֽׁם׃ ה' אֱלֹהִים גֵּרֵשׁ אוֹתוֹ מִגַּן עֵדֶן לַעֲבוֹד אֶת הָאֲדָמָה שֶׁהוּא נִלְקַח מִשָּׁם.





* * *



כד וַיְגָ֖רֶשׁ אֶת־הָֽאָדָ֑ם וַיַּשְׁכֵּן֩ מִקֶּ֨דֶם לְגַן־עֵ֜דֶן אֶת־הַכְּרֻבִ֗ים וְאֵ֨ת לַ֤הַט הַחֶ֙רֶב֙ הַמִּתְהַפֶּ֔כֶת לִשְׁמֹ֕ר אֶת־דֶּ֖רֶךְ עֵ֥ץ הַֽחַיִּֽים׃ הוּא גֵּרֵשׁ אֶת הָאָדָם, וְשָׂם לִפְנֵי גַּן עֵדֶן אֶת הַמַּלְאָכִים שֶׁאוֹחֲזִים חֶרֶב חַדָּה מִשְּׁנֵי צְדָדֶיהָ, כְּדֵי לִשְׁמוֹר אֶת דֶּרֶךְ עֵץ הַחַיִּים.

ס פָּרָשַׁת קַיִן וְהֶבֶל





* * *



ד א וְהָ֣אָדָ֔ם יָדַ֖ע אֶת־חַוָּ֣ה אִשְׁתּ֑וֹ וַתַּ֙הַר֙ וַתֵּ֣לֶד אֶת־קַ֔יִן וַתֹּ֕אמֶר קָנִ֥יתִי אִ֖ישׁ אֶת־יְהֹוָֽה׃ וְהָאָדָם הִתְחַבֵּר עִם חַוָּה אִשְׁתּוֹ, וְהִיא נִכְנְסָה לְהֵרָיוֹן וְיָלְדָה אֶת קַיִן וְאָמְרָה: קָנִיתִי אִישׁ מֵה'.





* * *



ב וַתֹּ֣סֶף לָלֶ֔דֶת אֶת־אָחִ֖יו אֶת־הָ֑בֶל וַֽיְהִי־הֶ֙בֶל֙ רֹ֣עֵה צֹ֔אן וְקַ֕יִן הָיָ֖ה עֹבֵ֥ד אֲדָמָֽה׃ הִיא הִמְשִׁיכָה וְיָלְדָה אֶת אָחִיו – אֶת הֶבֶל, הֶבֶל הָיָה רוֹעֵה צֹאן וְקַיִן הָיָה עוֹבֵד אֲדָמָה.





* * *



ג וַֽיְהִ֖י מִקֵּ֣ץ יָמִ֑ים וַיָּבֵ֨א קַ֜יִן מִפְּרִ֧י הָֽאֲדָמָ֛ה מִנְחָ֖ה לַֽיהֹוָֽה׃ בְּסוֹף הַשָּׁנָה קַיִן הֵבִיא מִפֵּרוֹת הָאֲדָמָה מִנְחָה לַה'.





* * *



ד וְהֶ֨בֶל הֵבִ֥יא גַם־ה֛וּא מִבְּכֹר֥וֹת צֹאנ֖וֹ וּמֵֽחֶלְבֵהֶ֑ן וַיִּ֣שַׁע יְהֹוָ֔ה אֶל־הֶ֖בֶל וְאֶל־מִנְחָתֽוֹ׃ וְהֶבֶל הֵבִיא גַּם הוּא מִבְּכוֹרוֹת צֹאנוֹ וּמֵהַשְּׁמֵנִים שֶׁבָּהֶם, ה' פָּנָה בְּרָצוֹן לְהֶבֶל וּלְמִנְחָתוֹ.





* * *



ה וְאֶל־קַ֥יִן וְאֶל־מִנְחָת֖וֹ לֹ֣א שָׁעָ֑ה וַיִּ֤חַר לְקַ֙יִן֙ מְאֹ֔ד וַֽיִּפְּל֖וּ פָּנָֽיו׃ וְאִלּוּ לְקַיִן וּלְמִנְחָתוֹ ה' לֹא פָּנָה, הַדָּבָר הִכְעִיס אֶת קַיִן מְאוֹד וּפָנָיו חָשְׁכוּ.





* * *



ו וַיֹּ֥אמֶר יְהֹוָ֖ה אֶל־קָ֑יִן לָ֚מָּה חָ֣רָה לָ֔ךְ וְלָ֖מָּה נָפְל֥וּ פָנֶֽיךָ׃ ה' אָמַר לְקַיִן, לָמָּה כָּעַסְתָּ וְלָמָּה פָּנֶיךָ חָשְׁכוּ?





* * *



ז הֲל֤וֹא אִם־תֵּיטִיב֙ שְׂאֵ֔ת וְאִם֙ לֹ֣א תֵיטִ֔יב לַפֶּ֖תַח חַטָּ֣את רֹבֵ֑ץ וְאֵלֶ֙יךָ֙ תְּשׁ֣וּקָת֔וֹ וְאַתָּ֖ה תִּמְשׇׁל־בּֽוֹ׃ הֲרֵי אִם תְּתַקֵּן אֶת מַעֲשֶׂיךָ יִסָּלַח לְךָ, וְאִם לֹא תְּתַקֵּן – יֵצֶר הָרַע רוֹבֵץ וְאוֹרֵב לְךָ, וְהוּא מִשְׁתּוֹקֵק לְהַחֲטִיא אוֹתְךָ, אֲבָל אִם תִּרְצֶה אַתָּה תִּמְשׁוֹל בּוֹ.





* * *



ח וַיֹּ֥אמֶר קַ֖יִן אֶל־הֶ֣בֶל אָחִ֑יו וַֽיְהִי֙ בִּהְיוֹתָ֣ם בַּשָּׂדֶ֔ה וַיָּ֥קׇם קַ֛יִן אֶל־הֶ֥בֶל אָחִ֖יו וַיַּהַרְגֵֽהוּ׃ קַיִן קָרָא לְהֶבֶל אָחִיו, וּכְשֶׁהָיוּ בַּשָּׂדֶה קַיִן קָם עַל הֶבֶל אָחִיו וְהָרַג אוֹתוֹ.





* * *



ט וַיֹּ֤אמֶר יְהֹוָה֙ אֶל־קַ֔יִן אֵ֖י הֶ֣בֶל אָחִ֑יךָ וַיֹּ֙אמֶר֙ לֹ֣א יָדַ֔עְתִּי הֲשֹׁמֵ֥ר אָחִ֖י אָנֹֽכִי׃ ה' אָמַר לְקַיִן: אֵיפֹה הֶבֶל אָחִיךָ? אָמַר לוֹ: לֹא יוֹדֵעַ! וְכִי אֲנִי הַשּׁוֹמֵר שֶׁל אָחִי?





* * *



י וַיֹּ֖אמֶר מֶ֣ה עָשִׂ֑יתָ ק֚וֹל דְּמֵ֣י אָחִ֔יךָ צֹעֲקִ֥ים אֵלַ֖י מִן־הָֽאֲדָמָֽה׃ אָמַר לוֹ: מֶה עָשִׂיתָ! קוֹל הַדָּם שֶׁל אָחִיךָ צוֹעֵק אֵלַי מֵהָאֲדָמָה!





* * *



יא וְעַתָּ֖ה אָר֣וּר אָ֑תָּה מִן־הָֽאֲדָמָה֙ אֲשֶׁ֣ר פָּצְתָ֣ה אֶת־פִּ֔יהָ לָקַ֛חַת אֶת־דְּמֵ֥י אָחִ֖יךָ מִיָּדֶֽךָ׃ וְעַכְשָׁיו אַתָּה מְקֻלָּל מֵהָאֲדָמָה שֶׁפָּתְחָה אֶת פִּיהָ לָקַחַת אֶת דַּם אָחִיךָ שֶׁשָּׁפַכְתָּ –





* * *



יב כִּ֤י תַֽעֲבֹד֙ אֶת־הָ֣אֲדָמָ֔ה לֹֽא־תֹסֵ֥ף תֵּת־כֹּחָ֖הּ לָ֑ךְ נָ֥ע וָנָ֖ד תִּֽהְיֶ֥ה בָאָֽרֶץ׃ כְּשֶׁתַּעֲבוֹד אֶת הָאֲדָמָה, הִיא לֹא תּוֹסִיף לָתֵת לְךָ אֶת כּוֹחָהּ, תִּהְיֶה גּוֹלֶה וְנַוָּד בָּאָרֶץ.





* * *



יג וַיֹּ֥אמֶר קַ֖יִן אֶל־יְהֹוָ֑ה גָּד֥וֹל עֲוֺנִ֖י מִנְּשֹֽׂא‏[1]׃ קַיִן אָמַר אֶל ה': וְכִי עֲוֹנִי כָּל כָּךְ גָּדוֹל עַד שֶׁלֹּא תּוּכַל לִסְבּוֹל אוֹתוֹ?





* * *



יד הֵן֩ גֵּרַ֨שְׁתָּ אֹתִ֜י הַיּ֗וֹם מֵעַל֙ פְּנֵ֣י הָֽאֲדָמָ֔ה וּמִפָּנֶ֖יךָ אֶסָּתֵ֑ר וְהָיִ֜יתִי נָ֤ע וָנָד֙ בָּאָ֔רֶץ וְהָיָ֥ה כׇל־מֹצְאִ֖י יַֽהַרְגֵֽנִי׃ הֲרֵי גֵּרַשְׁתָּ אוֹתִי הַיּוֹם מֵעַל פְּנֵי הָאֲדָמָה וּמִפָּנֶיךָ אֵינִי יָכוֹל לְהִסְתַּתֵּר, וְאֶהְיֶה גּוֹלֶה וְנַוָּד בָּאָרֶץ, וְכָל מִי שֶׁיִּמְצָא אוֹתִי יַהֲרוֹג אוֹתִי!





* * *



טו וַיֹּ֧אמֶר ל֣וֹ יְהֹוָ֗ה לָכֵן֙ כׇּל־הֹרֵ֣ג קַ֔יִן שִׁבְעָתַ֖יִם יֻקָּ֑ם וַיָּ֨שֶׂם יְהֹוָ֤ה לְקַ֙יִן֙ א֔וֹת לְבִלְתִּ֥י הַכּוֹת־אֹת֖וֹ כׇּל־מֹצְאֽוֹ׃ אָמַר לוֹ ה': לָכֵן מִי שֶׁיַּהֲרוֹג אֶת קַיִן יִנְקְמוּ בּוֹ פִּי שֶׁבַע! ה' עָשָׂה לְקַיִן סִימָן לְהַבְטָחָתוֹ כְּדֵי שֶׁלֹּא יַכֶּה אוֹתוֹ כָּל מִי שֶׁיִּמְצָא אוֹתוֹ.





* * *



טז וַיֵּ֥צֵא קַ֖יִן מִלִּפְנֵ֣י יְהֹוָ֑ה וַיֵּ֥שֶׁב בְּאֶֽרֶץ־נ֖וֹד קִדְמַת־עֵֽדֶן׃ קַיִן יָצָא מִלִּפְנֵי ה' לַגָּלוּת, וְהוּא גָּר וְנָדַד בָּאָרֶץ שֶׁמִּזְרָחִית לְעֵדֶן.





* * *



יז וַיֵּ֤דַע קַ֙יִן֙ אֶת־אִשְׁתּ֔וֹ וַתַּ֖הַר וַתֵּ֣לֶד אֶת־חֲנ֑וֹךְ וַֽיְהִי֙ בֹּ֣נֶה עִ֔יר וַיִּקְרָא֙ שֵׁ֣ם הָעִ֔יר כְּשֵׁ֖ם בְּנ֥וֹ חֲנֽוֹךְ׃ קַיִן הִתְחַבֵּר עִם אִשְׁתּוֹ וְהִיא נִכְנְסָה לְהֵרָיוֹן וְיָלְדָה אֶת חֲנוֹךְ, הוּא בָּנָה עִיר וְקָרָא אֶת שֵׁם הָעִיר כְּשֵׁם בְּנוֹ חֲנוֹךְ.





* * *



יח וַיִּוָּלֵ֤ד לַֽחֲנוֹךְ֙ אֶת־עִירָ֔ד וְעִירָ֕ד יָלַ֖ד אֶת־מְחֽוּיָאֵ֑ל וּמְחִיָּיאֵ֗ל יָלַד֙ אֶת־מְת֣וּשָׁאֵ֔ל וּמְתוּשָׁאֵ֖ל יָלַ֥ד אֶת־לָֽמֶךְ׃ לַחֲנוֹךְ נוֹלַד עִירָד, וְעִירָד יָלַד אֶת מְחוּיָאֵל, וּמְחִיָּיאֵל יָלַד אֶת מְתוּשָׁאֵל, וּמְתוּשָׁאֵל יָלַד אֶת לֶמֶךְ.





* * *





חמישי


יט וַיִּֽקַּֽח־ל֥וֹ לֶ֖מֶךְ שְׁתֵּ֣י נָשִׁ֑ים שֵׁ֤ם הָֽאַחַת֙ עָדָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית צִלָּֽה׃ לֶמֶךְ לָקַח לוֹ שְׁתֵּי נָשִׁים, שֵׁם הָאַחַת עָדָה וְשֵׁם הַשֵּׁנִית צִלָּה.





* * *



כ וַתֵּ֥לֶד עָדָ֖ה אֶת־יָבָ֑ל ה֣וּא הָיָ֔ה אֲבִ֕י יֹשֵׁ֥ב אֹ֖הֶל וּמִקְנֶֽה׃ עָדָה יָלְדָה אֶת יָבָל, הוּא הָיָה אֲבִי הַיּוֹשְׁבִים בְּאוֹהֲלִים וְרוֹעִים בְּהֵמוֹת.





* * *



כא וְשֵׁ֥ם אָחִ֖יו יוּבָ֑ל ה֣וּא הָיָ֔ה אֲבִ֕י כׇּל־תֹּפֵ֥שׂ כִּנּ֖וֹר וְעוּגָֽב׃ וְשֵׁם אָחִיו יוּבָל, הוּא הָיָה אֲבִי כָּל הַמְּנַגְּנִים בְּכִנּוֹר וְחָלִיל.





* * *



כב וְצִלָּ֣ה גַם־הִ֗וא יָֽלְדָה֙ אֶת־תּ֣וּבַל קַ֔יִן לֹטֵ֕שׁ כׇּל־חֹרֵ֥שׁ נְחֹ֖שֶׁת וּבַרְזֶ֑ל וַֽאֲח֥וֹת תּֽוּבַל־קַ֖יִן נַֽעֲמָֽה׃ וְצִלָּה גַּם הִיא יָלְדָה, אֶת תּוּבַל קַיִן – אֲבִי כָּל הָאֻמָּנִים בִּכְלֵי נְחוֹשֶׁת וּבַרְזֶל, וַאֲחוֹת תּוּבַל קַיִן הָיְתָה נַעֲמָה.





* * *



כג וַיֹּ֨אמֶר לֶ֜מֶךְ לְנָשָׁ֗יו עָדָ֤ה וְצִלָּה֙ שְׁמַ֣עַן קוֹלִ֔י נְשֵׁ֣י לֶ֔מֶךְ הַאְזֵ֖נָּה אִמְרָתִ֑י כִּ֣י אִ֤ישׁ הָרַ֙גְתִּי֙ לְפִצְעִ֔י וְיֶ֖לֶד לְחַבֻּרָתִֽי׃ לֶמֶךְ אָמַר לְנָשָׁיו: עָדָה וְצִלָּה שִׁמְעוּ קוֹלִי, נְשֵׁי לֶמֶךְ הַאֲזִינוּ לִדְבָרִי, כִּי הָאִישׁ שֶׁהָרַגְתִּי בְּשׁוֹגֵג הָיָה לְצַעֲרִי וּלְפִצְעִי, וְאוֹתוֹ הַיֶּלֶד שֶׁהָרַגְתִּי – זֶה הָיָה לְחַבּוּרָתִי וּלְמַכָּתִי.





* * *



כד כִּ֥י שִׁבְעָתַ֖יִם יֻקַּם־קָ֑יִן וְלֶ֖מֶךְ שִׁבְעִ֥ים וְשִׁבְעָֽה׃ הֲרֵי אִם יִנְקְמוּ בְּהוֹרֵג קַיִן פִּי שֶׁבַע, בְּהוֹרֵג לֶמֶךְ יִנְקְמוּ פִּי שִׁבְעִים וְשֶׁבַע!





* * *



כה וַיֵּ֨דַע אָדָ֥ם עוֹד֙ אֶת־אִשְׁתּ֔וֹ וַתֵּ֣לֶד בֵּ֔ן וַתִּקְרָ֥א אֶת־שְׁמ֖וֹ שֵׁ֑ת כִּ֣י שָֽׁת־לִ֤י אֱלֹהִים֙ זֶ֣רַע אַחֵ֔ר תַּ֣חַת הֶ֔בֶל כִּ֥י הֲרָג֖וֹ קָֽיִן׃ אָדָם הִתְחַבֵּר עוֹד עִם אִשְׁתּוֹ, וְהִיא יָלְדָה בֵּן וְקָרְאָה אֶת שְׁמוֹ שֵׁת מִלְּשׁוֹן תַּשְׁתִּית, כִּי אֱלֹהִים בִּסֵּס לִי הֶמְשֵׁךְ בְּצֶאֱצָא אַחֵר בִּמְקוֹם הֶבֶל שֶׁקַיִן הָרַג אוֹתוֹ.





* * *



כו וּלְשֵׁ֤ת גַּם־הוּא֙ יֻלַּד־בֵּ֔ן וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ אֱנ֑וֹשׁ אָ֣ז הוּחַ֔ל לִקְרֹ֖א בְּשֵׁ֥ם יְהֹוָֽה׃ וְגַם לְשֵׁת נוֹלַד בֵּן, וְהוּא קָרָא אֶת שְׁמוֹ אֱנוֹשׁ, אָז הִתְחִילוּ לִטְעוֹת וְלִקְרֹא לֶאֱלִילִים בְּשֵׁם ה'.





* * *





ששי


ס פָּרָשַׁת תּוֹלְדֹת בְּנֵי אָדָם





* * *



ה א זֶ֣ה סֵ֔פֶר‏[2] תּוֹלְדֹ֖ת אָדָ֑ם בְּי֗וֹם בְּרֹ֤א אֱלֹהִים֙ אָדָ֔ם בִּדְמ֥וּת אֱלֹהִ֖ים עָשָׂ֥ה אֹתֽוֹ׃ זֶה סִפּוּר צֶאֱצָאֵי אָדָם, בַּיּוֹם שֶׁאֱלֹהִים בָּרָא אָדָם הוּא עָשָׂה אוֹתוֹ דּוֹמֶה לֵאלֹהִים.





* * *



ב זָכָ֥ר וּנְקֵבָ֖ה בְּרָאָ֑ם וַיְבָ֣רֶךְ אֹתָ֗ם וַיִּקְרָ֤א אֶת־שְׁמָם֙ אָדָ֔ם בְּי֖וֹם הִבָּֽרְאָֽם׃ הוּא בָּרָא אוֹתָם זָכָר וּנְקֵבָה, וְהוּא בֵּרֵךְ אוֹתָם וְקָרָא אֶת שְׁמָם אָדָם בַּיּוֹם שֶׁנִּבְרְאוּ.





* * *



ג וַֽיְחִ֣י אָדָ֗ם שְׁלֹשִׁ֤ים וּמְאַת֙ שָׁנָ֔ה וַיּ֥וֹלֶד בִּדְמוּת֖וֹ כְּצַלְמ֑וֹ וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ שֵֽׁת׃ אָדָם חַי מֵאָה וּשְׁלוֹשִׁים שָׁנָה וְהוֹלִיד דּוֹמֶה לוֹ בִּדְמוּתוֹ וּכְצוּרָתוֹ וְקָרָא אֶת שְׁמוֹ שֵׁת.





* * *



ד וַיִּֽהְי֣וּ יְמֵי־אָדָ֗ם אַֽחֲרֵי֙ הוֹלִיד֣וֹ אֶת־שֵׁ֔ת שְׁמֹנֶ֥ה מֵאֹ֖ת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ יְמֵי אָדָם אַחֲרֵי שֶׁהוֹלִיד אֶת שֵׁת הָיוּ שְׁמוֹנֶה מֵאוֹת שָׁנָה, וְהוּא הוֹלִיד בָּנִים וּבָנוֹת.





* * *



ה וַיִּֽהְי֞וּ כׇּל־יְמֵ֤י אָדָם֙ אֲשֶׁר־חַ֔י תְּשַׁ֤ע מֵאוֹת֙ שָׁנָ֔ה וּשְׁלֹשִׁ֖ים שָׁנָ֑ה וַיָּמֹֽת׃ כָּל הַיָּמִים שֶׁאָדָם חַי הָיוּ תְּשַׁע מֵאוֹת וּשְׁלוֹשִׁים שָׁנָה, וְהוּא מֵת.





* * *



ס





* * *



ו וַֽיְחִי־שֵׁ֕ת חָמֵ֥שׁ שָׁנִ֖ים וּמְאַ֣ת שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־אֱנֽוֹשׁ׃ שֵׁת חַי מֵאָה וְחָמֵשׁ שָׁנִים, וְהוֹלִיד אֶת אֱנוֹשׁ.





* * *



ז וַֽיְחִי־שֵׁ֗ת אַֽחֲרֵי֙ הוֹלִיד֣וֹ אֶת־אֱנ֔וֹשׁ שֶׁ֣בַע שָׁנִ֔ים וּשְׁמֹנֶ֥ה מֵא֖וֹת שָׁנָ֑ה וַיּ֥וֹלֶד בָּנִ֖ים וּבָנֽוֹת׃ שֵׁת חַי אַחֲרֵי שֶׁהוֹלִיד אֶת אֱנוֹשׁ שְׁמוֹנֶה מֵאוֹת וְשֶׁבַע שָׁנִים, וְהוֹלִיד בָּנִים וּבָנוֹת.





* * *



ח וַיִּֽהְיוּ֙ כׇּל־יְמֵי־שֵׁ֔ת שְׁתֵּ֤ים עֶשְׂרֵה֙ שָׁנָ֔ה וּתְשַׁ֥ע מֵא֖וֹת שָׁנָ֑ה וַיָּמֹֽת׃ כָּל הַיָּמִים שֶׁשֵּׁת חַי הָיוּ תְּשַׁע מֵאוֹת וּשְׁתֵּים עֶשְׂרֵה שָׁנָה, וְהוּא מֵת.





* * *



ס





* * *



ט וַֽיְחִ֥י אֱנ֖וֹשׁ תִּשְׁעִ֣ים שָׁנָ֑ה וַיּ֖וֹלֶד אֶת־קֵינָֽן׃ אֱנוֹשׁ חַי תִּשְׁעִים שָׁנָה, וְהוֹלִיד אֶת קֵינָן.

ה וַיַּ֣רְא יְהֹוָ֔ה כִּ֥י רַבָּ֛ה רָעַ֥ת הָאָדָ֖ם בָּאָ֑רֶץ וְכׇל־יֵ֙צֶר֙ מַחְשְׁבֹ֣ת לִבּ֔וֹ רַ֥ק רַ֖ע כׇּל־הַיּֽוֹם׃ ה' רָאָה שֶׁרָעַת הָאָדָם בָּאָרֶץ רַבָּה, וְכָל יְצָרָיו וּמַחְשְׁבוֹתָיו כָּל הַיּוֹם הֵם רַק לַעֲשׂוֹת רַע.





* * *



ו וַיִּנָּ֣חֶם יְהֹוָ֔ה כִּֽי־עָשָׂ֥ה אֶת־הָֽאָדָ֖ם בָּאָ֑רֶץ וַיִּתְעַצֵּ֖ב אֶל־לִבּֽוֹ׃ ה' הִתְחָרֵט כִּבְיָכוֹל עַל שֶׁעָשָׂה אֶת הָאָדָם בָּאָרֶץ, וְהִתְעַצֵּב בְּלִבּוֹ.





* * *



ז וַיֹּ֣אמֶר יְהֹוָ֗ה אֶמְחֶ֨ה אֶת־הָאָדָ֤ם אֲשֶׁר־בָּרָ֙אתִי֙ מֵעַל֙ פְּנֵ֣י הָֽאֲדָמָ֔ה מֵֽאָדָם֙ עַד־בְּהֵמָ֔ה עַד־רֶ֖מֶשׂ וְעַד־ע֣וֹף הַשָּׁמָ֑יִם כִּ֥י נִחַ֖מְתִּי כִּ֥י עֲשִׂיתִֽם׃ ה' אָמַר: אֶמְחֶה אֶת הָאָדָם שֶׁבָּרָאתִי מֵעַל פְּנֵי הָאֲדָמָה, אֶמְחֶה אֶת הַכֹּל – מֵאָדָם עַד בְּהֵמָה עַד חָרָק וְעַד עוֹף הַשָּׁמַיִם כִּי הִתְחָרַטְתִּי עַל שֶׁעֲשִׂיתִי אוֹתָם.





* * *



ח וְנֹ֕חַ מָ֥צָא חֵ֖ן בְּעֵינֵ֥י יְהֹוָֽה׃ וְרַק נֹחַ מָצָא חֵן בְּעֵינֵי ה'.
