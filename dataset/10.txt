פָּרָשַׁת וָאֶתְחַנַּן

קְרִיאַת הַתּוֹרָה וְהַהַפְטָרָה · מִקְרָא וְתַרְגּוּם · שְׁנַֽיִם מִקְרָא וְאֶחָד תַּרְגּוּם

מִקְרָא וְתַרְגּוּם וְרַשִׁ"י (בהכנה) · מִקְרָא מְבוֹאָר





ראשון


ס פָּרָשַׁת תְּפִלַּת מֹשֶׁה





* * *



ג כג וָאֶתְחַנַּ֖ן אֶל־יְהֹוָ֑ה בָּעֵ֥ת הַהִ֖וא לֵאמֹֽר׃ בְּאוֹתוֹ הַזְּמַן הִתְחַנַּנְתִּי אֶל ה' וְאָמַרְתִּי:





* * *



כד אֲדֹנָ֣י יֱהֹוִ֗ה אַתָּ֤ה הַֽחִלּ֙וֹתָ֙ לְהַרְא֣וֹת אֶֽת־עַבְדְּךָ֔ אֶ֨ת־גׇּדְלְךָ֔ וְאֶת־יָדְךָ֖ הַחֲזָקָ֑ה אֲשֶׁ֤ר מִי־אֵל֙ בַּשָּׁמַ֣יִם וּבָאָ֔רֶץ אֲשֶׁר־יַעֲשֶׂ֥ה כְמַעֲשֶׂ֖יךָ וְכִגְבוּרֹתֶֽךָ׃ ה' אֱלֹהִים, אַתָּה הִתְחַלְתָּ לְהַרְאוֹת לְעַבְדְּךָ אֶת גְּדֻלָּתְךָ וְאֶת יָדְךָ הַחֲזָקָה – שֶׁאֵין אֵל בַּשָּׁמַיִם וּבָאָרֶץ שֶׁיַּעֲשֶׂה כְּמַעֲשֶׂיךָ וְכִגְבוּרוֹתֶיךָ!





* * *



כה אֶעְבְּרָה־נָּ֗א וְאֶרְאֶה֙ אֶת־הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן הָהָ֥ר הַטּ֛וֹב הַזֶּ֖ה וְהַלְּבָנֹֽן׃ אַפְשֵׁר לִי בְּבַקָּשָׁה לַעֲבוֹר וְלִרְאוֹת אֶת הָאָרֶץ הַטּוֹבָה שֶׁמֵּעֵבֶר לַיַּרְדֵּן, אֶת הָהָר הַטּוֹב הַזֶּה וְאֶת הַלְּבָנוֹן.





* * *



כו וַיִּתְעַבֵּ֨ר יְהֹוָ֥ה בִּי֙ לְמַ֣עַנְכֶ֔ם וְלֹ֥א שָׁמַ֖ע אֵלָ֑י וַיֹּ֨אמֶר יְהֹוָ֤ה אֵלַי֙ רַב־לָ֔ךְ אַל־תּ֗וֹסֶף דַּבֵּ֥ר אֵלַ֛י ע֖וֹד בַּדָּבָ֥ר הַזֶּֽה׃ ה' כָּעַס עָלַי בִּגְלַלְכֶם וְלֹא שָׁמַע לִי, ה' אָמַר לִי: מַסְפִּיק! אַל תַּמְשִׁיךְ לְדַבֵּר אֵלַי עוֹד בָּעִנְיָן הַזֶּה!





* * *



כז עֲלֵ֣ה ׀ רֹ֣אשׁ הַפִּסְגָּ֗ה וְשָׂ֥א עֵינֶ֛יךָ יָ֧מָּה וְצָפֹ֛נָה וְתֵימָ֥נָה וּמִזְרָ֖חָה וּרְאֵ֣ה בְעֵינֶ֑יךָ כִּי־לֹ֥א תַעֲבֹ֖ר אֶת־הַיַּרְדֵּ֥ן הַזֶּֽה׃ עֲלֵה לְרֹאשׁ הַפִּסְגָּה וְהַבֵּט לְמַעֲרָב, לְצָפוֹן, לְדָרוֹם, וּלְמִזְרָח, וּרְאֵה בְּעֵינֶיךָ אֶת הָאָרֶץ, כִּי אַתָּה לֹא תַּעֲבוֹר אֶת הַיַּרְדֵּן הַזֶּה.





* * *



כח וְצַ֥ו אֶת־יְהוֹשֻׁ֖עַ וְחַזְּקֵ֣הוּ וְאַמְּצֵ֑הוּ כִּי־ה֣וּא יַעֲבֹ֗ר לִפְנֵי֙ הָעָ֣ם הַזֶּ֔ה וְהוּא֙ יַנְחִ֣יל אוֹתָ֔ם אֶת־הָאָ֖רֶץ אֲשֶׁ֥ר תִּרְאֶֽה׃ וְצַוֵּה אֶת יְהוֹשֻׁעַ וְחַזֵּק אוֹתוֹ שֶׁיִּהְיֶה אַמִּיץ, כִּי הוּא יַעֲבוֹר בְּרֹאשׁ הָעָם הַזֶּה, וְהוּא יַנְחִיל אוֹתָם אֶת הָאָרֶץ שֶׁתִּרְאֶה.





* * *



כט וַנֵּ֣שֶׁב בַּגָּ֔יְא מ֖וּל בֵּ֥ית פְּעֽוֹר׃ יָשַׁבְנוּ בָּעֵמֶק מוּל בֵּית פְּעוֹר.





* * *



פ פָּרָשַׁת הַזְהָרַת הַבְּרִית





* * *



ד א וְעַתָּ֣ה יִשְׂרָאֵ֗ל שְׁמַ֤ע אֶל־הַֽחֻקִּים֙ וְאֶל־הַמִּשְׁפָּטִ֔ים אֲשֶׁ֧ר אָֽנֹכִ֛י מְלַמֵּ֥ד אֶתְכֶ֖ם לַעֲשׂ֑וֹת לְמַ֣עַן תִּֽחְי֗וּ וּבָאתֶם֙ וִֽירִשְׁתֶּ֣ם אֶת־הָאָ֔רֶץ אֲשֶׁ֧ר יְהֹוָ֛ה אֱלֹהֵ֥י אֲבֹתֵיכֶ֖ם נֹתֵ֥ן לָכֶֽם׃ וְעַכְשָׁיו יִשְׂרָאֵל, צַיֵּת לַחֻקִּים וְלַמִּשְׁפָּטִים שֶׁאֲנִי מְלַמֵּד אֶתְכֶם לַעֲשׂוֹת, בִּשְׁבִיל שֶׁתִּחְיוּ וְתָבוֹאוּ וְתִכְבְּשׁוּ אֶת הָאָרֶץ שֶׁה' אֱלֹהֵי אֲבוֹתֵיכֶם נוֹתֵן לָכֶם.





* * *



ב לֹ֣א תֹסִ֗פוּ עַל־הַדָּבָר֙ אֲשֶׁ֤ר אָנֹכִי֙ מְצַוֶּ֣ה אֶתְכֶ֔ם וְלֹ֥א תִגְרְע֖וּ מִמֶּ֑נּוּ לִשְׁמֹ֗ר אֶת־מִצְוֺת֙ יְהֹוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁ֥ר אָנֹכִ֖י מְצַוֶּ֥ה אֶתְכֶֽם׃ אַל תּוֹסִיפוּ עַל הַמִּצְווֹת שֶׁאֲנִי מְצַוֶּה אֶתְכֶם, וְאַל תַּחְסִירוּ מֵהֶם, שִׁמְרוּ אֶת מִצְווֹת ה' אֱלֹהֵיכֶם שֶׁאֲנִי מְצַוֶּה אֶתְכֶם.





* * *



ג עֵֽינֵיכֶם֙ הָֽרֹא֔וֹת אֵ֛ת אֲשֶׁר־עָשָׂ֥ה יְהֹוָ֖ה בְּבַ֣עַל פְּע֑וֹר כִּ֣י כׇל־הָאִ֗ישׁ אֲשֶׁ֤ר הָלַךְ֙ אַחֲרֵ֣י בַֽעַל־פְּע֔וֹר הִשְׁמִיד֛וֹ יְהֹוָ֥ה אֱלֹהֶ֖יךָ מִקִּרְבֶּֽךָ׃ עֵינֵיכֶם רָאוּ אֶת מַה שֶּׁה' עָשָׂה לְבַעַל פְּעוֹר, שֶׁכָּל אִישׁ שֶׁהָלַךְ אַחֲרֵי בַּעַל פְּעוֹר – ה' אֱלֹהֶיךָ הִשְׁמִיד אוֹתוֹ מִתּוֹכְךָ.





* * *



ד וְאַתֶּם֙ הַדְּבֵקִ֔ים בַּיהֹוָ֖ה אֱלֹהֵיכֶ֑ם חַיִּ֥ים כֻּלְּכֶ֖ם הַיּֽוֹם׃ וְאִלּוּ אַתֶּם שֶׁדְּבֵקִים בַּה' אֱלֹהֵיכֶם – כֻּלְּכֶם חַיִּים הַיּוֹם.





* * *





שני


ה רְאֵ֣ה ׀ לִמַּ֣דְתִּי אֶתְכֶ֗ם חֻקִּים֙ וּמִשְׁפָּטִ֔ים כַּאֲשֶׁ֥ר צִוַּ֖נִי יְהֹוָ֣ה אֱלֹהָ֑י לַעֲשׂ֣וֹת כֵּ֔ן בְּקֶ֣רֶב הָאָ֔רֶץ אֲשֶׁ֥ר אַתֶּ֛ם בָּאִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ רְאֵה לִמַּדְתִּי אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כְּמוֹ שֶׁה' אֱלֹהַי צִוָּה אוֹתִי, שֶׁתַּעֲשׂוּ אוֹתָם בְּתוֹךְ הָאָרֶץ שֶׁאַתֶּם בָּאִים לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.





* * *



ו וּשְׁמַרְתֶּם֮ וַעֲשִׂיתֶם֒ כִּ֣י הִ֤וא חׇכְמַתְכֶם֙ וּבִ֣ינַתְכֶ֔ם לְעֵינֵ֖י הָעַמִּ֑ים אֲשֶׁ֣ר יִשְׁמְע֗וּן אֵ֚ת כׇּל־הַחֻקִּ֣ים הָאֵ֔לֶּה וְאָמְר֗וּ רַ֚ק עַם־חָכָ֣ם וְנָב֔וֹן הַגּ֥וֹי הַגָּד֖וֹל הַזֶּֽה׃ הַקְפִּידוּ וְקַיְּמוּ אֶת הַתּוֹרָה כִּי הִיא חָכְמַתְכֶם וּבִינַתְכֶם לְעֵינֵי הָעַמִּים, שֶׁיִּשְׁמְעוּ אֶת כָּל הַחֻקִּים הָאֵלֶּה וְיֹאמְרוּ: רַק הָעָם הַגָּדוֹל הַזֶּה הוּא עַם חָכָם וְנָבוֹן!





* * *



ז כִּ֚י מִי־ג֣וֹי גָּד֔וֹל אֲשֶׁר־ל֥וֹ אֱלֹהִ֖ים קְרֹבִ֣ים אֵלָ֑יו כַּיהֹוָ֣ה אֱלֹהֵ֔ינוּ בְּכׇל־קׇרְאֵ֖נוּ אֵלָֽיו׃ הֲרֵי מִי עוֹד עַם גָּדוֹל כָּזֶה שֶׁיֵּשׁ לוֹ אֱלֹהִים קְרוֹבִים אֵלָיו, כַּה' אֱלֹהֵינוּ שֶׁשּׁוֹמֵעַ אֶת תְּפִלּוֹתֵנוּ בְּכָל קְרִיאוֹתֵינוּ אֵלָיו!





* * *



ח וּמִי֙ גּ֣וֹי גָּד֔וֹל אֲשֶׁר־ל֛וֹ חֻקִּ֥ים וּמִשְׁפָּטִ֖ים צַדִּיקִ֑ם כְּכֹל֙ הַתּוֹרָ֣ה הַזֹּ֔את אֲשֶׁ֧ר אָנֹכִ֛י נֹתֵ֥ן לִפְנֵיכֶ֖ם הַיּֽוֹם׃ וּמִי עַם גָּדוֹל כָּזֶה שֶׁיֵּשׁ לוֹ חֻקִּים וּמִשְׁפָּטִים הֲגוּנִים כְּמוֹ הַתּוֹרָה הַזֹּאת שֶׁאֲנִי נוֹתֵן לָכֶם הַיּוֹם!





* * *



ט רַ֡ק הִשָּׁ֣מֶר לְךָ֩ וּשְׁמֹ֨ר נַפְשְׁךָ֜ מְאֹ֗ד פֶּן־תִּשְׁכַּ֨ח אֶת־הַדְּבָרִ֜ים אֲשֶׁר־רָא֣וּ עֵינֶ֗יךָ וּפֶן־יָס֙וּרוּ֙ מִלְּבָ֣בְךָ֔ כֹּ֖ל יְמֵ֣י חַיֶּ֑יךָ וְהוֹדַעְתָּ֥ם לְבָנֶ֖יךָ וְלִבְנֵ֥י בָנֶֽיךָ׃ רַק הִזָּהֵר לְךָ וּשְׁמוֹר אֶת נַפְשְׁךָ מְאוֹד שֶׁלֹּא תִּשְׁכַּח אֶת הַדְּבָרִים שֶׁעֵינֶיךָ רָאוּ – שֶׁלֹּא יָסוּרוּ מִלִּבְּךָ כָּל יְמֵי חַיֶּיךָ, וְתוֹדִיעַ אוֹתָם לְבָנֶיךָ וְלִבְנֵי בָּנֶיךָ –





* * *



י י֗וֹם אֲשֶׁ֨ר עָמַ֜דְתָּ לִפְנֵ֨י יְהֹוָ֣ה אֱלֹהֶ֘יךָ֮ בְּחֹרֵב֒ בֶּאֱמֹ֨ר יְהֹוָ֜ה אֵלַ֗י הַקְהֶל־לִי֙ אֶת־הָעָ֔ם וְאַשְׁמִעֵ֖ם אֶת־דְּבָרָ֑י אֲשֶׁ֨ר יִלְמְד֜וּן לְיִרְאָ֣ה אֹתִ֗י כׇּל־הַיָּמִים֙ אֲשֶׁ֨ר הֵ֤ם חַיִּים֙ עַל־הָ֣אֲדָמָ֔ה וְאֶת־בְּנֵיהֶ֖ם יְלַמֵּדֽוּן׃ אֶת הַיּוֹם שֶׁעָמַדְתָּ לִפְנֵי ה' אֱלֹהֶיךָ בְּחוֹרֵב, כְּשֶׁה' אָמַר לִי: אֱסוֹף אֵלַי אֶת הָעָם וְאַשְׁמִיעַ לָהֶם אֶת דְּבָרַי, כְּדֵי שֶׁיִּלְמְדוּ לְכַבֵּד אוֹתִי כָּל הַיָּמִים שֶׁהֵם חַיִּים עַל הָאֲדָמָה, וִילַמְּדוּ גַּם אֶת בְּנֵיהֶם.





* * *



יא וַתִּקְרְב֥וּן וַתַּֽעַמְד֖וּן תַּ֣חַת הָהָ֑ר וְהָהָ֞ר בֹּעֵ֤ר בָּאֵשׁ֙ עַד־לֵ֣ב הַשָּׁמַ֔יִם חֹ֖שֶׁךְ עָנָ֥ן וַעֲרָפֶֽל׃ נִגַּשְׁתֶּם וַעֲמַדְתֶּם לְמַרְגְּלוֹת הָהָר, וְהָהָר בּוֹעֵר בָּאֵשׁ עַד לֵב הַשָּׁמַיִם, וְהָיָה חוֹשֶׁךְ וְעָנָן וַעֲרָפֶל.





* * *



יב וַיְדַבֵּ֧ר יְהֹוָ֛ה אֲלֵיכֶ֖ם מִתּ֣וֹךְ הָאֵ֑שׁ ק֤וֹל דְּבָרִים֙ אַתֶּ֣ם שֹׁמְעִ֔ים וּתְמוּנָ֛ה אֵינְכֶ֥ם רֹאִ֖ים זוּלָתִ֥י קֽוֹל׃ ה' דִּבֵּר אֲלֵיכֶם מִתּוֹךְ הָאֵשׁ, שְׁמַעְתֶּם קוֹל שֶׁל דִּבּוּר, אֲבָל לֹא רְאִיתֶם צוּרָה אֶלָּא רַק קוֹל.





* * *



יג וַיַּגֵּ֨ד לָכֶ֜ם אֶת־בְּרִית֗וֹ אֲשֶׁ֨ר צִוָּ֤ה אֶתְכֶם֙ לַעֲשׂ֔וֹת עֲשֶׂ֖רֶת הַדְּבָרִ֑ים וַֽיִּכְתְּבֵ֔ם עַל־שְׁנֵ֖י לֻח֥וֹת אֲבָנִֽים׃ הוּא הוֹדִיעַ לָכֶם אֶת הַהֶסְכֵּם אִתּוֹ, אֶת מַה שֶּׁצִּוָּה אֶתְכֶם לַעֲשׂוֹת – עֲשֶׂרֶת הַדִּבְּרוֹת, וְהוּא כָּתַב אוֹתָם עַל שְׁנֵי לוּחוֹת אֲבָנִים.





* * *



יד וְאֹתִ֞י צִוָּ֤ה יְהֹוָה֙ בָּעֵ֣ת הַהִ֔וא לְלַמֵּ֣ד אֶתְכֶ֔ם חֻקִּ֖ים וּמִשְׁפָּטִ֑ים לַעֲשֹׂתְכֶ֣ם אֹתָ֔ם בָּאָ֕רֶץ אֲשֶׁ֥ר אַתֶּ֛ם עֹבְרִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ וְאוֹתִי ה' צִוָּה בַּזְּמַן הַהוּא לְלַמֵּד אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים, שֶׁתַּעֲשׂוּ אוֹתָם בָּאָרֶץ שֶׁאַתֶּם עוֹבְרִים לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.





* * *



טו וְנִשְׁמַרְתֶּ֥ם מְאֹ֖ד לְנַפְשֹׁתֵיכֶ֑ם כִּ֣י לֹ֤א רְאִיתֶם֙ כׇּל־תְּמוּנָ֔ה בְּי֗וֹם דִּבֶּ֨ר יְהֹוָ֧ה אֲלֵיכֶ֛ם בְּחֹרֵ֖ב מִתּ֥וֹךְ הָאֵֽשׁ׃ הִזָּהֲרוּ מְאוֹד עַל נַפְשׁוֹתֵיכֶם וְאַל תַּעַבְדוּ לְגוּף גַּשְׁמִי, שֶׁהֲרֵי לֹא רְאִיתֶם שׁוּם צוּרָה בַּיּוֹם שֶׁה' דִּבֵּר אֲלֵיכֶם בְּחוֹרֵב מִתּוֹךְ הָאֵשׁ.





* * *



טז פֶּ֨ן־תַּשְׁחִת֔וּן וַעֲשִׂיתֶ֥ם לָכֶ֛ם פֶּ֖סֶל תְּמוּנַ֣ת כׇּל־סָ֑מֶל תַּבְנִ֥ית זָכָ֖ר א֥וֹ נְקֵבָֽה׃ הִזָּהֲרוּ שֶׁלֹּא תְּקַלְקְלוּ אֶת מַעֲשֵׂיכֶם, וְתַעֲשׂוּ לָכֶם פֶּסֶל, דְּמוּת שֶׁל אֵיזוֹ צוּרָה, צוּרַת זָכָר אוֹ נְקֵבָה.





* * *



יז תַּבְנִ֕ית כׇּל־בְּהֵמָ֖ה אֲשֶׁ֣ר בָּאָ֑רֶץ תַּבְנִית֙ כׇּל־צִפּ֣וֹר כָּנָ֔ף אֲשֶׁ֥ר תָּע֖וּף בַּשָּׁמָֽיִם׃ – צוּרָה שֶׁל אַחַת מִכָּל הַבְּהֵמוֹת שֶׁבָּאָרֶץ אוֹ מִכָּל הַצִּפּוֹרִים – בַּעֲלֵי כָּנָף שֶׁעָפִים בַּשָּׁמַיִם.





* * *



יח תַּבְנִ֕ית כׇּל־רֹמֵ֖שׂ בָּאֲדָמָ֑ה תַּבְנִ֛ית כׇּל־דָּגָ֥ה אֲשֶׁר־בַּמַּ֖יִם מִתַּ֥חַת לָאָֽרֶץ׃ אוֹ צוּרָה שֶׁל אֶחָד מִכָּל הַזּוֹחֲלִים עַל הָאֲדָמָה אוֹ מִכָּל הַדָּגִים שֶׁבַּמַּיִם מִתַּחַת לְגוֹבַהּ הַיַּבָּשָׁה.





* * *



יט וּפֶן־תִּשָּׂ֨א עֵינֶ֜יךָ הַשָּׁמַ֗יְמָה וְֽ֠רָאִ֠יתָ אֶת־הַשֶּׁ֨מֶשׁ וְאֶת־הַיָּרֵ֜חַ וְאֶת־הַכּֽוֹכָבִ֗ים כֹּ֚ל צְבָ֣א הַשָּׁמַ֔יִם וְנִדַּחְתָּ֛ וְהִשְׁתַּחֲוִ֥יתָ לָהֶ֖ם וַעֲבַדְתָּ֑ם אֲשֶׁ֨ר חָלַ֜ק יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ אֹתָ֔ם לְכֹל֙ הָֽעַמִּ֔ים תַּ֖חַת כׇּל־הַשָּׁמָֽיִם׃ וְשֶׁלֹּא תָּרִים אֶת מַבָּטְךָ לַשָּׁמַיִם, וְתִרְאֶה אֶת הַשֶּׁמֶשׁ וְאֶת הַיָּרֵחַ וְאֶת הַכּוֹכָבִים – כָּל גַּרְמֵי הַשָּׁמַיִם, וְתִטְעֶה וְתִשְׁתַּחֲוֶה לָהֶם וְתַעֲבוֹד אוֹתָם, אֶת אֵלֶּה שֶׁה' אֱלֹהֶיךָ חִלֵּק לְכָל הָעַמִּים שֶׁתַּחַת כָּל הַשָּׁמַיִם, לִטְעוֹת אַחֲרֵיהֶם.





* * *



כ וְאֶתְכֶם֙ לָקַ֣ח יְהֹוָ֔ה וַיּוֹצִ֥א אֶתְכֶ֛ם מִכּ֥וּר הַבַּרְזֶ֖ל מִמִּצְרָ֑יִם לִהְי֥וֹת ל֛וֹ לְעַ֥ם נַחֲלָ֖ה כַּיּ֥וֹם הַזֶּֽה׃ וְאִלּוּ אֶתְכֶם ה' לָקַח וְהוֹצִיא אֶתְכֶם מִכִּבְשַׁן הִתּוּךְ וְיִצּוּר הַבַּרְזֶל – מִמִּצְרַיִם, שֶׁתִּהְיוּ לוֹ עַמּוֹ וְנַחֲלָתוֹ, כְּמוֹ הַיּוֹם.





* * *



כא וַֽיהֹוָ֥ה הִתְאַנַּף־בִּ֖י עַל־דִּבְרֵיכֶ֑ם וַיִּשָּׁבַ֗ע לְבִלְתִּ֤י עׇבְרִי֙ אֶת־הַיַּרְדֵּ֔ן וּלְבִלְתִּי־בֹא֙ אֶל־הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁר֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַחֲלָֽה׃ וַה' כָּעַס עָלַי בִּגְלַלְכֶם, הוּא נִשְׁבַּע שֶׁלֹּא אֶעֱבוֹר אֶת הַיַּרְדֵּן, וְשֶׁלֹּא אָבוֹא לָאָרֶץ הַטּוֹבָה, שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ נַחֲלָה.





* * *



כב כִּ֣י אָנֹכִ֥י מֵת֙ בָּאָ֣רֶץ הַזֹּ֔את אֵינֶ֥נִּי עֹבֵ֖ר אֶת־הַיַּרְדֵּ֑ן וְאַתֶּם֙ עֹֽבְרִ֔ים וִֽירִשְׁתֶּ֕ם אֶת־הָאָ֥רֶץ הַטּוֹבָ֖ה הַזֹּֽאת׃ אֶלָּא אֲנִי אָמוּת בָּאָרֶץ הַזֹּאת, אֵינֶנִּי עוֹבֵר אֶת הַיַּרְדֵּן, וְאִלּוּ אַתֶּם תַּעַבְרוּ וְתִכְבְּשׁוּ אֶת הָאָרֶץ הַטּוֹבָה הַזֹּאת.





* * *



כג הִשָּׁמְר֣וּ לָכֶ֗ם פֶּֽן־תִּשְׁכְּחוּ֙ אֶת־בְּרִ֤ית יְהֹוָה֙ אֱלֹ֣הֵיכֶ֔ם אֲשֶׁ֥ר כָּרַ֖ת עִמָּכֶ֑ם וַעֲשִׂיתֶ֨ם לָכֶ֥ם פֶּ֙סֶל֙ תְּמ֣וּנַת כֹּ֔ל אֲשֶׁ֥ר צִוְּךָ֖ יְהֹוָ֥ה אֱלֹהֶֽיךָ׃ הִזָּהֲרוּ לָכֶם שֶׁלֹּא תִּשְׁכְּחוּ אֶת הַהֶסְכֵּם עִם ה' אֱלֹהֵיכֶם שֶׁעָשָׂה אִתְּכֶם, וְתַעֲשׂוּ לָכֶם פֶּסֶל שֶׁל כָּל צוּרָה שֶׁהִיא, שֶׁה' אֱלֹהֶיךָ צִוָּה אוֹתְךָ שֶׁלֹּא תַּעֲשֶׂה.





* * *



כד כִּ֚י יְהֹוָ֣ה אֱלֹהֶ֔יךָ אֵ֥שׁ אֹכְלָ֖ה ה֑וּא אֵ֖ל קַנָּֽא׃ כִּי ה' אֱלֹהֶיךָ מַעֲנִישׁ כְּמוֹ אֵשׁ אוֹכֶלֶת – הוּא אֵל נוֹקֵם.





* * *



פ פָּרָשַׁת הַזְהָרַת הַגָּלוּת





* * *



כה כִּֽי־תוֹלִ֤יד בָּנִים֙ וּבְנֵ֣י בָנִ֔ים וְנוֹשַׁנְתֶּ֖ם בָּאָ֑רֶץ וְהִשְׁחַתֶּ֗ם וַעֲשִׂ֤יתֶם פֶּ֙סֶל֙ תְּמ֣וּנַת כֹּ֔ל וַעֲשִׂיתֶ֥ם הָרַ֛ע בְּעֵינֵ֥י יְהֹוָה־אֱלֹהֶ֖יךָ לְהַכְעִיסֽוֹ׃ אִם תּוֹלִיד בָּנִים וּנְכָדִים, וְתִהְיוּ וָתִיקִים בָּאָרֶץ, וּתְקַלְקְלוּ אֶת מַעֲשֵׂיכֶם, וְתַעֲשׂוּ פֶּסֶל שֶׁל כָּל צוּרָה שֶׁהִיא, וְתַעֲשׂוּ אֶת הָרַע בְּעֵינֵי ה' אֱלֹהֶיךָ, לְהַכְעִיס אוֹתוֹ.





* * *



כו הַעִידֹ֩תִי֩ בָכֶ֨ם הַיּ֜וֹם אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֗רֶץ כִּֽי־אָבֹ֣ד תֹּאבֵדוּן֮ מַהֵר֒ מֵעַ֣ל הָאָ֔רֶץ אֲשֶׁ֨ר אַתֶּ֜ם עֹבְרִ֧ים אֶת־הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּ֑הּ לֹֽא־תַאֲרִיכֻ֤ן יָמִים֙ עָלֶ֔יהָ כִּ֥י הִשָּׁמֵ֖ד תִּשָּׁמֵדֽוּן׃ אֲנִי מֵעִיד בָּכֶם הַיּוֹם אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ, שֶׁתֹּאבְדוּ מַהֵר מֵהָאָרֶץ שֶׁאַתֶּם עוֹבְרִים אֶת הַיַּרְדֵּן לְשָׁם לִכְבּוֹשׁ אוֹתָהּ, לֹא תִּתְקַיְּמוּ עָלֶיהָ זְמַן רָב, אֶלָּא תֻּשְׁמְדוּ.





* * *



כז וְהֵפִ֧יץ יְהֹוָ֛ה אֶתְכֶ֖ם בָּעַמִּ֑ים וְנִשְׁאַרְתֶּם֙ מְתֵ֣י מִסְפָּ֔ר בַּגּוֹיִ֕ם אֲשֶׁ֨ר יְנַהֵ֧ג יְהֹוָ֛ה אֶתְכֶ֖ם שָֽׁמָּה׃ ה' יְפַזֵּר אֶתְכֶם בֵּין הָעַמִּים וְתִשָּׁאֲרוּ מְעַט אֲנָשִׁים, בָּעַמִּים שֶׁה' יוֹלִיךְ אֶתְכֶם לְשָׁם.





* * *



כח וַעֲבַדְתֶּם־שָׁ֣ם אֱלֹהִ֔ים מַעֲשֵׂ֖ה יְדֵ֣י אָדָ֑ם עֵ֣ץ וָאֶ֔בֶן אֲשֶׁ֤ר לֹֽא־יִרְאוּן֙ וְלֹ֣א יִשְׁמְע֔וּן וְלֹ֥א יֹֽאכְל֖וּן וְלֹ֥א יְרִיחֻֽן׃ וְתַעַבְדוּ שָׁם עוֹבְדֵי אֱלִילִים מַעֲשֵׂה יְדֵי אָדָם, עֲשׂוּיִים עֵץ וָאֶבֶן, שֶׁלֹּא רוֹאִים וְלֹא שׁוֹמְעִים וְלֹא אוֹכְלִים וְלֹא מְרִיחִים.





* * *



כט וּבִקַּשְׁתֶּ֥ם מִשָּׁ֛ם אֶת־יְהֹוָ֥ה אֱלֹהֶ֖יךָ וּמָצָ֑אתָ כִּ֣י תִדְרְשֶׁ֔נּוּ בְּכׇל־לְבָבְךָ֖ וּבְכׇל־נַפְשֶֽׁךָ׃ וּתְחַפְּשׂוּ מִשָּׁם אֶת הַדֶּרֶךְ לַה' אֱלֹהֶיךָ, וְתִמְצָא, כִּי תְּחַפֵּשׂ אוֹתוֹ בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ.





* * *



ל בַּצַּ֣ר לְךָ֔ וּמְצָא֕וּךָ כֹּ֖ל הַדְּבָרִ֣ים הָאֵ֑לֶּה בְּאַחֲרִית֙ הַיָּמִ֔ים וְשַׁבְתָּ֙ עַד־יְהֹוָ֣ה אֱלֹהֶ֔יךָ וְשָׁמַעְתָּ֖ בְּקֹלֽוֹ׃ כְּשֶׁיִּהְיוּ לְךָ צָרוֹת וְיִקְרוּ לְךָ כָּל הַדְּבָרִים הָאֵלֶּה, בְּאַחֲרִית הַיָּמִים, תָּשׁוּב אֶל ה' אֱלֹהֶיךָ וְתִשְׁמַע בְּקוֹלוֹ.





* * *



לא כִּ֣י אֵ֤ל רַחוּם֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ לֹ֥א יַרְפְּךָ֖ וְלֹ֣א יַשְׁחִיתֶ֑ךָ וְלֹ֤א יִשְׁכַּח֙ אֶת־בְּרִ֣ית אֲבֹתֶ֔יךָ אֲשֶׁ֥ר נִשְׁבַּ֖ע לָהֶֽם׃ כִּי ה' אֱלֹהֶיךָ אֵל רַחוּם – הוּא לֹא יַעֲזוֹב אוֹתְךָ וְלֹא יַשְׁמִיד אוֹתְךָ, וְלֹא יִשְׁכַּח אֶת הַהֶסְכֵּם עִם אֲבוֹתֶיךָ שֶׁהוּא נִשְׁבַּע לָהֶם.





* * *



לב כִּ֣י שְׁאַל־נָא֩ לְיָמִ֨ים רִֽאשֹׁנִ֜ים אֲשֶׁר־הָי֣וּ לְפָנֶ֗יךָ לְמִן־הַיּוֹם֙ אֲשֶׁר֩ בָּרָ֨א אֱלֹהִ֤ים ׀ אָדָם֙ עַל־הָאָ֔רֶץ וּלְמִקְצֵ֥ה הַשָּׁמַ֖יִם וְעַד־קְצֵ֣ה הַשָּׁמָ֑יִם הֲנִֽהְיָ֗ה כַּדָּבָ֤ר הַגָּדוֹל֙ הַזֶּ֔ה א֖וֹ הֲנִשְׁמַ֥ע כָּמֹֽהוּ׃ הֲרֵי, שְׁאַל נָא עַל הַיָּמִים הָרִאשׁוֹנִים שֶׁהָיוּ לְפָנֶיךָ – מֵהַיּוֹם שֶׁאֱלֹהִים בָּרָא אָדָם עַל הָאָרֶץ, וּמִקְצֵה הַשָּׁמַיִם וְעַד קְצֵה הַשָּׁמַיִם, הַאִם הָיָה כַּדָּבָר הַגָּדוֹל הַזֶּה, אוֹ הַאִם נִשְׁמַע כָּמוֹהוּ?





* * *



לג הֲשָׁ֣מַֽע עָם֩ ק֨וֹל אֱלֹהִ֜ים מְדַבֵּ֧ר מִתּוֹךְ־הָאֵ֛שׁ כַּאֲשֶׁר־שָׁמַ֥עְתָּ אַתָּ֖ה וַיֶּֽחִי׃ הַאִם עָם שָׁמַע קוֹל אֱלֹהִים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כְּמוֹ שֶׁשָּׁמַעְתָּ אַתָּה, וְנִשְׁאַר חָי?





* * *



לד א֣וֹ ׀ הֲנִסָּ֣ה אֱלֹהִ֗ים לָ֠ב֠וֹא לָקַ֨חַת ל֣וֹ גוֹי֮ מִקֶּ֣רֶב גּוֹי֒ בְּמַסֹּת֩ בְּאֹתֹ֨ת וּבְמוֹפְתִ֜ים וּבְמִלְחָמָ֗ה וּבְיָ֤ד חֲזָקָה֙ וּבִזְר֣וֹעַ נְטוּיָ֔ה וּבְמוֹרָאִ֖ים גְּדֹלִ֑ים כְּ֠כֹ֠ל אֲשֶׁר־עָשָׂ֨ה לָכֶ֜ם יְהֹוָ֧ה אֱלֹהֵיכֶ֛ם בְּמִצְרַ֖יִם לְעֵינֶֽיךָ׃ אוֹ הַאִם נִסָּה אֵיזֶה אֱלֹהִים לָבוֹא לָקַחַת לוֹ עַם מִתּוֹךְ עַם אַחֵר, בְּנִסִּים בְּסִימָנִים וּבְהוֹכָחוֹת וּבְמִלְחָמָה וּבְיָד חֲזָקָה וּבִזְרוֹעַ מוּנֶפֶת לְהַכּוֹת וּבְמַרְאוֹת גְּדוֹלִים, כְּכָל מַה שֶּׁה' אֱלֹהֵיכֶם עָשָׂה לָכֶם בְּמִצְרַיִם לְעֵינֶיךָ?





* * *



לה אַתָּה֙ הׇרְאֵ֣תָ לָדַ֔עַת כִּ֥י יְהֹוָ֖ה ה֣וּא הָאֱלֹהִ֑ים אֵ֥ין ע֖וֹד מִלְּבַדּֽוֹ׃ הוּא הֶרְאָה לְךָ אֶת כָּל זֹאת כְּדֵי שֶׁתֵּדַע שֶׁה' הוּא הָאֱלֹהִים – אֵין עוֹד מִלְּבַדּוֹ.





* * *



לו מִן־הַשָּׁמַ֛יִם הִשְׁמִֽיעֲךָ֥ אֶת־קֹל֖וֹ לְיַסְּרֶ֑ךָּ וְעַל־הָאָ֗רֶץ הֶרְאֲךָ֙ אֶת־אִשּׁ֣וֹ הַגְּדוֹלָ֔ה וּדְבָרָ֥יו שָׁמַ֖עְתָּ מִתּ֥וֹךְ הָאֵֽשׁ׃ מֵהַשָּׁמַיִם הוּא הִשְׁמִיעַ לְךָ אֶת קוֹלוֹ כְּדֵי לְלַמֵּד אוֹתְךָ מוּסָר, וְעַל הָאָרֶץ הוּא הֶרְאָה לְךָ אֶת הָאֵשׁ הַגְּדוֹלָה שֶׁלּוֹ, וְאֶת דְּבָרָיו שָׁמַעְתָּ מִתּוֹךְ הָאֵשׁ.





* * *



לז וְתַ֗חַת כִּ֤י אָהַב֙ אֶת־אֲבֹתֶ֔יךָ וַיִּבְחַ֥ר בְּזַרְע֖וֹ אַחֲרָ֑יו וַיּוֹצִֽאֲךָ֧ בְּפָנָ֛יו בְּכֹח֥וֹ הַגָּדֹ֖ל מִמִּצְרָֽיִם׃ וְכֵיוָן שֶׁאָהַב אֶת אֲבוֹתֶיךָ הוּא בָּחַר בְּצֶאֱצָאֵיהֶם שֶׁיִּהְיוּ הֶמְשֵׁכָם, וְהוֹצִיא אוֹתְךָ בִּרְצוֹנוֹ הַטּוֹב בְּכוֹחוֹ הַגָּדוֹל מִמִּצְרַיִם.





* * *



לח לְהוֹרִ֗ישׁ גּוֹיִ֛ם גְּדֹלִ֧ים וַעֲצֻמִ֛ים מִמְּךָ֖ מִפָּנֶ֑יךָ לַהֲבִֽיאֲךָ֗ לָֽתֶת־לְךָ֧ אֶת־אַרְצָ֛ם נַחֲלָ֖ה כַּיּ֥וֹם הַזֶּֽה׃ לְאַפְשֵׁר לְךָ לִכְבּוֹשׁ עַמִּים גְּדוֹלִים וַעֲצוּמִים מִמְּךָ, לַהֲבִיא אוֹתְךָ וְלָתֵת לְךָ אֶת אַרְצָם לְנַחֲלָה, כְּמוֹ שֶׁעָשָׂה הַיּוֹם – שֶׁכְּבַשְׁתֶּם אֶת סִיחוֹן וְעוֹג.





* * *



לט וְיָדַעְתָּ֣ הַיּ֗וֹם וַהֲשֵׁבֹתָ֮ אֶל־לְבָבֶ֒ךָ֒ כִּ֤י יְהֹוָה֙ ה֣וּא הָֽאֱלֹהִ֔ים בַּשָּׁמַ֣יִם מִמַּ֔עַל וְעַל־הָאָ֖רֶץ מִתָּ֑חַת אֵ֖ין עֽוֹד׃ וְעָלֶיךָ לָדַעַת הַיּוֹם וּלְהַפְנִים בְּלִבְּךָ, שֶׁה' הוּא הָאֱלֹהִים בַּשָּׁמַיִם מִמַּעַל וְעַל הָאָרֶץ מִתַּחַת, וְאֵין עוֹד אֱלֹהִים.





* * *



מ וְשָׁמַרְתָּ֞ אֶת־חֻקָּ֣יו וְאֶת־מִצְוֺתָ֗יו אֲשֶׁ֨ר אָנֹכִ֤י מְצַוְּךָ֙ הַיּ֔וֹם אֲשֶׁר֙ יִיטַ֣ב לְךָ֔ וּלְבָנֶ֖יךָ אַחֲרֶ֑יךָ וּלְמַ֨עַן תַּאֲרִ֤יךְ יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר יְהֹוָ֧ה אֱלֹהֶ֛יךָ נֹתֵ֥ן לְךָ֖ כׇּל־הַיָּמִֽים׃ הַקְפֵּד לְקַיֵּם אֶת חֻקָּיו וְאֶת מִצְווֹתָיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם, כְּדֵי שֶׁיִּיטַב לְךָ וּלְבָנֶיךָ אַחֲרֶיךָ, וּכְדֵי שֶׁתַּאֲרִיךְ יָמִים תָּמִיד עַל הָאֲדָמָה שֶׁה' אֱלֹהֶיךָ נוֹתֵן לְךָ.





* * *





שלישי


פ פָּרָשַׁת עֵבֶר הַיַּרְדֵּן





* * *



מא אָ֣ז יַבְדִּ֤יל מֹשֶׁה֙ שָׁלֹ֣שׁ עָרִ֔ים בְּעֵ֖בֶר הַיַּרְדֵּ֑ן מִזְרְחָ֖ה שָֽׁמֶשׁ׃ מֹשֶׁה הִבְדִּיל אָז שָׁלוֹשׁ עָרִים בְּעֵבֶר הַיַּרְדֵּן הַמִּזְרָחִי.





* * *



מב לָנֻ֨ס שָׁ֜מָּה רוֹצֵ֗חַ אֲשֶׁ֨ר יִרְצַ֤ח אֶת־רֵעֵ֙הוּ֙ בִּבְלִי־דַ֔עַת וְה֛וּא לֹא־שֹׂנֵ֥א ל֖וֹ מִתְּמֹ֣ל שִׁלְשֹׁ֑ם וְנָ֗ס אֶל־אַחַ֛ת מִן־הֶעָרִ֥ים הָאֵ֖ל וָחָֽי׃ כְּדֵי שֶׁמִּי שֶׁהָרַג אֶת חֲבֵרוֹ בְּלִי כַּוָּנָה וְהוּא לֹא שׂוֹנֵא אוֹתוֹ מֵאֶתְמוֹל וְשִׁלְשׁוֹם יוּכַל לִבְרוֹחַ לְשָׁם – הוּא יִבְרַח לְאַחַת מֵהֶעָרִים הָאֵלֶּה וְיִשָּׁאֵר חַי.





* * *



מג אֶת־בֶּ֧צֶר בַּמִּדְבָּ֛ר בְּאֶ֥רֶץ הַמִּישֹׁ֖ר לָרֽאוּבֵנִ֑י וְאֶת־רָאמֹ֤ת בַּגִּלְעָד֙ לַגָּדִ֔י וְאֶת־גּוֹלָ֥ן בַּבָּשָׁ֖ן לַֽמְנַשִּֽׁי׃ לְשֵׁבֶט רְאוּבֵן אֶת הָעִיר בֶּצֶר בַּמִּדְבָּר בְּאֶרֶץ הַמִּישׁוֹר, לְשֵׁבֶט גָּד אֶת רָמוֹת בַּגִּלְעָד, וּלְשֵׁבֶט מְנַשֶּׁה אֶת גּוֹלָן בַּבָּשָׁן.





* * *



מד וְזֹ֖את הַתּוֹרָ֑ה אֲשֶׁר־שָׂ֣ם מֹשֶׁ֔ה לִפְנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ הַדְּבָרִים הָאֵלֶּה – זֹאת הַתּוֹרָה שֶׁמֹּשֶׁה שָׂם לִפְנֵי בְּנֵי יִשְׂרָאֵל.





* * *



מה אֵ֚לֶּה הָֽעֵדֹ֔ת וְהַֽחֻקִּ֖ים וְהַמִּשְׁפָּטִ֑ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל בְּצֵאתָ֖ם מִמִּצְרָֽיִם׃ אֵלֶּה הָאַזְהָרוֹת הַחֻקִּים וְהַמִּשְׁפָּטִים שֶׁמֹּשֶׁה דִּבֵּר עֲלֵיהֶם אֶל בְּנֵי יִשְׂרָאֵל כְּשֶׁיָּצְאוּ מִמִּצְרַיִם.





* * *



מו בְּעֵ֨בֶר הַיַּרְדֵּ֜ן בַּגַּ֗יְא מ֚וּל בֵּ֣ית פְּע֔וֹר בְּאֶ֗רֶץ סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן אֲשֶׁ֨ר הִכָּ֤ה מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל בְּצֵאתָ֖ם מִמִּצְרָֽיִם׃ בְּעֵבֶר הַיַּרְדֵּן בָּעֵמֶק מוּל בֵּית-פְּעוֹר בְּאֶרֶץ סִיחוֹן מֶלֶךְ הָאֱמוֹרִי שֶׁיּוֹשֵׁב בְּחֶשְׁבּוֹן, שֶׁמֹּשֶׁה וּבְנֵי יִשְׂרָאֵל הֵבִיסוּ כְּשֶׁיָּצְאוּ מִמִּצְרַיִם.





* * *



מז וַיִּֽירְשׁ֨וּ אֶת־אַרְצ֜וֹ וְאֶת־אֶ֣רֶץ ׀ ע֣וֹג מֶֽלֶךְ־הַבָּשָׁ֗ן שְׁנֵי֙ מַלְכֵ֣י הָֽאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִזְרַ֖ח שָֽׁמֶשׁ׃ הֵם כָּבְשׁוּ אֶת אַרְצוֹ וְאֶת אֶרֶץ עוֹג מֶלֶךְ הַבָּשָׁן – שְׁנֵי מַלְכֵי הָאֱמוֹרִי שֶׁבְּעֵבֶר הַיַּרְדֵּן הַמִּזְרָחִי.





* * *

יח אֶֽת־הַדְּבָרִ֣ים הָאֵ֡לֶּה דִּבֶּר֩ יְהֹוָ֨ה אֶל־כׇּל־קְהַלְכֶ֜ם בָּהָ֗ר מִתּ֤וֹךְ הָאֵשׁ֙ הֶֽעָנָ֣ן וְהָֽעֲרָפֶ֔ל ק֥וֹל גָּד֖וֹל וְלֹ֣א יָסָ֑ף וַֽיִּכְתְּבֵ֗ם עַל־שְׁנֵי֙ לֻחֹ֣ת אֲבָנִ֔ים וַֽיִּתְּנֵ֖ם אֵלָֽי׃ אֶת הַדְּבָרִים הָאֵלֶּה ה' דִּבֵּר אֶל כָּל קְהַלְכֶם בָּהָר מִתּוֹךְ הָאֵשׁ הֶעָנָן וְהָעֲרָפֶל, קוֹל גָּדוֹל שֶׁלֹּא הִמְשִׁיךְ אַחַר כָּךְ, וְהוּא כָּתַב אוֹתָם עַל שְׁנֵי לוּחוֹת אֲבָנִים וּמָסַר אוֹתָם אֵלַי.





* * *



יט וַיְהִ֗י כְּשׇׁמְעֲכֶ֤ם אֶת־הַקּוֹל֙ מִתּ֣וֹךְ הַחֹ֔שֶׁךְ וְהָהָ֖ר בֹּעֵ֣ר בָּאֵ֑שׁ וַתִּקְרְב֣וּן אֵלַ֔י כׇּל־רָאשֵׁ֥י שִׁבְטֵיכֶ֖ם וְזִקְנֵיכֶֽם׃ כְּשֶׁשְּׁמַעְתֶּם אֶת הַקּוֹל מִתּוֹךְ הַחוֹשֶׁךְ וְהָהָר בּוֹעֵר בָּאֵשׁ, נִגַּשְׁתֶּם אֵלַי כָּל רָאשֵׁי שִׁבְטֵיכֶם וְזִקְנֵיכֶם.





* * *



כ וַתֹּאמְר֗וּ הֵ֣ן הֶרְאָ֜נוּ יְהֹוָ֤ה אֱלֹהֵ֙ינוּ֙ אֶת־כְּבֹד֣וֹ וְאֶת־גׇּדְל֔וֹ וְאֶת־קֹל֥וֹ שָׁמַ֖עְנוּ מִתּ֣וֹךְ הָאֵ֑שׁ הַיּ֤וֹם הַזֶּה֙ רָאִ֔ינוּ כִּֽי־יְדַבֵּ֧ר אֱלֹהִ֛ים אֶת־הָֽאָדָ֖ם וָחָֽי׃ וַאֲמַרְתֶּם, הֲרֵי ה' אֱלֹהֵינוּ הֶרְאָה לָנוּ אֶת כְּבוֹדוֹ וְאֶת גְּדֻלָּתוֹ, וְשָׁמַעְנוּ אֶת קוֹלוֹ מִתּוֹךְ הָאֵשׁ, הַיּוֹם הַזֶּה רָאִינוּ שֶׁאֶפְשָׁרִי שֶׁאֱלֹהִים יְדַבֵּר אֶל אָדָם וְהוּא יִחְיֶה.





* * *



כא וְעַתָּה֙ לָ֣מָּה נָמ֔וּת כִּ֣י תֹֽאכְלֵ֔נוּ הָאֵ֥שׁ הַגְּדֹלָ֖ה הַזֹּ֑את אִם־יֹסְפִ֣ים ׀ אֲנַ֗חְנוּ לִ֠שְׁמֹ֠עַ אֶת־ק֨וֹל יְהֹוָ֧ה אֱלֹהֵ֛ינוּ ע֖וֹד וָמָֽתְנוּ׃ וְעַכְשָׁיו, לָמָּה שֶׁנָּמוּת – שֶׁתֹּאכַל אוֹתָנוּ הָאֵשׁ הַגְּדוֹלָה הַזֹּאת? אִם נַמְשִׁיךְ עוֹד לִשְׁמוֹעַ אֶת קוֹל ה' אֱלֹהֵינוּ, נָמוּת!





* * *



כב כִּ֣י מִ֣י כׇל־בָּשָׂ֡ר אֲשֶׁ֣ר שָׁמַ֣ע קוֹל֩ אֱלֹהִ֨ים חַיִּ֜ים מְדַבֵּ֧ר מִתּוֹךְ־הָאֵ֛שׁ כָּמֹ֖נוּ וַיֶּֽחִי׃ כִּי מִי מִכָּל בָּשָׂר וָדָם שָׁמַע קוֹל אֱלֹהִים חַיִּים מְדַבֵּר מִתּוֹךְ הָאֵשׁ כָּמוֹנוּ וְנִשְׁאַר בַּחַיִּים?!





* * *



כג קְרַ֤ב אַתָּה֙ וּֽשְׁמָ֔ע אֵ֛ת כׇּל־אֲשֶׁ֥ר יֹאמַ֖ר יְהֹוָ֣ה אֱלֹהֵ֑ינוּ וְאַ֣תְּ ׀ תְּדַבֵּ֣ר אֵלֵ֗ינוּ אֵת֩ כׇּל־אֲשֶׁ֨ר יְדַבֵּ֜ר יְהֹוָ֧ה אֱלֹהֵ֛ינוּ אֵלֶ֖יךָ וְשָׁמַ֥עְנוּ וְעָשִֽׂינוּ׃ גַּשׁ אַתָּה וּשְׁמַע אֶת כָּל מַה שֶּׁה' אֱלֹהֵינוּ יֹאמַר, וְאַתָּה תְּדַבֵּר אֵלֵינוּ אֶת כָּל מַה שֶּׁה' אֱלֹהֵינוּ יְדַבֵּר אֵלֶיךָ, וְנִשְׁמַע וְנַעֲשֶׂה.





* * *



כד וַיִּשְׁמַ֤ע יְהֹוָה֙ אֶת־ק֣וֹל דִּבְרֵיכֶ֔ם בְּדַבֶּרְכֶ֖ם אֵלָ֑י וַיֹּ֨אמֶר יְהֹוָ֜ה אֵלַ֗י שָׁ֠מַ֠עְתִּי אֶת־ק֨וֹל דִּבְרֵ֜י הָעָ֤ם הַזֶּה֙ אֲשֶׁ֣ר דִּבְּר֣וּ אֵלֶ֔יךָ הֵיטִ֖יבוּ כׇּל־אֲשֶׁ֥ר דִּבֵּֽרוּ׃ ה' שָׁמַע אֶת קוֹל דִּבְרֵיכֶם כְּשֶׁדִּבַּרְתֶּם אִיתִּי, וְאָמַר לִי ה': שָׁמַעְתִּי אֶת קוֹל דִּבְרֵי הָעָם הַזֶּה שֶׁדִּבְּרוּ אִתְּךָ, וְכָל הַדְּבָרִים שֶׁדִּבְּרוּ טוֹבִים.





* * *



כה מִֽי־יִתֵּ֡ן וְהָיָה֩ לְבָבָ֨ם זֶ֜ה לָהֶ֗ם לְיִרְאָ֥ה אֹתִ֛י וְלִשְׁמֹ֥ר אֶת־כׇּל־מִצְוֺתַ֖י כׇּל־הַיָּמִ֑ים לְמַ֨עַן יִיטַ֥ב לָהֶ֛ם וְלִבְנֵיהֶ֖ם לְעֹלָֽם׃ הַלְוַאי שֶׁרְצוֹנָם זֶה – לְכַבֵּד אוֹתִי וְלִשְׁמוֹר אֶת כָּל מִצְווֹתַי, יִשָּׁאֵר אֶצְלָם כָּל הַיָּמִים, כְּדֵי שֶׁיִּיטַב לָהֶם וְלִבְנֵיהֶם לְעוֹלָם.





* * *



כו לֵ֖ךְ אֱמֹ֣ר לָהֶ֑ם שׁ֥וּבוּ לָכֶ֖ם לְאׇהֳלֵיכֶֽם׃ לֵךְ אֱמוֹר לָהֶם חִזְרוּ לָכֶם לְאָהֳלֵיכֶם.





* * *



כז וְאַתָּ֗ה פֹּה֮ עֲמֹ֣ד עִמָּדִי֒ וַאֲדַבְּרָ֣ה אֵלֶ֗יךָ אֵ֧ת כׇּל־הַמִּצְוָ֛ה וְהַחֻקִּ֥ים וְהַמִּשְׁפָּטִ֖ים אֲשֶׁ֣ר תְּלַמְּדֵ֑ם וְעָשׂ֣וּ בָאָ֔רֶץ אֲשֶׁ֧ר אָנֹכִ֛י נֹתֵ֥ן לָהֶ֖ם לְרִשְׁתָּֽהּ׃ וְאַתָּה עֲמוֹד פֹּה אִתִּי, וְאוֹמַר לְךָ אֶת כָּל הַמִּצְווֹת וְהַחֻקִּים וְהַמִּשְׁפָּטִים שֶׁתְּלַמֵּד אוֹתָם, שֶׁיַּעֲשׂוּ אוֹתָם בָּאָרֶץ שֶׁאֲנִי נוֹתֵן לָהֶם לִכְבּוֹשׁ אוֹתָהּ.





* * *



כח וּשְׁמַרְתֶּ֣ם לַעֲשׂ֔וֹת כַּאֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵיכֶ֖ם אֶתְכֶ֑ם לֹ֥א תָסֻ֖רוּ יָמִ֥ין וּשְׂמֹֽאל׃ הַקְפִּידוּ לַעֲשׂוֹת כְּמוֹ שֶׁה' אֱלֹהֵיכֶם צִוָּה אֶתְכֶם, אַל תִּסְטוּ לְיָמִין אוֹ לִשְׂמֹאל.





* * *



כט בְּכׇל־הַדֶּ֗רֶךְ אֲשֶׁ֨ר צִוָּ֜ה יְהֹוָ֧ה אֱלֹהֵיכֶ֛ם אֶתְכֶ֖ם תֵּלֵ֑כוּ לְמַ֤עַן תִּֽחְיוּן֙ וְט֣וֹב לָכֶ֔ם וְהַאֲרַכְתֶּ֣ם יָמִ֔ים בָּאָ֖רֶץ אֲשֶׁ֥ר תִּֽירָשֽׁוּן׃ לְכוּ בְּכָל הַדֶּרֶךְ שֶׁה' אֱלֹהֵיכֶם צִוָּה אֶתְכֶם, כְּדֵי שֶׁתִּחְיוּ וְיִהְיֶה לָכֶם טוֹב, וְחַיֵּיכֶם יִתְאָרְכוּ בָּאָרֶץ שֶׁתִּכְבְּשׁוּ.





* * *



פ





* * *



ו א וְזֹ֣את הַמִּצְוָ֗ה הַֽחֻקִּים֙ וְהַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵיכֶ֖ם לְלַמֵּ֣ד אֶתְכֶ֑ם לַעֲשׂ֣וֹת בָּאָ֔רֶץ אֲשֶׁ֥ר אַתֶּ֛ם עֹבְרִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ אֵלֶּה הַמִּצְווֹת הַחֻקִּים וְהַמִּשְׁפָּטִים שֶׁה' אֱלֹהֵיכֶם צִוָּה אוֹתִי לְלַמֵּד אֶתְכֶם שֶׁתְּקַיְּמוּ אוֹתָם בָּאָרֶץ שֶׁאַתֶּם עוֹבְרִים לְשָׁם לִכְבּוֹשׁ אוֹתָהּ.





* * *



ב לְמַ֨עַן תִּירָ֜א אֶת־יְהֹוָ֣ה אֱלֹהֶ֗יךָ לִ֠שְׁמֹ֠ר אֶת־כׇּל־חֻקֹּתָ֣יו וּמִצְוֺתָיו֮ אֲשֶׁ֣ר אָנֹכִ֣י מְצַוֶּ֒ךָ֒ אַתָּה֙ וּבִנְךָ֣ וּבֶן־בִּנְךָ֔ כֹּ֖ל יְמֵ֣י חַיֶּ֑יךָ וּלְמַ֖עַן יַאֲרִכֻ֥ן יָמֶֽיךָ׃ כְּדֵי שֶׁתְּכַבֵּד אֶת ה' אֱלֹהֶיךָ וְתַקְפִּיד לְקַיֵּם אֶת כָּל חֻקָּיו וּמִצְווֹתָיו שֶׁאֲנִי מְצַוֶּה אוֹתְךָ, אַתָּה וּבִנְךָ וְנֶכְדְּךָ כָּל יְמֵי חַיֶּיךָ, וּכְדֵי שֶׁיִּתְאָרְכוּ יְמֵי חַיֶּיךָ.





* * *



ג וְשָׁמַעְתָּ֤ יִשְׂרָאֵל֙ וְשָׁמַרְתָּ֣ לַעֲשׂ֔וֹת אֲשֶׁר֙ יִיטַ֣ב לְךָ֔ וַאֲשֶׁ֥ר תִּרְבּ֖וּן מְאֹ֑ד כַּאֲשֶׁר֩ דִּבֶּ֨ר יְהֹוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֙יךָ֙ לָ֔ךְ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ שְׁמַע אוֹתָם, עַם יִשְׂרָאֵל, וְהַקְפֵּד לַעֲשׂוֹת אוֹתָם, כְּדֵי שֶׁיִּהְיֶה לְךָ טוֹב וְשֶׁתִּתְרַבּוּ מְאוֹד, כְּמוֹ שֶׁה' אֱלֹהֵי אֲבוֹתֶיךָ הִבְטִיחַ לְךָ – אֶרֶץ שׁוֹפַעַת חָלָב וּדְבָשׁ.

ששי


פ פָּרָשַׁת שְׁמַע





* * *



ד שְׁמַ֖ע יִשְׂרָאֵ֑ל יְהֹוָ֥ה אֱלֹהֵ֖ינוּ יְהֹוָ֥ה ׀ אֶחָֽד׃ שְׁמַע יִשְׂרָאֵל, ה' הוּא אֱלֹהֵינוּ, ה' הוּא הַיָּחִיד.





* * *



ה וְאָ֣הַבְתָּ֔ אֵ֖ת יְהֹוָ֣ה אֱלֹהֶ֑יךָ בְּכׇל־לְבָבְךָ֥ וּבְכׇל־נַפְשְׁךָ֖ וּבְכׇל־מְאֹדֶֽךָ׃ אֱהוֹב אֶת ה' אֱלֹהֶיךָ בְּכָל לִבְּךָ וּבְכָל רְצוֹנְךָ וּבְכָל רְכוּשְׁךָ.





* * *



ו וְהָי֞וּ הַדְּבָרִ֣ים הָאֵ֗לֶּה אֲשֶׁ֨ר אָנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם עַל־לְבָבֶֽךָ׃ הַדְּבָרִים הָאֵלֶּה שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם יִהְיוּ שְׁמוּרִים בְּלִבְּךָ.





* * *



ז וְשִׁנַּנְתָּ֣ם לְבָנֶ֔יךָ וְדִבַּרְתָּ֖ בָּ֑ם בְּשִׁבְתְּךָ֤ בְּבֵיתֶ֙ךָ֙ וּבְלֶכְתְּךָ֣ בַדֶּ֔רֶךְ וּֽבְשׇׁכְבְּךָ֖ וּבְקוּמֶֽךָ׃ שַׁנֵּן אוֹתָם לְבָנֶיךָ וְדַבֵּר בָּהֶם כְּשֶׁאַתָּה יוֹשֵׁב בְּבֵיתְךָ וּכְשֶׁאַתָּה הוֹלֵךְ בַּדֶּרֶךְ וּכְשֶׁאַתָּה שׁוֹכֵב וּכְשֶׁאַתָּה קָם.





* * *



ח וּקְשַׁרְתָּ֥ם לְא֖וֹת עַל־יָדֶ֑ךָ וְהָי֥וּ לְטֹטָפֹ֖ת בֵּ֥ין עֵינֶֽיךָ׃ קְשׁוֹר אוֹתָם לְסִימָן עַל יָדְךָ, וְיִהְיוּ תַּכְשִׁיט בֵּין עֵינֶיךָ.





* * *



ט וּכְתַבְתָּ֛ם עַל־מְזֻז֥וֹת בֵּיתֶ֖ךָ וּבִשְׁעָרֶֽיךָ׃ כְּתוֹב אוֹתָם עַל מְזוּזוֹת בֵּיתְךָ, וּבִשְׁעָרֶיךָ.





* * *



ס פָּרָשַׁת הַזְהָרַת עֲבוֹדָה זָרָה





* * *



י וְהָיָ֞ה כִּ֥י יְבִיאֲךָ֣ ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ אֶל־הָאָ֜רֶץ אֲשֶׁ֨ר נִשְׁבַּ֧ע לַאֲבֹתֶ֛יךָ לְאַבְרָהָ֛ם לְיִצְחָ֥ק וּֽלְיַעֲקֹ֖ב לָ֣תֶת לָ֑ךְ עָרִ֛ים גְּדֹלֹ֥ת וְטֹבֹ֖ת אֲשֶׁ֥ר לֹא־בָנִֽיתָ׃ כְּשֶׁה' אֱלֹהֶיךָ יָבִיא אוֹתְךָ לָאָרֶץ שֶׁהוּא נִשְׁבַּע לַאֲבוֹתֶיךָ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לָתֵת לְךָ – עָרִים גְּדוֹלוֹת וְטוֹבוֹת שֶׁלֹּא בָּנִיתָ.





* * *



יא וּבָ֨תִּ֜ים מְלֵאִ֣ים כׇּל־טוּב֮ אֲשֶׁ֣ר לֹא־מִלֵּ֒אתָ֒ וּבֹרֹ֤ת חֲצוּבִים֙ אֲשֶׁ֣ר לֹא־חָצַ֔בְתָּ כְּרָמִ֥ים וְזֵיתִ֖ים אֲשֶׁ֣ר לֹא־נָטָ֑עְתָּ וְאָכַלְתָּ֖ וְשָׂבָֽעְתָּ׃ וּבָתִּים מְלֵאִים כָּל טוּב שֶׁלֹּא מִלֵּאתָ וּבוֹרוֹת חֲצוּבִים שֶׁלֹּא חָצַבְתָּ, כְּרָמִים וְזֵיתִים שֶׁלֹּא נָטַעְתָּ, וְתֹאכַל וְתִשְׂבַּע.





* * *



יב הִשָּׁ֣מֶר לְךָ֔ פֶּן־תִּשְׁכַּ֖ח אֶת־יְהֹוָ֑ה אֲשֶׁ֧ר הוֹצִֽיאֲךָ֛ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ הִזָּהֵר לְךָ שֶׁלֹּא תִּשְׁכַּח אֶת ה' שֶׁהוֹצִיא אוֹתְךָ מֵאֶרֶץ מִצְרַיִם מִבֵּית עֲבָדִים.





* * *



יג אֶת־יְהֹוָ֧ה אֱלֹהֶ֛יךָ תִּירָ֖א וְאֹת֣וֹ תַעֲבֹ֑ד וּבִשְׁמ֖וֹ תִּשָּׁבֵֽעַ׃ כַּבֵּד אֶת ה' אֱלֹהֶיךָ וְאוֹתוֹ תַּעֲבוֹד וְרַק בִּשְׁמוֹ תִּשָּׁבַע.





* * *



יד לֹ֣א תֵֽלְכ֔וּן אַחֲרֵ֖י אֱלֹהִ֣ים אֲחֵרִ֑ים מֵאֱלֹהֵי֙ הָֽעַמִּ֔ים אֲשֶׁ֖ר סְבִיבוֹתֵיכֶֽם׃ אַל תֵּלְכוּ אַחֲרֵי אֱלֹהִים אֲחֵרִים מֵאֱלֹהֵי הָעַמִּים שֶׁסְּבִיבוֹתֵיכֶם לַעֲבוֹד אוֹתָם.





* * *



טו כִּ֣י אֵ֥ל קַנָּ֛א יְהֹוָ֥ה אֱלֹהֶ֖יךָ בְּקִרְבֶּ֑ךָ פֶּן־יֶ֠חֱרֶ֠ה אַף־יְהֹוָ֤ה אֱלֹהֶ֙יךָ֙ בָּ֔ךְ וְהִשְׁמִ֣ידְךָ֔ מֵעַ֖ל פְּנֵ֥י הָאֲדָמָֽה׃ כִּי ה' אֱלֹהֶיךָ שֶׁנִּמְצָא בְּתוֹכְךָ הוּא אֵל נוֹקֵם – שֶׁה' אֱלֹהֶיךָ לֹא יִכְעַס עָלֶיךָ וְיַשְׁמִיד אוֹתְךָ מֵעַל פְּנֵי הָאֲדָמָה.





* * *



ס





* * *



טז לֹ֣א תְנַסּ֔וּ אֶת־יְהֹוָ֖ה אֱלֹהֵיכֶ֑ם כַּאֲשֶׁ֥ר נִסִּיתֶ֖ם בַּמַּסָּֽה׃ אַל תְּנַסּוּ אֶת ה' אֱלֹהֵיכֶם כְּמוֹ שֶׁנִּסִּיתֶם בְּמַסָּה וּמְרִיבָה, שֶׁאֲמַרְתֶּם: "הַאִם יֵשׁ ה' בְּתוֹכֵנוּ אוֹ לֹא".





* * *



יז שָׁמ֣וֹר תִּשְׁמְר֔וּן אֶת־מִצְוֺ֖ת יְהֹוָ֣ה אֱלֹהֵיכֶ֑ם וְעֵדֹתָ֥יו וְחֻקָּ֖יו אֲשֶׁ֥ר צִוָּֽךְ׃ הַקְפִּידוּ לְקַיֵּם אֶת מִצְווֹת ה' אֱלֹהֵיכֶם וְאֶת אַזְהֲרוֹתָיו וְחֻקָּיו שֶׁצִּוָּה אוֹתְךָ.





* * *



יח וְעָשִׂ֛יתָ הַיָּשָׁ֥ר וְהַטּ֖וֹב בְּעֵינֵ֣י יְהֹוָ֑ה לְמַ֙עַן֙ יִ֣יטַב לָ֔ךְ וּבָ֗אתָ וְיָֽרַשְׁתָּ֙ אֶת־הָאָ֣רֶץ הַטֹּבָ֔ה אֲשֶׁר־נִשְׁבַּ֥ע יְהֹוָ֖ה לַאֲבֹתֶֽיךָ׃ עֲשֵׂה אֶת הַיָּשָׁר וְהַטּוֹב בְּעֵינֵי ה' כְּדֵי שֶׁיִּיטַב לְךָ, שֶׁתָּבוֹא וְתִכְבּוֹשׁ אֶת הָאָרֶץ הַטּוֹבָה שֶׁה' נִשְׁבַּע לַאֲבוֹתֶיךָ.





* * *



יט לַהֲדֹ֥ף אֶת־כׇּל־אֹיְבֶ֖יךָ מִפָּנֶ֑יךָ כַּאֲשֶׁ֖ר דִּבֶּ֥ר יְהֹוָֽה׃ וְתַהֲדוֹף אֶת כָּל אוֹיְבֶיךָ מִפָּנֶיךָ כְּמוֹ שֶׁה' הִבְטִיחַ.





* * *



ס





* * *



כ כִּֽי־יִשְׁאָלְךָ֥ בִנְךָ֛ מָחָ֖ר לֵאמֹ֑ר מָ֣ה הָעֵדֹ֗ת וְהַֽחֻקִּים֙ וְהַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר צִוָּ֛ה יְהֹוָ֥ה אֱלֹהֵ֖ינוּ אֶתְכֶֽם׃ אִם בִּנְךָ יִשְׁאַל אוֹתְךָ בֶּעָתִיד: לְשֵׁם מָה הָאַזְהָרוֹת הַחֻקִּים וְהַמִּשְׁפָּטִים שֶׁה' אֱלֹהֵינוּ צִוָּה אֶתְכֶם?





* * *



כא וְאָמַרְתָּ֣ לְבִנְךָ֔ עֲבָדִ֛ים הָיִ֥ינוּ לְפַרְעֹ֖ה בְּמִצְרָ֑יִם וַיֹּצִיאֵ֧נוּ יְהֹוָ֛ה מִמִּצְרַ֖יִם בְּיָ֥ד חֲזָקָֽה׃ אֱמוֹר לְבִנְךָ: הָיִינוּ עֲבָדִים לְפַרְעֹה בְּמִצְרַיִם, וְה' הוֹצִיא אוֹתָנוּ מִמִּצְרַיִם בְּיָד חֲזָקָה.





* * *



כב וַיִּתֵּ֣ן יְהֹוָ֡ה אוֹתֹ֣ת וּ֠מֹפְתִ֠ים גְּדֹלִ֨ים וְרָעִ֧ים ׀ בְּמִצְרַ֛יִם בְּפַרְעֹ֥ה וּבְכׇל־בֵּית֖וֹ לְעֵינֵֽינוּ׃ ה' עָשָׂה לְעֵינֵינוּ סִימָנִים וְהוֹכָחוֹת גְּדוֹלִים וְרָעִים בְּמִצְרַיִם – בְּפַרְעֹה וּבְכָל בֵּיתוֹ.





* * *



כג וְאוֹתָ֖נוּ הוֹצִ֣יא מִשָּׁ֑ם לְמַ֙עַן֙ הָבִ֣יא אֹתָ֔נוּ לָ֤תֶת לָ֙נוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖ע לַאֲבֹתֵֽינוּ׃ וְאוֹתָנוּ הוֹצִיא מִשָּׁם כְּדֵי לְהָבִיא אוֹתָנוּ לָתֵת לָנוּ אֶת הָאָרֶץ שֶׁנִּשְׁבַּע לַאֲבוֹתֵינוּ.





* * *



כד וַיְצַוֵּ֣נוּ יְהֹוָ֗ה לַעֲשׂוֹת֙ אֶת־כׇּל־הַחֻקִּ֣ים הָאֵ֔לֶּה לְיִרְאָ֖ה אֶת־יְהֹוָ֣ה אֱלֹהֵ֑ינוּ לְט֥וֹב לָ֙נוּ֙ כׇּל־הַיָּמִ֔ים לְחַיֹּתֵ֖נוּ כְּהַיּ֥וֹם הַזֶּֽה׃ ה' צִוָּה אוֹתָנוּ לַעֲשׂוֹת אֶת כָּל הַחֻקִּים הָאֵלֶּה – לְכַבֵּד אֶת ה' אֱלֹהֵינוּ, כְּדֵי לְהֵטִיב לָנוּ כָּל הַיָּמִים, לְהַחֲיוֹת אוֹתָנוּ כְּמוֹ הַיּוֹם הַזֶּה.





* * *



כה וּצְדָקָ֖ה תִּֽהְיֶה־לָּ֑נוּ כִּֽי־נִשְׁמֹ֨ר לַעֲשׂ֜וֹת אֶת־כׇּל־הַמִּצְוָ֣ה הַזֹּ֗את לִפְנֵ֛י יְהֹוָ֥ה אֱלֹהֵ֖ינוּ כַּאֲשֶׁ֥ר צִוָּֽנוּ׃ וְתִהְיֶה לָנוּ זְכוּת בְּכָךְ שֶׁנַּקְפִּיד לְקַיֵּם אֶת כָּל הַמִּצְווֹת הָאֵלֶּה לִפְנֵי ה' אֱלֹהֵינוּ כְּמוֹ שֶׁהוּא צִוָּה אוֹתָנוּ.





* * *





שביעי


ס פָּרָשַׁת הַזְהָרַת שִׁבְעַת הָעַמִּים





* * *



ז א כִּ֤י יְבִֽיאֲךָ֙ יְהֹוָ֣ה אֱלֹהֶ֔יךָ אֶל־הָאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּ֑הּ וְנָשַׁ֣ל גּֽוֹיִם־רַבִּ֣ים ׀ מִפָּנֶ֡יךָ הַֽחִתִּי֩ וְהַגִּרְגָּשִׁ֨י וְהָאֱמֹרִ֜י וְהַכְּנַעֲנִ֣י וְהַפְּרִזִּ֗י וְהַֽחִוִּי֙ וְהַיְבוּסִ֔י שִׁבְעָ֣ה גוֹיִ֔ם רַבִּ֥ים וַעֲצוּמִ֖ים מִמֶּֽךָּ׃ כְּשֶׁה' אֱלֹהֶיךָ יָבִיא אוֹתְךָ לָאָרֶץ שֶׁאַתָּה בָּא לְשָׁם לִכְבּוֹשׁ אוֹתָהּ, וִיגָרֵשׁ מִפָּנֶיךָ עַמִּים רַבִּים – אֶת הַחִתִּי הַגִּרְגָּשִׁי הָאֱמוֹרִי הַכְּנַעֲנִי הַפְּרִזִּי הַחִוִּי וְהַיְּבוּסִי, שִׁבְעָה עַמִּים גְּדוֹלִים וַחֲזָקִים מִמְּךָ.





* * *



ב וּנְתָנָ֞ם יְהֹוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ וְהִכִּיתָ֑ם הַחֲרֵ֤ם תַּחֲרִים֙ אֹתָ֔ם לֹא־תִכְרֹ֥ת לָהֶ֛ם בְּרִ֖ית וְלֹ֥א תְחׇנֵּֽם׃ וַה' אֱלֹהֶיךָ יִתֵּן אוֹתָם לְפָנֶיךָ וְתָבִיס אוֹתָם – הַשְׁמֵד אוֹתָם! אַל תַּעֲשֶׂה אִתָּם הֶסְכֵּם וְאַל תְּרַחֵם עֲלֵיהֶם.





* * *



ג וְלֹ֥א תִתְחַתֵּ֖ן בָּ֑ם בִּתְּךָ֙ לֹא־תִתֵּ֣ן לִבְנ֔וֹ וּבִתּ֖וֹ לֹא־תִקַּ֥ח לִבְנֶֽךָ׃ אַל תִּתְחַתֵּן אִתָּם – אֶת בִּתְּךָ אַל תִּתֵּן לִבְנוֹ וְאֶת בִּתּוֹ אַל תִּקַּח לְבִנְךָ.





* * *



ד כִּֽי־יָסִ֤יר אֶת־בִּנְךָ֙ מֵֽאַחֲרַ֔י וְעָבְד֖וּ אֱלֹהִ֣ים אֲחֵרִ֑ים וְחָרָ֤ה אַף־יְהֹוָה֙ בָּכֶ֔ם וְהִשְׁמִידְךָ֖ מַהֵֽר׃ שֶׁלֹּא יָסִיט אֶת בִּנְךָ מִלָּלֶכֶת אַחֲרַי לַעֲבוֹד אוֹתִי, וְיַעַבְדוּ אֱלֹהִים אֲחֵרִים, וַה' יִכְעַס עֲלֵיכֶם וְיַשְׁמִיד אוֹתְךָ מַהֵר.





* * *



ה כִּֽי־אִם־כֹּ֤ה תַעֲשׂוּ֙ לָהֶ֔ם מִזְבְּחֹתֵיהֶ֣ם תִּתֹּ֔צוּ וּמַצֵּבֹתָ֖ם תְּשַׁבֵּ֑רוּ וַאֲשֵֽׁירֵהֶם֙ תְּגַדֵּע֔וּן וּפְסִילֵיהֶ֖ם תִּשְׂרְפ֥וּן בָּאֵֽשׁ׃ אֶלָּא כָּךְ תַּעֲשׂוּ לָהֶם: תְּנַתְּצוּ אֶת מִזְבְּחוֹתֵיהֶם וְתִשְׁבְּרוּ אֶת מַצֵּבוֹתֵיהֶם וְתִגְדְּעוּ אֶת הָעֵצִים שֶׁהֵם עוֹבְדִים וְתִשְׂרְפוּ בָּאֵשׁ אֶת פִּסְלֵיהֶם.





* * *



ו כִּ֣י עַ֤ם קָדוֹשׁ֙ אַתָּ֔ה לַיהֹוָ֖ה אֱלֹהֶ֑יךָ בְּךָ֞ בָּחַ֣ר ׀ יְהֹוָ֣ה אֱלֹהֶ֗יךָ לִהְי֥וֹת לוֹ֙ לְעַ֣ם סְגֻלָּ֔ה מִכֹּל֙ הָֽעַמִּ֔ים אֲשֶׁ֖ר עַל־פְּנֵ֥י הָאֲדָמָֽה׃ כִּי אַתָּה עַם קָדוֹשׁ לַה' אֱלֹהֶיךָ, בְּךָ ה' אֱלֹהֶיךָ בָּחַר מִכָּל הָעַמִּים שֶׁעַל פְּנֵי הָאֲדָמָה שֶׁתִּהְיֶה לוֹ הָעָם הֶחָבִיב.





* * *



ז לֹ֣א מֵֽרֻבְּכֶ֞ם מִכׇּל־הָֽעַמִּ֗ים חָשַׁ֧ק יְהֹוָ֛ה בָּכֶ֖ם וַיִּבְחַ֣ר בָּכֶ֑ם כִּֽי־אַתֶּ֥ם הַמְעַ֖ט מִכׇּל־הָעַמִּֽים׃ ה' לֹא חָשַׁק בָּכֶם וּבָחַר בָּכֶם מִפְּנֵי שֶׁאַתֶּם מְרֻבִּים מִכָּל הָעַמִּים, שֶׁהֲרֵי אַתֶּם דַּוְקָא הַמְּעַטִּים מִכָּל הָעַמִּים.





* * *



ח כִּי֩ מֵאַֽהֲבַ֨ת יְהֹוָ֜ה אֶתְכֶ֗ם וּמִשׇּׁמְר֤וֹ אֶת־הַשְּׁבֻעָה֙ אֲשֶׁ֤ר נִשְׁבַּע֙ לַאֲבֹ֣תֵיכֶ֔ם הוֹצִ֧יא יְהֹוָ֛ה אֶתְכֶ֖ם בְּיָ֣ד חֲזָקָ֑ה וַֽיִּפְדְּךָ֙ מִבֵּ֣ית עֲבָדִ֔ים מִיַּ֖ד פַּרְעֹ֥ה מֶֽלֶךְ־מִצְרָֽיִם׃ אֶלָּא מִתּוֹךְ אַהֲבַת ה' אֶתְכֶם וּמֵהַקְפָּדָתוֹ לְקַיֵּם אֶת הַשְּׁבוּעָה שֶׁנִּשְׁבַּע לַאֲבוֹתֵיכֶם – ה' הוֹצִיא אֶתְכֶם בְּיָד חֲזָקָה וּפָדָה אוֹתְךָ מִבֵּית עֲבָדִים מִיַּד פַּרְעֹה מֶלֶךְ מִצְרַיִם.





* * *





מפטיר


ט וְיָ֣דַעְתָּ֔ כִּֽי־יְהֹוָ֥ה אֱלֹהֶ֖יךָ ה֣וּא הָֽאֱלֹהִ֑ים הָאֵל֙ הַֽנֶּאֱמָ֔ן שֹׁמֵ֧ר הַבְּרִ֣ית וְהַחֶ֗סֶד לְאֹהֲבָ֛יו וּלְשֹׁמְרֵ֥י מִצְוֺתָ֖ו לְאֶ֥לֶף דּֽוֹר׃ דַּע שֶׁה' אֱלֹהֶיךָ הוּא הָאֵל הַשּׁוֹלֵט, הָאֵל הַנֶּאֱמָן – מְקַיֵּם הַבְטָחוֹת וַחֲסָדִים לְאוֹהֲבָיו וּלְשׁוֹמְרֵי מִצְווֹתָיו לְאֶלֶף דּוֹרוֹת.





* * *



י וּמְשַׁלֵּ֧ם לְשֹׂנְאָ֛יו אֶל־פָּנָ֖יו לְהַאֲבִיד֑וֹ לֹ֤א יְאַחֵר֙ לְשֹׂ֣נְא֔וֹ אֶל־פָּנָ֖יו יְשַׁלֶּם־לֽוֹ׃ וּמְשַׁלֵּם לְשׂוֹנְאָיו אֶת שְׂכָרוֹ הַמּוּעָט בְּחַיָּיו כְּדֵי לְהַאֲבִיד אוֹתוֹ מֵהָעוֹלָם הַבָּא, הוּא לֹא יִשְׁמוֹר לְשׂוֹנְאוֹ אֶת הַשָּׂכָר לָעוֹלָם הַבָּא, אֶלָּא יְשַׁלֵּם לוֹ בְּחַיָּיו.





* * *



יא וְשָׁמַרְתָּ֨ אֶת־הַמִּצְוָ֜ה וְאֶת־הַֽחֻקִּ֣ים וְאֶת־הַמִּשְׁפָּטִ֗ים אֲשֶׁ֨ר אָנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם לַעֲשׂוֹתָֽם׃ הַקְפֵּד לְקַיֵּם אֶת הַמִּצְווֹת וְאֶת הַחֻקִּים וְאֶת הַמִּשְׁפָּטִים שֶׁאֲנִי מְצַוֶּה אוֹתְךָ הַיּוֹם לַעֲשׂוֹת אוֹתָם.
