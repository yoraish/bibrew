פָּרָשַׁת בְּמִדְבַּר

קְרִיאַת הַתּוֹרָה וְהַהַפְטָרָה · מִקְרָא וְתַרְגּוּם · שְׁנַֽיִם מִקְרָא וְאֶחָד תַּרְגּוּם

מִקְרָא וְתַרְגּוּם וְרַשִׁ"י (בהכנה) · מִקְרָא מְבוֹאָר





ראשון


פָּרָשַׁת פְּקוּדֵי יִשְׂרָאֵל





* * *



א א וַיְדַבֵּ֨ר יְהֹוָ֧ה אֶל־מֹשֶׁ֛ה בְּמִדְבַּ֥ר סִינַ֖י בְּאֹ֣הֶל מוֹעֵ֑ד בְּאֶחָד֩ לַחֹ֨דֶשׁ הַשֵּׁנִ֜י בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית לְצֵאתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בְּמִדְבַּר סִינַי בְּאוֹהֶל הַהִתְכַּנְּסוּת, בַּיּוֹם הָרִאשׁוֹן שֶׁל הַחוֹדֶשׁ הַשֵּׁנִי בַּשָּׁנָה הַשְּׁנִיָּה לְצֵאתָם מֵאֶרֶץ מִצְרַיִם, וְאָמַר:





* * *



ב שְׂא֗וּ אֶת־רֹאשׁ֙ כׇּל־עֲדַ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֔וֹת כׇּל־זָכָ֖ר לְגֻלְגְּלֹתָֽם׃ סִפְרוּ אֶת כָּל הַצִּבּוּר שֶׁל בְּנֵי יִשְׂרָאֵל לְמִשְׁפְּחוֹתֵיהֶם בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם, כָּל אֶחָד בִּשְׁמוֹ, כָּל אִישׁ זָכָר.





* * *



ג מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כׇּל־יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵ֑ל תִּפְקְד֥וּ אֹתָ֛ם לְצִבְאֹתָ֖ם אַתָּ֥ה וְאַהֲרֹֽן׃ מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה – כָּל מִתְגַּיֵּס לַצָּבָא בְּיִשְׂרָאֵל, אַתָּה וְאַהֲרֹן תִּסְפְּרוּ אוֹתָם בִּשְׁבִיל הַצָּבָא.





* * *



ד וְאִתְּכֶ֣ם יִהְי֔וּ אִ֥ישׁ אִ֖ישׁ לַמַּטֶּ֑ה אִ֛ישׁ רֹ֥אשׁ לְבֵית־אֲבֹתָ֖יו הֽוּא׃ וְאִתְּכֶם יִהְיוּ אִישׁ אֶחָד מִכָּל שֵׁבֶט – וְהוּא הַמַּנְהִיג שֶׁל מִשְׁפְּחוֹתָיו.





* * *



ה וְאֵ֙לֶּה֙ שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁ֥ר יַֽעַמְד֖וּ אִתְּכֶ֑ם לִרְאוּבֵ֕ן אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ וְאֵלֶּה שְׁמוֹת הָאֲנָשִׁים שֶׁיַּעַמְדוּ אִתְּכֶם: לִרְאוּבֵן – אֱלִיצוּר בֶּן שְׁדֵיאוּר.





* * *



ו לְשִׁמְע֕וֹן שְׁלֻמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ לְשִׁמְעוֹן – שְׁלוּמִיאֵל בֶּן צוּרִישַׁדָּי.





* * *



ז לִֽיהוּדָ֕ה נַחְשׁ֖וֹן בֶּן־עַמִּינָדָֽב׃ לִיהוּדָה – נַחְשׁוֹן בֶּן עַמִּינָדָב.





* * *



ח לְיִ֨שָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ לְיִשָּׂשכָר – נְתַנְאֵל בֶּן צוּעָר.





* * *



ט לִזְבוּלֻ֕ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ לִזְבוּלוּן – אֱלִיאָב בֶּן חֵלוֹן.





* * *



י לִבְנֵ֣י יוֹסֵ֔ף לְאֶפְרַ֕יִם אֱלִישָׁמָ֖ע בֶּן־עַמִּיה֑וּד לִמְנַשֶּׁ֕ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ לִבְנֵי יוֹסֵף: לְאֶפְרַיִם – אֱלִישָׁמָע בֶּן עַמִּיהוּד, לִמְנַשֶּׁה – גַּמְלִיאֵל בֶּן פְּדָהצוּר.





* * *



יא לְבִ֨נְיָמִ֔ן אֲבִידָ֖ן בֶּן־גִּדְעֹנִֽי׃ לְבִנְיָמִין – אֲבִידָן בֶּן גִּדְעוֹנִי.





* * *



יב לְדָ֕ן אֲחִיעֶ֖זֶר בֶּן־עַמִּֽישַׁדָּֽי׃ לְדָן – אֲחִיעֶזֶר בֶּן עַמִּישַׁדָּי.





* * *



יג לְאָשֵׁ֕ר פַּגְעִיאֵ֖ל בֶּן־עׇכְרָֽן׃ לְאָשֵׁר – פַּגְעִיאֵל בֶּן עָכְרָן.





* * *



יד לְגָ֕ד אֶלְיָסָ֖ף בֶּן־דְּעוּאֵֽל׃ לְגָד – אֶלְיָסָף בֶּן דְּעוּאֵל.





* * *



טו לְנַ֨פְתָּלִ֔י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ לְנַפְתָּלִי – אֲחִירַע בֶּן עֵינָן.





* * *



טז אֵ֚לֶּה קריאי קְרוּאֵ֣י הָעֵדָ֔ה נְשִׂיאֵ֖י מַטּ֣וֹת אֲבוֹתָ֑ם רָאשֵׁ֛י אַלְפֵ֥י יִשְׂרָאֵ֖ל הֵֽם׃ אֵלֶּה הֵם הַמֻּזְמָנִים מֵהַצִּבּוּר לִסְפּוֹר – נְשִׂיאֵי שִׁבְטֵי אֲבוֹתֵיהֶם, הֵם רָאשֵׁי אַלְפֵי בְּנֵי יִשְׂרָאֵל.





* * *



יז וַיִּקַּ֥ח מֹשֶׁ֖ה וְאַהֲרֹ֑ן אֵ֚ת הָאֲנָשִׁ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר נִקְּב֖וּ בְּשֵׁמֹֽת‏[1]׃ מֹשֶׁה וְאַהֲרֹן לָקְחוּ אִתָּם אֶת הָאֲנָשִׁים הָאֵלֶּה שֶׁשְּׁמוֹתֵיהֶם הֻזְכְּרוּ.





* * *



יח וְאֵ֨ת כׇּל־הָעֵדָ֜ה הִקְהִ֗ילוּ בְּאֶחָד֙ לַחֹ֣דֶשׁ הַשֵּׁנִ֔י וַיִּתְיַֽלְד֥וּ עַל־מִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְגֻלְגְּלֹתָֽם׃ וְאָסְפוּ אֶת כָּל הַצִּבּוּר בַּיּוֹם הָרִאשׁוֹן שֶׁל חוֹדֶשׁ אִיָּר, וְהֵם הִצִּיגוּ אֶת יִחוּסֵי מִשְׁפְּחוֹתֵיהֶם לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם, שְׁמוֹת הָאֲנָשִׁים נִסְפְּרוּ מִגִּיל עֶשְׂרִים וָמַעְלָה, כָּל אֶחָד וְאֶחָד.





* * *



יט כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶׁ֑ה וַֽיִּפְקְדֵ֖ם בְּמִדְבַּ֥ר סִינָֽי׃ הֵם עָשׂוּ כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה, וְהוּא סָפַר אוֹתָם בְּמִדְבַּר סִינַי.





* * *





שני


ס





* * *



כ וַיִּהְי֤וּ בְנֵֽי־רְאוּבֵן֙ בְּכֹ֣ר יִשְׂרָאֵ֔ל תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כׇּל־זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ בְּנֵי רְאוּבֵן בְּכוֹר יִשְׂרָאֵל, צֶאֱצָאֵי מִשְׁפְּחוֹתֵיהֶם בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם, שֶׁנִּסְפְּרוּ כָּל אֶחָד בִּשְׁמוֹ, כָּל זָכָר מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה – כָּל מִתְגַּיֵּס לַצָּבָא –





* * *



כא פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה רְאוּבֵ֑ן שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ מִנְיָנָם שֶׁל שֵׁבֶט רְאוּבֵן הָיָה אַרְבָּעִים וְשִׁשָּׁה אֶלֶף וַחֲמֵשׁ מֵאוֹת.





* * *



פ





* * *



כב לִבְנֵ֣י שִׁמְע֔וֹן תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם פְּקֻדָ֗יו בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כׇּל־זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי שִׁמְעוֹן צֶאֱצָאֵי מִשְׁפְּחוֹתֵיהֶם לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם – מִנְיָנָם שֶׁנִּסְפְּרוּ כָּל אֶחָד בִּשְׁמוֹ, כָּל אִישׁ זָכָר מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה כָּל מִתְגַּיֵּס לַצָּבָא –





* * *



כג פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה שִׁמְע֑וֹן תִּשְׁעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ מִנְיָנָם שֶׁל שֵׁבֶט שִׁמְעוֹן חֲמִשִּׁים וְתִשְׁעָה אֶלֶף וּשְׁלוֹשׁ מֵאוֹת.





* * *



פ





* * *



כד לִבְנֵ֣י גָ֔ד תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי גָּד צֶאֱצָאֵי מִשְׁפְּחוֹתֵיהֶם לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם – מִנְיָנָם שֶׁנִּסְפְּרוּ כָּל אֶחָד בִּשְׁמוֹ, כָּל אִישׁ זָכָר מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה כָּל מִתְגַּיֵּס לַצָּבָא –





* * *



כה פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה גָ֑ד חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים׃ מִנְיָנָם שֶׁל שֵׁבֶט גָּד אַרְבָּעִים וַחֲמִשָּׁה אֶלֶף וְשֵׁשׁ מֵאוֹת וַחֲמִשִּׁים.





* * *



פ





* * *



כו לִבְנֵ֣י יְהוּדָ֔ה תּוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי יְהוּדָה צֶאֱצָאֵי מִשְׁפְּחוֹתֵיהֶם לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם – מִנְיָנָם שֶׁנִּסְפְּרוּ כָּל אֶחָד בִּשְׁמוֹ, כָּל אִישׁ זָכָר מִגִּיל עֶשְׂרִים שָׁנָה וָמַעְלָה כָּל מִתְגַּיֵּס לַצָּבָא –





* * *



כז פְּקֻדֵיהֶ֖ם לְמַטֵּ֣ה יְהוּדָ֑ה אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ מִנְיָנָם שֶׁל שֵׁבֶט יְהוּדָה שִׁבְעִים וְאַרְבָּעָה אֶלֶף וְשֵׁשׁ מֵאוֹת.

מט אַ֣ךְ אֶת־מַטֵּ֤ה לֵוִי֙ לֹ֣א תִפְקֹ֔ד וְאֶת־רֹאשָׁ֖ם לֹ֣א תִשָּׂ֑א בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ אֲבָל אֶת שֵׁבֶט לֵוִי אַל תִּמְנֶה וְאַל תִּסְפּוֹר אוֹתָם יַחַד עִם בְּנֵי יִשְׂרָאֵל.

נ וְאַתָּ֡ה הַפְקֵ֣ד אֶת־הַלְוִיִּם֩ עַל־מִשְׁכַּ֨ן הָעֵדֻ֜ת וְעַ֣ל כׇּל־כֵּלָיו֮ וְעַ֣ל כׇּל־אֲשֶׁר־לוֹ֒ הֵ֜מָּה יִשְׂא֤וּ אֶת־הַמִּשְׁכָּן֙ וְאֶת־כׇּל־כֵּלָ֔יו וְהֵ֖ם יְשָׁרְתֻ֑הוּ וְסָבִ֥יב לַמִּשְׁכָּ֖ן יַחֲנֽוּ׃ וְאַתָּה תְּמַנֶּה אֶת הַלְוִיִּים עַל מִשְׁכַּן לוּחוֹת הַבְּרִית, וְעַל כָּל כֵּלָיו וְעַל כָּל הַדְּבָרִים שֶׁשַּׁיָּכִים אֵלָיו – הֵם יִסְחֲבוּ אֶת הַמִּשְׁכָּן וְאֶת כָּל כֵּלָיו וְהֵם יִהְיוּ הַשָּׁרָתִים שֶׁלּוֹ וְיַחֲנוּ מִסָּבִיב לַמִּשְׁכָּן.





* * *



נא וּבִנְסֹ֣עַ הַמִּשְׁכָּ֗ן יוֹרִ֤ידוּ אֹתוֹ֙ הַלְוִיִּ֔ם וּבַחֲנֹת֙ הַמִּשְׁכָּ֔ן יָקִ֥ימוּ אֹת֖וֹ הַלְוִיִּ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ כְּשֶׁהַמִּשְׁכָּן יִסַּע הַלְוִיִּים יְפָרְקוּ אוֹתוֹ וּכְשֶׁהַמִּשְׁכָּן יַחֲנֶה הַלְוִיִּים יָקִימוּ אוֹתוֹ, וְאִישׁ זָר שֶׁיִּגַּשׁ לַעֲשׂוֹת אֶת זֶה – יוּמַת.





* * *



נב וְחָנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל אִ֧ישׁ עַֽל־מַחֲנֵ֛הוּ וְאִ֥ישׁ עַל־דִּגְל֖וֹ לְצִבְאֹתָֽם׃ וּבְנֵי יִשְׂרָאֵל יַחֲנוּ כָּל אֶחָד בַּמַּחֲנֶה שֶׁלּוֹ – כָּל אִישׁ תַּחַת דִּגְלוֹ לִקְהִלּוֹתֵיהֶם.





* * *



נג וְהַלְוִיִּ֞ם יַחֲנ֤וּ סָבִיב֙ לְמִשְׁכַּ֣ן הָעֵדֻ֔ת וְלֹֽא־יִהְיֶ֣ה קֶ֔צֶף עַל־עֲדַ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְשָׁמְרוּ֙ הַלְוִיִּ֔ם אֶת־מִשְׁמֶ֖רֶת מִשְׁכַּ֥ן הָעֵדֽוּת׃ וְהַלְוִיִּים יַחֲנוּ מִסָּבִיב לְמִשְׁכַּן לוּחוֹת הַבְּרִית וְלֹא יִהְיֶה כַּעַס עַל הַצִּבּוּר שֶׁל בְּנֵי יִשְׂרָאֵל, וְהַלְוִיִּים יִשְׁמְרוּ אֶת מִשְׁמַר מִשְׁכַּן לוּחוֹת הַבְּרִית.





* * *



נד וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹ֠ל אֲשֶׁ֨ר צִוָּ֧ה יְהֹוָ֛ה אֶת־מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ׃ וּבְנֵי יִשְׂרָאֵל בִּצְּעוּ, כְּכָל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה – כָּךְ הֵם עָשׂוּ.





* * *





שלישי


פ פָּרָשַׁת הַדְּגָלִים





* * *



ב א וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:





* * *



ב אִ֣ישׁ עַל־דִּגְל֤וֹ בְאֹתֹת֙ לְבֵ֣ית אֲבֹתָ֔ם יַחֲנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל מִנֶּ֕גֶד סָבִ֥יב לְאֹֽהֶל־מוֹעֵ֖ד יַחֲנֽוּ׃ בְּנֵי יִשְׂרָאֵל יַחֲנוּ כָּל אֶחָד לְיַד דֶּגֶל שִׁבְטוֹ וְסִמְלוֹ לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם, הֵם יַחֲנוּ מוּל אוֹהֶל הַהִתְכַּנְּסוּת מִסָּבִיב.





* * *



ג וְהַחֹנִים֙ קֵ֣דְמָה מִזְרָ֔חָה דֶּ֛גֶל מַחֲנֵ֥ה יְהוּדָ֖ה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י יְהוּדָ֔ה נַחְשׁ֖וֹן בֶּן־עַמִּינָדָֽב׃ וְהַחוֹנִים בְּצַד מִזְרָח – דֶּגֶל מַחֲנֵה יְהוּדָה וּקְהִלּוֹתָיו, הַנָּשִׂיא שֶׁל בְּנֵי יְהוּדָה – נַחְשׁוֹן בֶּן עַמִּינָדָב.





* * *



ד וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ שִׁבְעִים וְאַרְבָּעָה אֶלֶף וְשֵׁשׁ מֵאוֹת.





* * *



ה וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה יִשָּׂשכָ֑ר וְנָשִׂיא֙ לִבְנֵ֣י יִשָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ וְהַחוֹנִים אִתּוֹ: שֵׁבֶט יִשָּׂשכָר, הַנָּשִׂיא שֶׁל בְּנֵי יִשָּׂשכָר – נְתַנְאֵל בֶּן צוּעָר.





* * *



ו וּצְבָא֖וֹ וּפְקֻדָ֑יו אַרְבָּעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ חֲמִשִּׁים וְאַרְבָּעָה אֶלֶף וְאַרְבַּע מֵאוֹת.





* * *



ז מַטֵּ֖ה זְבוּלֻ֑ן וְנָשִׂיא֙ לִבְנֵ֣י זְבוּלֻ֔ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ וְשֵׁבֶט זְבוּלוּן, הַנָּשִׂיא שֶׁל בְּנֵי זְבוּלוּן – אֱלִיאָב בֶּן חֵלוֹן.





* * *



ח וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ חֲמִשִּׁים וְשִׁבְעָה אֶלֶף וְאַרְבַּע מֵאוֹת.





* * *



ט כׇּֽל־הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה יְהוּדָ֗ה מְאַ֨ת אֶ֜לֶף וּשְׁמֹנִ֥ים אֶ֛לֶף וְשֵֽׁשֶׁת־אֲלָפִ֥ים וְאַרְבַּע־מֵא֖וֹת לְצִבְאֹתָ֑ם רִאשֹׁנָ֖ה יִסָּֽעוּ׃ כָּל הַמִּנְיָן שֶׁל מַחֲנֵה יְהוּדָה מֵאָה וּשְׁמוֹנִים וְשִׁשָּׁה אֶלֶף וְאַרְבַּע מֵאוֹת לִקְהִלּוֹתֵיהֶם, הֵם יִסְעוּ רִאשׁוֹנִים.





* * *



ס





* * *



י דֶּ֣גֶל מַחֲנֵ֧ה רְאוּבֵ֛ן תֵּימָ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י רְאוּבֵ֔ן אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ דֶּגֶל מַחֲנֵה רְאוּבֵן וּקְהִלּוֹתָיו יַחֲנוּ בַּדָּרוֹם, הַנָּשִׂיא לִבְנֵי רְאוּבֵן – אֱלִיצוּר בֶּן שְׁדֵיאוּר.





* * *



יא וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ אַרְבָּעִים וְשִׁשָּׁה אֶלֶף וַחֲמֵשׁ מֵאוֹת.





* * *



יב וְהַחוֹנִ֥ם עָלָ֖יו מַטֵּ֣ה שִׁמְע֑וֹן וְנָשִׂיא֙ לִבְנֵ֣י שִׁמְע֔וֹן שְׁלֻמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ וְהַחוֹנִים אִתּוֹ: שֵׁבֶט שִׁמְעוֹן, הַנָּשִׂיא לִבְנֵי שִׁמְעוֹן – שְׁלוּמִיאֵל בֶּן צוּרִישַׁדָּי.





* * *



יג וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם תִּשְׁעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ חֲמִשִּׁים וְתִשְׁעָה אֶלֶף וּשְׁלוֹשׁ מֵאוֹת.





* * *



יד וּמַטֵּ֖ה גָּ֑ד וְנָשִׂיא֙ לִבְנֵ֣י גָ֔ד אֶלְיָסָ֖ף בֶּן־רְעוּאֵֽל׃ וְשֵׁבֶט גָּד, הַנָּשִׂיא שֶׁל בְּנֵי גָּד – אֶלְיָסָף בֶּן רְעוּאֵל.





* * *



טו וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ אַרְבָּעִים וַחֲמִשָּׁה אֶלֶף וְשֵׁשׁ מֵאוֹת וַחֲמִשִּׁים.





* * *



טז כׇּֽל־הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה רְאוּבֵ֗ן מְאַ֨ת אֶ֜לֶף וְאֶחָ֨ד וַחֲמִשִּׁ֥ים אֶ֛לֶף וְאַרְבַּע־מֵא֥וֹת וַחֲמִשִּׁ֖ים לְצִבְאֹתָ֑ם וּשְׁנִיִּ֖ם יִסָּֽעוּ׃ כָּל הַמִּנְיָן שֶׁל מַחֲנֵה רְאוּבֵן מֵאָה וַחֲמִשִּׁים וְאֶחָד אֶלֶף וְאַרְבַּע מֵאוֹת וַחֲמִשִּׁים לִקְהִלּוֹתֵיהֶם, וְהֵם יִסְעוּ שְׁנִיִּים.





* * *



ס





* * *



יז וְנָסַ֧ע אֹֽהֶל־מוֹעֵ֛ד מַחֲנֵ֥ה הַלְוִיִּ֖ם בְּת֣וֹךְ הַֽמַּחֲנֹ֑ת כַּאֲשֶׁ֤ר יַחֲנוּ֙ כֵּ֣ן יִסָּ֔עוּ אִ֥ישׁ עַל־יָד֖וֹ לְדִגְלֵיהֶֽם׃ אוֹהֶל הַהִתְכַּנְּסוּת יִסַּע עִם מַחֲנֵה הַלְוִיִּים בְּאֶמְצַע הַמַּחֲנוֹת, כְּמוֹ צוּרַת חֲנִיָּתָם כָּךְ תִּהְיֶה צוּרַת נְסִיעָתָם – כָּל אֶחָד בִּמְקוֹמוֹ מְסֻדָּרִים לְפִי דִּגְלֵיהֶם.





* * *



ס





* * *



יח דֶּ֣גֶל מַחֲנֵ֥ה אֶפְרַ֛יִם לְצִבְאֹתָ֖ם יָ֑מָּה וְנָשִׂיא֙ לִבְנֵ֣י אֶפְרַ֔יִם אֱלִישָׁמָ֖ע בֶּן־עַמִּיהֽוּד׃ דֶּגֶל מַחֲנֵה אֶפְרַיִם וּקְהִלּוֹתָיו בְּצַד מַעֲרָב, הַנָּשִׂיא שֶׁל בְּנֵי אֶפְרַיִם – אֱלִישָׁמָע בֶּן עַמִּיהוּד.





* * *



יט וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ אַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת.





* * *



כ וְעָלָ֖יו מַטֵּ֣ה מְנַשֶּׁ֑ה וְנָשִׂיא֙ לִבְנֵ֣י מְנַשֶּׁ֔ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ וְאִתּוֹ שֵׁבֶט מְנַשֶּׁה, הַנָּשִׂיא שֶׁל בְּנֵי מְנַשֶּׁה גַּמְלִיאֵל בֶּן פְּדָהצוּר.





* * *



כא וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וּמָאתָֽיִם׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ שְׁלוֹשִׁים וּשְׁנַיִם אֶלֶף וּמָאתַיִם.





* * *



כב וּמַטֵּ֖ה בִּנְיָמִ֑ן וְנָשִׂיא֙ לִבְנֵ֣י בִנְיָמִ֔ן אֲבִידָ֖ן בֶּן־גִּדְעֹנִֽי׃ וְשֵׁבֶט בִּנְיָמִין, הַנָּשִׂיא שֶׁל בְּנֵי בִּנְיָמִין – אֲבִידָן בֶּן גִּדְעוֹנִי.





* * *



כג וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם חֲמִשָּׁ֧ה וּשְׁלֹשִׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ שְׁלוֹשִׁים וַחֲמִשָּׁה אֶלֶף וְאַרְבַּע מֵאוֹת.





* * *



כד כׇּֽל־הַפְּקֻדִ֞ים לְמַחֲנֵ֣ה אֶפְרַ֗יִם מְאַ֥ת אֶ֛לֶף וּשְׁמֹֽנַת־אֲלָפִ֥ים וּמֵאָ֖ה לְצִבְאֹתָ֑ם וּשְׁלִשִׁ֖ים יִסָּֽעוּ׃ כָּל הַמִּנְיָן שֶׁל מַחֲנֵה אֶפְרַיִם מֵאָה וּשְׁמוֹנָה אֶלֶף וּמֵאָה לִקְהִלּוֹתֵיהֶם, וְהֵם יִסְעוּ שְׁלִישִׁים.





* * *



ס





* * *



כה דֶּ֣גֶל מַחֲנֵ֥ה דָ֛ן צָפֹ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י דָ֔ן אֲחִיעֶ֖זֶר בֶּן־עַמִּֽישַׁדָּֽי׃ דֶּגֶל מַחֲנֵה דָּן יַחֲנֶה בְּצַד צָפוֹן לִקְהִלּוֹתָיו, הַנָּשִׂיא שֶׁל בְּנֵי דָּן – אֲחִיעֶזֶר בֶּן עַמִּישַׁדָּי.





* * *



כו וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁנַ֧יִם וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ שִׁשִּׁים וּשְׁנַיִם אֶלֶף וּשְׁבַע מֵאוֹת.





* * *



כז וְהַחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה אָשֵׁ֑ר וְנָשִׂיא֙ לִבְנֵ֣י אָשֵׁ֔ר פַּגְעִיאֵ֖ל בֶּן־עׇכְרָֽן׃ וְהַחוֹנִים אִתּוֹ: שֵׁבֶט אָשֵׁר, הַנָּשִׂיא שֶׁל בְּנֵי אָשֵׁר – פַּגְעִיאֵל בֶּן עָכְרָן.





* * *



כח וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם אֶחָ֧ד וְאַרְבָּעִ֛ים אֶ֖לֶף וַחֲמֵ֥שׁ מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ אַרְבָּעִים וְאֶחָד אֶלֶף וַחֲמֵשׁ מֵאוֹת.





* * *



כט וּמַטֵּ֖ה נַפְתָּלִ֑י וְנָשִׂיא֙ לִבְנֵ֣י נַפְתָּלִ֔י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ וְשֵׁבֶט נַפְתָּלִי, הַנָּשִׂיא שֶׁל בְּנֵי נַפְתָּלִי – אֲחִירַע בֶּן עֵינָן.





* * *



ל וּצְבָא֖וֹ וּפְקֻדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וּמִסְפַּר אַנְשֵׁי צְבָאוֹ חֲמִשִּׁים וּשְׁלוֹשָׁה אֶלֶף וְאַרְבַּע מֵאוֹת.





* * *



לא כׇּל־הַפְּקֻדִים֙ לְמַ֣חֲנֵה דָ֔ן מְאַ֣ת אֶ֗לֶף וְשִׁבְעָ֧ה וַחֲמִשִּׁ֛ים אֶ֖לֶף וְשֵׁ֣שׁ מֵא֑וֹת לָאַחֲרֹנָ֥ה יִסְע֖וּ לְדִגְלֵיהֶֽם׃ כָּל הַמִּנְיָן שֶׁל מַחֲנֵה דָּן מֵאָה חֲמִשִּׁים וְשִׁבְעָה אֶלֶף וְשֵׁשׁ מֵאוֹת, דִּגְלֵיהֶם יִסְעוּ אַחֲרוֹנִים.





* * *



פ





* * *



לב אֵ֛לֶּה פְּקוּדֵ֥י בְנֵֽי־יִשְׂרָאֵ֖ל לְבֵ֣ית אֲבֹתָ֑ם כׇּל־פְּקוּדֵ֤י הַֽמַּחֲנֹת֙ לְצִבְאֹתָ֔ם שֵׁשׁ־מֵא֥וֹת אֶ֙לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַחֲמֵ֥שׁ מֵא֖וֹת וַחֲמִשִּֽׁים׃ אֵלֶּה מִנְיָנֵי בְּנֵי יִשְׂרָאֵל בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם כָּל מִנְיְנֵי הַמַּחֲנוֹת לִקְהִלּוֹתֵיהֶם – שֵׁשׁ מֵאוֹת אֶלֶף וּשְׁלוֹשֶׁת אֲלָפִים וַחֲמֵשׁ מֵאוֹת וַחֲמִשִּׁים.





* * *



לג וְהַ֨לְוִיִּ֔ם לֹ֣א הׇתְפָּקְד֔וּ בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ וְהַלְוִיִּים לֹא נִמְנוּ יַחַד עִם בְּנֵי יִשְׂרָאֵל, כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.





* * *



לד וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹ֠ל אֲשֶׁר־צִוָּ֨ה יְהֹוָ֜ה אֶת־מֹשֶׁ֗ה כֵּֽן־חָנ֤וּ לְדִגְלֵיהֶם֙ וְכֵ֣ן נָסָ֔עוּ אִ֥ישׁ לְמִשְׁפְּחֹתָ֖יו עַל־בֵּ֥ית אֲבֹתָֽיו׃ וּבְנֵי יִשְׂרָאֵל עָשׂוּ, כְּכָל מַה שֶּׁה' צִוָּה אֶת מֹשֶׁה כָּךְ חָנוּ מְסֻדָּרִים לְפִי דִּגְלֵיהֶם וְכָךְ נָסְעוּ, כָּל אֶחָד עִם מִשְׁפַּחְתּוֹ לְיַד מִשְׁפְּחוֹת אֲבוֹתָיו.





* * *





רביעי


פ פָּרָשַׁת תּוֹלְדֹת אַהֲרֹן





* * *



ג א וְאֵ֛לֶּה תּוֹלְדֹ֥ת אַהֲרֹ֖ן וּמֹשֶׁ֑ה בְּי֗וֹם דִּבֶּ֧ר יְהֹוָ֛ה אֶת־מֹשֶׁ֖ה בְּהַ֥ר סִינָֽי׃ וְאֵלֶּה צֶאֱצָאֵי אַהֲרֹן וּמֹשֶׁה בַּיּוֹם שֶׁה' דִּבֵּר עִם מֹשֶׁה בְּהַר סִינַי.





* * *



ב וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי־אַהֲרֹ֖ן הַבְּכֹ֣ר ׀ נָדָ֑ב וַאֲבִיה֕וּא אֶלְעָזָ֖ר וְאִיתָמָֽר׃ אֵלֶּה שְׁמוֹת בְּנֵי אַהֲרֹן: הַבְּכוֹר נָדָב, וַאֲבִיהוּא, אֶלְעָזָר, וְאִיתָמָר.





* * *



ג אֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י אַהֲרֹ֔ן הַכֹּהֲנִ֖ים הַמְּשֻׁחִ֑ים אֲשֶׁר־מִלֵּ֥א יָדָ֖ם לְכַהֵֽן׃ אֵלֶּה שְׁמוֹת בְּנֵי אַהֲרֹן הַכֹּהֲנִים שֶׁנִּמְרְחוּ בְּשֶׁמֶן, שֶׁמִּנּוּ אוֹתָם לְכַהֵן.





* * *



ד וַיָּ֣מׇת נָדָ֣ב וַאֲבִיה֣וּא לִפְנֵ֣י יְהֹוָ֡ה בְּֽהַקְרִבָם֩ אֵ֨שׁ זָרָ֜ה לִפְנֵ֤י יְהֹוָה֙ בְּמִדְבַּ֣ר סִינַ֔י וּבָנִ֖ים לֹא־הָי֣וּ לָהֶ֑ם וַיְכַהֵ֤ן אֶלְעָזָר֙ וְאִ֣יתָמָ֔ר עַל־פְּנֵ֖י אַהֲרֹ֥ן אֲבִיהֶֽם׃ נָדָב וַאֲבִיהוּא מֵתוּ לִפְנֵי ה' כְּשֶׁהִקְרִיבוּ לִפְנֵי ה' בְּמִדְבַּר סִינַי אֵשׁ זָרָה שֶׁלֹּא נִצְטַוּוּ עָלֶיהָ, וְלֹא הָיוּ לָהֶם בָּנִים, אֶלְעָזָר וְאִיתָמָר כִּהֲנוּ יַחַד עִם אַהֲרֹן אֲבִיהֶם.





* * *



פ פָּרָשַׁת מִשְׁמֶרֶת הַלְוִיִּים





* * *



ה וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:





* * *



ו הַקְרֵב֙ אֶת־מַטֵּ֣ה לֵוִ֔י וְהַֽעֲמַדְתָּ֣ אֹת֔וֹ לִפְנֵ֖י אַהֲרֹ֣ן הַכֹּהֵ֑ן וְשֵׁרְת֖וּ אֹתֽוֹ׃ הַגֵּשׁ אֶת שֵׁבֶט לֵוִי וְהַעֲמֵד אוֹתוֹ לִפְנֵי אַהֲרֹן הַכֹּהֵן וְהֵם יְשָׁרְתוּ אוֹתוֹ.





* * *



ז וְשָׁמְר֣וּ אֶת־מִשְׁמַרְתּ֗וֹ וְאֶת־מִשְׁמֶ֙רֶת֙ כׇּל־הָ֣עֵדָ֔ה לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד לַעֲבֹ֖ד אֶת־עֲבֹדַ֥ת הַמִּשְׁכָּֽן׃ וְיַעֲשׂוּ אֶת הָעֲבוֹדָה הַמֻּטֶּלֶת עָלָיו וְעַל כָּל הַצִּבּוּר לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת – לַעֲבוֹד אֶת עֲבוֹדַת הַמִּשְׁכָּן.





* * *



ח וְשָׁמְר֗וּ אֶֽת־כׇּל־כְּלֵי֙ אֹ֣הֶל מוֹעֵ֔ד וְאֶת־מִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל לַעֲבֹ֖ד אֶת־עֲבֹדַ֥ת הַמִּשְׁכָּֽן׃ הֵם יִשְׁמְרוּ אֶת כָּל כְּלֵי אוֹהֶל הַהִתְכַּנְּסוּת וְיַעַבְדוּ אֶת הָעֲבוֹדָה הַמֻּטֶּלֶת עַל בְּנֵי יִשְׂרָאֵל, לַעֲבוֹד אֶת עֲבוֹדַת הַמִּשְׁכָּן.





* * *



ט וְנָתַתָּה֙ אֶת־הַלְוִיִּ֔ם לְאַהֲרֹ֖ן וּלְבָנָ֑יו נְתוּנִ֨ם נְתוּנִ֥ם הֵ֙מָּה֙ ל֔וֹ מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ תִּתֵּן אֶת הַלְוִיִּים לְאַהֲרֹן וּלְבָנָיו, הֵם נְתוּנִים לוֹ מֵאֵת בְּנֵי יִשְׂרָאֵל לְעֶזְרָה.





* * *



י וְאֶת־אַהֲרֹ֤ן וְאֶת־בָּנָיו֙ תִּפְקֹ֔ד וְשָׁמְר֖וּ אֶת־כְּהֻנָּתָ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ וְאֶת אַהֲרֹן וְאֶת בָּנָיו תְּמַנֶּה שֶׁיִּשְׁמְרוּ עַל כְּהֻנָּתָם – וְזָר שֶׁיִּגַּשׁ לַעֲשׂוֹת בִּמְקוֹמָם יוּמַת.





* * *



פ





* * *



יא וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:





* * *



יב וַאֲנִ֞י הִנֵּ֧ה לָקַ֣חְתִּי אֶת־הַלְוִיִּ֗ם מִתּוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל תַּ֧חַת כׇּל־בְּכ֛וֹר פֶּ֥טֶר רֶ֖חֶם מִבְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם׃ הִנֵּה אֲנִי לָקַחְתִּי אֶת הַלְוִיִּים מִתּוֹךְ בְּנֵי יִשְׂרָאֵל בִּמְקוֹם כָּל בְּכוֹר מִבְּנֵי יִשְׂרָאֵל שֶׁיָּצָא רִאשׁוֹן מִפֶּתַח הָרֶחֶם, הַלְוִיִּים יִהְיוּ שֶׁלִּי.





* * *



יג כִּ֣י לִי֮ כׇּל־בְּכוֹר֒ בְּיוֹם֩ הַכֹּתִ֨י כׇל־בְּכ֜וֹר בְּאֶ֣רֶץ מִצְרַ֗יִם הִקְדַּ֨שְׁתִּי לִ֤י כׇל־בְּכוֹר֙ בְּיִשְׂרָאֵ֔ל מֵאָדָ֖ם עַד־בְּהֵמָ֑ה לִ֥י יִהְי֖וּ אֲנִ֥י יְהֹוָֽה׃ כִּי כָּל הַבְּכוֹרוֹת הֵם שֶׁלִּי, כִּי בַּיּוֹם שֶׁהִכֵּיתִי אֶת כָּל הַבְּכוֹרוֹת בְּאֶרֶץ מִצְרַיִם הִקְדַּשְׁתִּי לִי אֶת כָּל הַבְּכוֹרוֹת בְּיִשְׂרָאֵל מֵאָדָם עַד בְּהֵמָה, הֵם יִהְיוּ שֶׁלִּי, אֲנִי ה'!





* * *





חמישי


פ פָּרָשַׁת פְּקוּדֵי הַלְוִיִּים





* * *



יד וַיְדַבֵּ֤ר יְהֹוָה֙ אֶל־מֹשֶׁ֔ה בְּמִדְבַּ֥ר סִינַ֖י לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה בְּמִדְבַּר סִינַי וְאָמַר:





* * *



טו פְּקֹד֙ אֶת־בְּנֵ֣י לֵוִ֔י לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָ֑ם כׇּל־זָכָ֛ר מִבֶּן־חֹ֥דֶשׁ וָמַ֖עְלָה תִּפְקְדֵֽם׃ סְפוֹר אֶת בְּנֵי לֵוִי לְמִשְׁפְּחוֹתֵיהֶם בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם, תִּסְפּוֹר אוֹתָם – כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמַעְלָה.





* * *



טז וַיִּפְקֹ֥ד אֹתָ֛ם מֹשֶׁ֖ה עַל־פִּ֣י יְהֹוָ֑ה כַּאֲשֶׁ֖ר צֻוָּֽה׃ מֹשֶׁה סָפַר אוֹתָם עַל פִּי דִּבְרֵי ה' כְּמוֹ שֶׁנִּצְטַוָּה.





* * *



יז וַיִּֽהְיוּ־אֵ֥לֶּה בְנֵֽי־לֵוִ֖י בִּשְׁמֹתָ֑ם גֵּרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִֽי׃ אֵלֶּה הָיוּ בְּנֵי לֵוִי בִּשְׁמוֹתָם: גֵּרְשׁוֹן קְהָת וּמְרָרִי.





* * *



יח וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי־גֵרְשׁ֖וֹן לְמִשְׁפְּחֹתָ֑ם לִבְנִ֖י וְשִׁמְעִֽי׃ וְאֵלֶּה שְׁמוֹת בְּנֵי גֵּרְשׁוֹן לְמִשְׁפְּחוֹתֵיהֶם: לִבְנִי וְשִׁמְעִי.





* * *



יט וּבְנֵ֥י קְהָ֖ת לְמִשְׁפְּחֹתָ֑ם עַמְרָ֣ם וְיִצְהָ֔ר חֶבְר֖וֹן וְעֻזִּיאֵֽל׃ וּבְנֵי קְהָת לְמִשְׁפְּחוֹתֵיהֶם: עַמְרָם וְיִצְהָר חֶבְרוֹן וְעֻזִּיאֵל.





* * *



כ וּבְנֵ֧י מְרָרִ֛י לְמִשְׁפְּחֹתָ֖ם מַחְלִ֣י וּמוּשִׁ֑י אֵ֥לֶּה הֵ֛ם מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְבֵ֥ית אֲבֹתָֽם׃ וּבְנֵי מְרָרִי לְמִשְׁפְּחוֹתֵיהֶם: מַחְלִי וּמוּשִׁי אֵלֶּה הֵם מִשְׁפְּחוֹת הַלֵּוִי לְמִשְׁפְּחוֹת אֲבוֹתֵיהֶם.





* * *



כא לְגֵ֣רְשׁ֔וֹן מִשְׁפַּ֙חַת֙ הַלִּבְנִ֔י וּמִשְׁפַּ֖חַת הַשִּׁמְעִ֑י אֵ֣לֶּה הֵ֔ם מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּֽי׃ לְגֵרְשׁוֹן: מִשְׁפַּחַת הַלִּבְנִי וּמִשְׁפַּחַת הַשִּׁמְעִי אֵלֶּה הֵם מִשְׁפְּחוֹת הַגֵּרְשֻׁנִּי.





* * *



כב פְּקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כׇּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה פְּקֻ֣דֵיהֶ֔ם שִׁבְעַ֥ת אֲלָפִ֖ים וַחֲמֵ֥שׁ מֵאֽוֹת׃ מִסְפָּרָם כְּשֶׁסָּפְרוּ כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמָעְלָה – שִׁבְעַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת.





* * *



כג מִשְׁפְּחֹ֖ת הַגֵּרְשֻׁנִּ֑י אַחֲרֵ֧י הַמִּשְׁכָּ֛ן יַחֲנ֖וּ יָֽמָּה׃ מִשְׁפְּחוֹת הַגֵּרְשֻׁנִּי יַחֲנוּ מֵאַחוֹרֵי הַמִּשְׁכָּן – בְּצַד מַעֲרָב.





* * *



כד וּנְשִׂ֥יא בֵֽית־אָ֖ב לַגֵּרְשֻׁנִּ֑י אֶלְיָסָ֖ף בֶּן־לָאֵֽל׃ וּנְשִׂיא הַמִּשְׁפָּחָה שֶׁל מִשְׁפְּחוֹת הַגֵּרְשֻׁנִּי – אֶלְיָסָף בֶּן לָאֵל.





* * *



כה וּמִשְׁמֶ֤רֶת בְּנֵֽי־גֵרְשׁוֹן֙ בְּאֹ֣הֶל מוֹעֵ֔ד הַמִּשְׁכָּ֖ן וְהָאֹ֑הֶל מִכְסֵ֕הוּ וּמָסַ֕ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ הַתַּפְקִיד שֶׁל בְּנֵי גֵּרְשׁוֹן בְּאוֹהֶל הַהִתְכַּנְּסוּת: כִּסּוּי הַמִּשְׁכָּן, וְהָאוֹהֶל, וְהַכִּסּוּי שֶׁעָלָיו, וְהַמָּסַךְ שֶׁל פֶּתַח אוֹהֶל הַהִתְכַּנְּסוּת,





* * *



כו וְקַלְעֵ֣י הֶֽחָצֵ֗ר וְאֶת־מָסַךְ֙ פֶּ֣תַח הֶֽחָצֵ֔ר אֲשֶׁ֧ר עַל־הַמִּשְׁכָּ֛ן וְעַל־הַמִּזְבֵּ֖חַ סָבִ֑יב וְאֵת֙ מֵֽיתָרָ֔יו לְכֹ֖ל עֲבֹדָתֽוֹ׃ וְהַגְּדֵרוֹת הַקְּלוּעוֹת שֶׁל הֶחָצֵר, וְהַמָּסַךְ שֶׁל פֶּתַח הֶחָצֵר שֶׁסְּבִיב הַמִּשְׁכָּן וְהַמִּזְבֵּחַ, וּמֵיתְרֵי הָאוֹהֶל הַמְּשַׁמְּשִׂים לְכָל צָרְכֵי עֲבוֹדָתוֹ.





* * *



ס





* * *



כז וְלִקְהָ֗ת מִשְׁפַּ֤חַת הַֽעַמְרָמִי֙ וּמִשְׁפַּ֣חַת הַיִּצְהָרִ֔י וּמִשְׁפַּ֙חַת֙ הַֽחֶבְרֹנִ֔י וּמִשְׁפַּ֖חַת הָעׇזִּֽיאֵלִ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת הַקְּהָתִֽי׃ וְלִקְהָת: מִשְׁפַּחַת הַעַמְרָמִי, וּמִשְׁפַּחַת הַיִּצְהָרִי, וּמִשְׁפַּחַת הַחֶבְרוֹנִי, וּמִשְׁפַּחַת הָעָזִּיאֵלִי; אֵלֶּה הֵם מִשְׁפְּחוֹת הַקְּהָתִי.





* * *



כח בְּמִסְפַּר֙ כׇּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה שְׁמֹנַ֤ת אֲלָפִים֙ וְשֵׁ֣שׁ מֵא֔וֹת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ׃ בִּסְפִירַת כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמָעְלָה נִמְצָא מִנְיָנָם: שְׁמוֹנַת אֲלָפִים וְשֵׁשׁ מֵאוֹת שׁוֹמְרֵי מִשְׁמֶרֶת הַקּוֹדֶשׁ.





* * *



כט מִשְׁפְּחֹ֥ת בְּנֵי־קְהָ֖ת יַחֲנ֑וּ עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן תֵּימָֽנָה׃ מִשְׁפְּחוֹת בְּנֵי קְהָת יַחֲנוּ בְּצַד הַמִּשְׁכָּן מִדָּרוֹם.





* * *



ל וּנְשִׂ֥יא בֵֽית־אָ֖ב לְמִשְׁפְּחֹ֣ת הַקְּהָתִ֑י אֱלִיצָפָ֖ן בֶּן־עֻזִּיאֵֽל׃ וּנְשִׂיא הַמִּשְׁפָּחָה שֶׁל מִשְׁפְּחוֹת הַקְּהָתִי – אֱלִיצָפָן בֶּן עֻזִּיאֵל.





* * *



לא וּמִשְׁמַרְתָּ֗ם הָאָרֹ֤ן וְהַשֻּׁלְחָן֙ וְהַמְּנֹרָ֣ה וְהַֽמִּזְבְּחֹ֔ת וּכְלֵ֣י הַקֹּ֔דֶשׁ אֲשֶׁ֥ר יְשָׁרְת֖וּ בָּהֶ֑ם וְהַ֨מָּסָ֔ךְ וְכֹ֖ל עֲבֹדָתֽוֹ׃ וְתַפְקִידָם: הָאָרוֹן, וְהַשֻּׁלְחָן, וְהַמְּנוֹרָה, וְהַמִּזְבְּחוֹת וּכְלֵי הַקּוֹדֶשׁ שֶׁיְּשָׁרְתוּ בָּהֶם, וְהַמָּסָךְ וְכָל עֲבוֹדָתוֹ.





* * *



לב וּנְשִׂיא֙ נְשִׂיאֵ֣י הַלֵּוִ֔י אֶלְעָזָ֖ר בֶּן־אַהֲרֹ֣ן הַכֹּהֵ֑ן פְּקֻדַּ֕ת שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ׃ וְהַנָּשִׂיא שֶׁל נְשִׂיאֵי הַלֵּוִי – אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן – מְפַקֵּד עוֹבְדֵי עֲבוֹדַת הַקּוֹדֶשׁ.





* * *



לג לִמְרָרִ֕י מִשְׁפַּ֙חַת֙ הַמַּחְלִ֔י וּמִשְׁפַּ֖חַת הַמּוּשִׁ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת מְרָרִֽי׃ לִמְרָרִי: מִשְׁפַּחַת הַמַּחְלִי, וּמִשְׁפַּחַת הַמּוּשִׁי. אֵלֶּה הֵם מִשְׁפְּחוֹת מְרָרִי.





* * *



לד וּפְקֻדֵיהֶם֙ בְּמִסְפַּ֣ר כׇּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה שֵׁ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם׃ וּמִסְפָּרָם כְּשֶׁסָּפְרוּ כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמַעְלָה – שֵׁשֶׁת אֲלָפִים וּמָאתַיִם.





* * *



לה וּנְשִׂ֤יא בֵֽית־אָב֙ לְמִשְׁפְּחֹ֣ת מְרָרִ֔י צוּרִיאֵ֖ל בֶּן־אֲבִיחָ֑יִל עַ֣ל יֶ֧רֶךְ הַמִּשְׁכָּ֛ן יַחֲנ֖וּ צָפֹֽנָה׃ וּנְשִׂיא הַמִּשְׁפָּחָה שֶׁל מִשְׁפְּחוֹת מְרָרִי – צוּרִיאֵל בֶּן אֲבִיחַיִל, הֵם יַחֲנוּ בְּצַד הַמִּשְׁכָּן מִצָּפוֹן.





* * *



לו וּפְקֻדַּ֣ת מִשְׁמֶ֘רֶת֮ בְּנֵ֣י מְרָרִי֒ קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמֻּדָ֣יו וַאֲדָנָ֑יו וְכׇ֨ל־כֵּלָ֔יו וְכֹ֖ל עֲבֹדָתֽוֹ׃ וּבְנֵי מְרָרִי מֻפְקָדִים עַל קַרְשֵׁי הַמִּשְׁכָּן, וּבְרִיחָיו, וְעַמּוּדָיו, וּבְסִיסָיו, וְכָל כֵּלָיו וְכָל עֲבוֹדָתוֹ,





* * *



לז וְעַמֻּדֵ֧י הֶחָצֵ֛ר סָבִ֖יב וְאַדְנֵיהֶ֑ם וִיתֵדֹתָ֖ם וּמֵֽיתְרֵיהֶֽם׃ וְעַמּוּדֵי הֶחָצֵר מִסָּבִיב, וּבְסִיסֵיהֶם וִיתֵדוֹתָם וּמֵיתְרֵיהֶם.





* * *



לח וְהַחֹנִ֣ים לִפְנֵ֣י הַמִּשְׁכָּ֡ן קֵ֣דְמָה לִפְנֵי֩ אֹֽהֶל־מוֹעֵ֨ד ׀ מִזְרָ֜חָה מֹשֶׁ֣ה ׀ וְאַהֲרֹ֣ן וּבָנָ֗יו שֹֽׁמְרִים֙ מִשְׁמֶ֣רֶת הַמִּקְדָּ֔שׁ לְמִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ וְהַחוֹנִים לִפְנֵי הַמִּשְׁכָּן בַּמִּזְרָח – לִפְנֵי אוֹהֶל הַהִתְכַּנְּסוּת בְּצַד מִזְרָח: מֹשֶׁה וְאַהֲרֹן וּבָנָיו – שׁוֹמְרִים אֶת מִשְׁמַר הַמִּקְדָּשׁ הַמֻּטָּל עַל בְּנֵי יִשְׂרָאֵל, וְזָר שֶׁיִּגַּשׁ לַעֲשׂוֹת אֶת עֲבוֹדָתָם – יוּמַת.





* * *



לט כׇּל־פְּקוּדֵ֨י הַלְוִיִּ֜ם אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְׄאַׄהֲׄרֹ֛ׄןׄ עַל־פִּ֥י יְהֹוָ֖ה לְמִשְׁפְּחֹתָ֑ם כׇּל־זָכָר֙ מִבֶּן־חֹ֣דֶשׁ וָמַ֔עְלָה שְׁנַ֥יִם וְעֶשְׂרִ֖ים אָֽלֶף׃ כָּל מִנְיַן הַלְוִיִּים שֶׁמֹּשֶׁה וְאַהֲרֹן סָפְרוּ עַל פִּי ה' לְמִשְׁפְּחוֹתֵיהֶם – כָּל זָכָר מִגִּיל חוֹדֶשׁ וָמַעְלָה עֶשְׂרִים וּשְׁנַיִם אֶלֶף.





* * *





שישי


ס פָּרָשַׁת פִּדְיוֹן הַבְּכוֹרוֹת





* * *



מ וַיֹּ֨אמֶר יְהֹוָ֜ה אֶל־מֹשֶׁ֗ה פְּקֹ֨ד כׇּל־בְּכֹ֤ר זָכָר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה וְשָׂ֕א אֵ֖ת מִסְפַּ֥ר שְׁמֹתָֽם׃ ה' אָמַר לְמֹשֶׁה: סְפוֹר כָּל בְּכוֹר זָכָר מִבְּנֵי יִשְׂרָאֵל מִגִּיל חוֹדֶשׁ וָמַעְלָה, וְחַשֵּׁב אֶת מִסְפַּר שְׁמוֹתֵיהֶם.





* * *



מא וְלָקַחְתָּ֨ אֶת־הַלְוִיִּ֥ם לִי֙ אֲנִ֣י יְהֹוָ֔ה תַּ֥חַת כׇּל־בְּכֹ֖ר בִּבְנֵ֣י יִשְׂרָאֵ֑ל וְאֵת֙ בֶּהֱמַ֣ת הַלְוִיִּ֔ם תַּ֣חַת כׇּל־בְּכ֔וֹר בְּבֶהֱמַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ אֲנִי ה'! קַח אֶת הַלְוִיִּים לַעֲבוֹדָתִי בִּמְקוֹם כָּל בְּכוֹר בִּבְנֵי יִשְׂרָאֵל, וְאֶת בֶּהֱמוֹת הַלְוִיִּים בִּמְקוֹם כָּל בְּכוֹרוֹת בֶּהֱמוֹת בְּנֵי יִשְׂרָאֵל.





* * *



מב וַיִּפְקֹ֣ד מֹשֶׁ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֹת֑וֹ אֶֽת־כׇּל־בְּכ֖וֹר בִּבְנֵ֥י יִשְׂרָאֵֽל׃ מֹשֶׁה סָפַר אֶת כָּל הַבְּכוֹרוֹת שֶׁל בְּנֵי יִשְׂרָאֵל כְּמוֹ שֶׁה' צִוָּה אוֹתוֹ.





* * *



מג וַיְהִי֩ כׇל־בְּכ֨וֹר זָכָ֜ר בְּמִסְפַּ֥ר שֵׁמֹ֛ת מִבֶּן־חֹ֥דֶשׁ וָמַ֖עְלָה לִפְקֻדֵיהֶ֑ם שְׁנַ֤יִם וְעֶשְׂרִים֙ אֶ֔לֶף שְׁלֹשָׁ֥ה וְשִׁבְעִ֖ים וּמָאתָֽיִם׃ מִסְפַּר שְׁמוֹת כָּל הַבְּכוֹרוֹת הַזְּכָרִים מִגִּיל חוֹדֶשׁ וָמַעְלָה הָיָה עֶשְׂרִים וּשְׁנַיִם אֶלֶף מָאתַיִם שִׁבְעִים וּשְׁלוֹשָׁה.





* * *



פ





* * *



מד וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאָמַר:





* * *



מה קַ֣ח אֶת־הַלְוִיִּ֗ם תַּ֤חַת כׇּל־בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְאֶת־בֶּהֱמַ֥ת הַלְוִיִּ֖ם תַּ֣חַת בְּהֶמְתָּ֑ם וְהָיוּ־לִ֥י הַלְוִיִּ֖ם אֲנִ֥י יְהֹוָֽה׃ קַח אֶת הַלְוִיִּים בִּמְקוֹם כָּל בְּכוֹר בִּבְנֵי יִשְׂרָאֵל וְאֶת בֶּהֱמוֹת הַלְוִיִּים בִּמְקוֹם בֶּהֱמוֹת יִשְׂרָאֵל, וְהַלְוִיִּים יֻקְדְּשׁוּ לַעֲבוֹדָתִי, אֲנִי ה'!





* * *



מו וְאֵת֙ פְּדוּיֵ֣י הַשְּׁלֹשָׁ֔ה וְהַשִּׁבְעִ֖ים וְהַמָּאתָ֑יִם הָעֹֽדְפִים֙ עַל־הַלְוִיִּ֔ם מִבְּכ֖וֹר בְּנֵ֥י יִשְׂרָאֵֽל׃ וּכְדֵי לִפְדּוֹת אֶת מָאתַיִם שִׁבְעִים וּשְׁלוֹשָׁה בְּכוֹרֵי בְּנֵי יִשְׂרָאֵל שֶׁעוֹדְפִים עַל הַלְוִיִּים –





* * *



מז וְלָקַחְתָּ֗ חֲמֵ֧שֶׁת חֲמֵ֛שֶׁת שְׁקָלִ֖ים לַגֻּלְגֹּ֑לֶת בְּשֶׁ֤קֶל הַקֹּ֙דֶשׁ֙ תִּקָּ֔ח עֶשְׂרִ֥ים גֵּרָ֖ה הַשָּֽׁקֶל׃ קַח מִשְׁקַל שֶׁל חֲמִשָּׁה שְׁקָלִים (70.5 גְּרַם) לְכָל אִישׁ, קַח לְּפִי מִשְׁקַל שִׁקְלֵי קוֹדֶשׁ – מִשְׁקַל הַשֶּׁקֶל עֶשְׂרִים גֵּרָה (14.1 גְּרַם).





* * *



מח וְנָתַתָּ֣ה הַכֶּ֔סֶף לְאַהֲרֹ֖ן וּלְבָנָ֑יו פְּדוּיֵ֕י הָעֹדְפִ֖ים בָּהֶֽם׃ וְתִתֵּן אֶת הַכֶּסֶף לְאַהֲרֹן וּלְבָנָיו – פִּדְיוֹן הַבְּכוֹרוֹת הָעוֹדְפִים עַל הַלְוִיִּים.





* * *



מט וַיִּקַּ֣ח מֹשֶׁ֔ה אֵ֖ת כֶּ֣סֶף הַפִּדְי֑וֹם מֵאֵת֙ הָעֹ֣דְפִ֔ים עַ֖ל פְּדוּיֵ֥י הַלְוִיִּֽם׃ מֹשֶׁה לָקַח אֶת כֶּסֶף הַפִּדְיוֹן מֵהַבְּכוֹרוֹת הָעוֹדְפִים עַל אֵלֶּה שֶׁנִּפְדּוּ בַּלְּוִיִּים.





* * *



נ מֵאֵ֗ת בְּכ֛וֹר בְּנֵ֥י יִשְׂרָאֵ֖ל לָקַ֣ח אֶת־הַכָּ֑סֶף חֲמִשָּׁ֨ה וְשִׁשִּׁ֜ים וּשְׁלֹ֥שׁ מֵא֛וֹת וָאֶ֖לֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ הוּא לָקַח אֶת הַכֶּסֶף מִבְּכוֹרֵי בְּנֵי יִשְׂרָאֵל – מִשְׁקַל שֶׁל אֶלֶף וּשְׁלוֹשׁ מֵאוֹת וְשִׁשִּׁים וַחֲמִשָּׁה שְׁקָלִים בְּשִׁקְלֵי הַקּוֹדֶשׁ (כִּ19 וָרֶבַע קִילוֹ).





* * *



נא וַיִּתֵּ֨ן מֹשֶׁ֜ה אֶת־כֶּ֧סֶף הַפְּדֻיִ֛ם לְאַהֲרֹ֥ן וּלְבָנָ֖יו עַל־פִּ֣י יְהֹוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהֹוָ֖ה אֶת־מֹשֶֽׁה׃ מֹשֶׁה נָתַן אֶת הַכֶּסֶף שֶׁנָּתְנוּ הַפְּדוּיִים לְאַהֲרֹן וּלְבָנָיו עַל פִּי דִּבְרֵי ה', כְּמוֹ שֶׁה' צִוָּה אֶת מֹשֶׁה.





* * *





שביעי


פ פָּרָשַׁת בְּנֵי קְהָת





* * *



ד א וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:





* * *



ב נָשֹׂ֗א אֶת־רֹאשׁ֙ בְּנֵ֣י קְהָ֔ת מִתּ֖וֹךְ בְּנֵ֣י לֵוִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם׃ סִפְרוּ אֶת בְּנֵי קְהָת מִתּוֹךְ בְּנֵי לֵוִי לְמִשְׁפְּחוֹתֵיהֶם בַּחֲלוּקָה לְפִי מִשְׁפְּחוֹת אֲבוֹתֵיהֶם.





* * *



ג מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כׇּל־בָּא֙ לַצָּבָ֔א לַעֲשׂ֥וֹת מְלָאכָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִגִּיל שְׁלוֹשִׁים שָׁנָה וָמַעְלָה וְעַד גִּיל חֲמִשִּׁים שָׁנָה – כָּל הַבָּאִים לַקָּהָל שֶׁל עוֹשֵׂי הַמְּלָאכָה בְּאוֹהֶל הַהִתְכַּנְּסוּת.





* * *



ד זֹ֛את עֲבֹדַ֥ת בְּנֵי־קְהָ֖ת בְּאֹ֣הֶל מוֹעֵ֑ד קֹ֖דֶשׁ הַקֳּדָשִֽׁים׃ זֹאת עֲבוֹדַת בְּנֵי קְהָת בְּאוֹהֶל הַהִתְכַּנְּסוּת – הָעֲבוֹדָה הַקְּדוֹשָׁה בְּיוֹתֵר.





* * *



ה וּבָ֨א אַהֲרֹ֤ן וּבָנָיו֙ בִּנְסֹ֣עַ הַֽמַּחֲנֶ֔ה וְהוֹרִ֕דוּ אֵ֖ת פָּרֹ֣כֶת הַמָּסָ֑ךְ וְכִ֨סּוּ־בָ֔הּ אֵ֖ת אֲרֹ֥ן הָעֵדֻֽת׃ אַהֲרֹן וּבָנָיו יָבוֹאוּ כְּשֶׁהַמַּחֲנֶה יִסַּע, וְיוֹרִידוּ אֶת מָסַךְ הַפָּרוֹכֶת וִיכַסּוּ בָּהּ אֶת אֲרוֹן לוּחוֹת הַבְּרִית.





* * *



ו וְנָתְנ֣וּ עָלָ֗יו כְּסוּי֙ ע֣וֹר תַּ֔חַשׁ וּפָרְשׂ֧וּ בֶֽגֶד־כְּלִ֛יל תְּכֵ֖לֶת מִלְמָ֑עְלָה וְשָׂמ֖וּ בַּדָּֽיו׃ וְיָשִׂימוּ עָלָיו כִּסּוּי מֵעוֹר שֶׁל הַחַיָּה תַּחַשׁ וְיִפְרְשׂוּ מִלְּמַעְלָה בֶּגֶד צָבוּעַ כֻּלּוֹ תְּכֵלֶת, וִיסַדְּרוּ אֶת הַמּוֹטוֹת שֶׁלּוֹ לְהָרִימוֹ.





* * *



ז וְעַ֣ל ׀ שֻׁלְחַ֣ן הַפָּנִ֗ים יִפְרְשׂוּ֮ בֶּ֣גֶד תְּכֵ֒לֶת֒ וְנָתְנ֣וּ עָ֠לָ֠יו אֶת־הַקְּעָרֹ֤ת וְאֶת־הַכַּפֹּת֙ וְאֶת־הַמְּנַקִּיֹּ֔ת וְאֵ֖ת קְשׂ֣וֹת הַנָּ֑סֶךְ וְלֶ֥חֶם הַתָּמִ֖יד עָלָ֥יו יִהְיֶֽה׃ וְעַל שֻׁלְחַן הַלֶּחֶם הַמְּכֻבָּד יִפְרְשׂוּ בֶּגֶד תְּכֵלֶת וְיָשִׂימוּ עָלָיו אֶת הַקְּעָרוֹת וְאֶת הַכַּפּוֹת וְאֶת הָעַמּוּדִים שֶׁמַּחְזִיקִים אֶת הַמַּקְלוֹת, וְאֶת הַמַּקְלוֹת שֶׁסּוֹכְכִים עַל הַשֻּׁלְחָן, שֶׁעֲלֵיהֶם מֻנָּח הַלֶּחֶם, וְעָלָיו יִהְיֶה לֶחֶם הַתָּמִיד.





* * *



ח וּפָרְשׂ֣וּ עֲלֵיהֶ֗ם בֶּ֚גֶד תּוֹלַ֣עַת שָׁנִ֔י וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת־בַּדָּֽיו׃ וְיִפְרְשׂוּ עֲלֵיהֶם בֶּגֶד צָבוּעַ בְּאָדוֹם מִתּוֹלַעַת וִיכַסּוּ אוֹתוֹ בְּכִסּוּי שֶׁעָשׂוּי מֵעוֹר הַחַיָּה תַּחַשׁ וְיָשִׂימוּ אֶת הַמּוֹטוֹת שֶׁלּוֹ.





* * *



ט וְלָקְח֣וּ ׀ בֶּ֣גֶד תְּכֵ֗לֶת וְכִסּ֞וּ אֶת־מְנֹרַ֤ת הַמָּאוֹר֙ וְאֶת־נֵ֣רֹתֶ֔יהָ וְאֶת־מַלְקָחֶ֖יהָ וְאֶת־מַחְתֹּתֶ֑יהָ וְאֵת֙ כׇּל־כְּלֵ֣י שַׁמְנָ֔הּ אֲשֶׁ֥ר יְשָׁרְתוּ־לָ֖הּ בָּהֶֽם׃ וְיִקְחוּ בֶּגֶד צָבוּעַ בִּתְכֵלֶת וִיכַסּוּ אֶת מְנוֹרַת הַתְּאוּרָה וְאֶת נֵרוֹתֶיהָ וְאֶת הַמֶּלְקְחַיִם וְאֶת הַיָעִים שֶׁמְּנַקִּים אוֹתָהּ בָּהֶם, וְאֶת כָּל כְּלֵי הַשֶּׁמֶן שֶׁלָּהּ שֶׁמִּשְׁתַּמְּשִׁים בָּהֶם עֲבוּרָהּ.





* * *



י וְנָתְנ֤וּ אֹתָהּ֙ וְאֶת־כׇּל־כֵּלֶ֔יהָ אֶל־מִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל־הַמּֽוֹט׃ וְיָשִׂימוּ אוֹתָהּ וְאֶת כָּל כֵּלֶיהָ בְּכִסּוּי שֶׁעָשׂוּי מֵעוֹר הַחַיָּה תַּחַשׁ, וְיָשִׂימוּ עַל מוֹט לִנְשִׂיאָה.





* * *



יא וְעַ֣ל ׀ מִזְבַּ֣ח הַזָּהָ֗ב יִפְרְשׂוּ֙ בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת־בַּדָּֽיו׃ וְעַל מִזְבַּח הַזָּהָב יִפְרְשׂוּ בֶּגֶד תְּכֵלֶת וִיכַסּוּ אוֹתוֹ בְּכִסּוּי שֶׁעָשׂוּי מֵעוֹר תַּחַשׁ וְיָשִׂימוּ אֶת הַמּוֹטוֹת שֶׁלּוֹ.





* * *



יב וְלָקְחוּ֩ אֶת־כׇּל־כְּלֵ֨י הַשָּׁרֵ֜ת אֲשֶׁ֧ר יְשָֽׁרְתוּ־בָ֣ם בַּקֹּ֗דֶשׁ וְנָֽתְנוּ֙ אֶל־בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אוֹתָ֔ם בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָתְנ֖וּ עַל־הַמּֽוֹט׃ וְיִקְחוּ אֶת כָּל כְּלֵי הַשָּׁרֵת שֶׁיְּשָׁרְתוּ בָּהֶם בַּקּוֹדֶשׁ וִיכַסּוּ בְּבֶגֶד תְּכֵלֶת וִיכַסּוּ אוֹתָם בְּכִסּוּי מֵעוֹר הַחַיָּה תַּחַשׁ וְיָשִׂימוּ עַל מוֹט.





* * *



יג וְדִשְּׁנ֖וּ אֶת־הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֔יו בֶּ֖גֶד אַרְגָּמָֽן׃ וִינַקּוּ אֶת הַמִּזְבֵּחַ וְיִפְרְשׂוּ עָלָיו בֶּגֶד צֶמֶר צָבוּעַ בְּאַרְגָּמָן.





* * *



יד וְנָתְנ֣וּ עָ֠לָ֠יו אֶֽת־כׇּל־כֵּלָ֞יו אֲשֶׁ֣ר יְֽשָׁרְת֧וּ עָלָ֣יו בָּהֶ֗ם אֶת־הַמַּחְתֹּ֤ת אֶת־הַמִּזְלָגֹת֙ וְאֶת־הַיָּעִ֣ים וְאֶת־הַמִּזְרָקֹ֔ת כֹּ֖ל כְּלֵ֣י הַמִּזְבֵּ֑חַ וּפָרְשׂ֣וּ עָלָ֗יו כְּס֛וּי ע֥וֹר תַּ֖חַשׁ וְשָׂמ֥וּ בַדָּֽיו׃ וְיָשִׂימוּ עָלָיו אֶת כָּל כֵּלָיו שֶׁמִּשְׁתַּמְּשִׁים בָּהֶם עָלָיו – אֶת הַיָּעִים לִלְקִיחַת הַגֶּחָלִים וְאֶת הַקִּלְּשׁוֹנִים לַהֲפִיכַת הַבָּשָׂר וְאֶת מַגְרְפוֹת נִקּוּי הָאֵפֶר וְאֶת כּוֹסוֹת קַבָּלַת הַדָּם וּזְרִיקָתוֹ – כָּל כְּלֵי הַמִּזְבֵּחַ, וְיִפְרְשׂוּ עָלָיו כִּסּוּי שֶׁעָשׂוּי מֵעוֹר הַחַיָּה תַּחַשׁ וְיָשִׂימוּ אֶת הַמּוֹטוֹת שֶׁלּוֹ.





* * *



טו וְכִלָּ֣ה אַֽהֲרֹן־וּ֠בָנָ֠יו לְכַסֹּ֨ת אֶת־הַקֹּ֜דֶשׁ וְאֶת־כׇּל־כְּלֵ֣י הַקֹּ֘דֶשׁ֮ בִּנְסֹ֣עַ הַֽמַּחֲנֶה֒ וְאַחֲרֵי־כֵ֗ן יָבֹ֤אוּ בְנֵי־קְהָת֙ לָשֵׂ֔את וְלֹֽא־יִגְּע֥וּ אֶל־הַקֹּ֖דֶשׁ וָמֵ֑תוּ אֵ֛לֶּה מַשָּׂ֥א בְנֵֽי־קְהָ֖ת בְּאֹ֥הֶל מוֹעֵֽד׃ אַהֲרֹן וּבָנָיו יִגְמְרוּ לְכַסּוֹת אֶת הַקּוֹדֶשׁ וְאֶת כָּל כְּלֵי הַקּוֹדֶשׁ כְּשֶׁהַמַּחֲנֶה יִסַּע, וְאַחֲרֵי כֵן בְּנֵי קְהָת יָבוֹאוּ לִסְחוֹב וְלֹא יִגְּעוּ בִּכְלֵי הַקּוֹדֶשׁ וְיָמוּתוּ; זֶה הַמַּשָּׂא שֶׁל בְּנֵי קְהָת בְּאוֹהֶל הַהִתְכַּנְּסוּת.





* * *



טז וּפְקֻדַּ֞ת אֶלְעָזָ֣ר ׀ בֶּן־אַהֲרֹ֣ן הַכֹּהֵ֗ן שֶׁ֤מֶן הַמָּאוֹר֙ וּקְטֹ֣רֶת הַסַּמִּ֔ים וּמִנְחַ֥ת הַתָּמִ֖יד וְשֶׁ֣מֶן הַמִּשְׁחָ֑ה פְּקֻדַּ֗ת כׇּל־הַמִּשְׁכָּן֙ וְכׇל־אֲשֶׁר־בּ֔וֹ בְּקֹ֖דֶשׁ וּבְכֵלָֽיו׃ וְתַפְקִיד אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן: הַשֶּׁמֶן לַתְּאוּרָה וּקְטוֹרֶת הַבְּשָׂמִים וּמִנְחַת הַתָּמִיד וְהַשֶּׁמֶן שֶׁמּוֹרְחִים בּוֹ, וְהַפִּקּוּד עַל כָּל הַמִּשְׁכָּן וְכָל מַה שֶּׁבּוֹ בַּקּוֹדֶשׁ וּבְכֵלָיו.





* * *





מפטיר


פ





* * *



יז וַיְדַבֵּ֣ר יְהֹוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ ה' דִּבֵּר אֶל מֹשֶׁה וְאֶל אַהֲרֹן וְאָמַר:





* * *



יח אַל־תַּכְרִ֕יתוּ אֶת־שֵׁ֖בֶט מִשְׁפְּחֹ֣ת הַקְּהָתִ֑י מִתּ֖וֹךְ הַלְוִיִּֽם׃ אַל תַּשְׁמִידוּ אֶת שֵׁבֶט מִשְׁפְּחוֹת הַקְּהָתִי מִתּוֹךְ הַלְוִיִּים.





* * *



יט וְזֹ֣את ׀ עֲשׂ֣וּ לָהֶ֗ם וְחָיוּ֙ וְלֹ֣א יָמֻ֔תוּ בְּגִשְׁתָּ֖ם אֶת־קֹ֣דֶשׁ הַקֳּדָשִׁ֑ים אַהֲרֹ֤ן וּבָנָיו֙ יָבֹ֔אוּ וְשָׂמ֣וּ אוֹתָ֗ם אִ֥ישׁ אִ֛ישׁ עַל־עֲבֹדָת֖וֹ וְאֶל־מַשָּׂאֽוֹ׃ כָּךְ עֲשׂוּ לָהֶם כְּדֵי שֶׁיִּחְיוּ וְלֹא יָמוּתוּ כְּשֶׁהֵם נִגָּשִׁים לָעֲבוֹדָה הַקְּדוֹשָׁה בְּיוֹתֵר – אַהֲרֹן וּבָנָיו יָבוֹאוּ וִימַנּוּ אוֹתָם כָּל אֶחָד עַל עֲבוֹדָתוֹ וְעַל מַשָּׂאוֹ.





* * *



כ וְלֹא־יָבֹ֧אוּ לִרְא֛וֹת כְּבַלַּ֥ע אֶת־הַקֹּ֖דֶשׁ וָמֵֽתוּ׃ וְלֹא יָבוֹאוּ לִרְאוֹת כְּשֶׁמְּכַסִּים אֶת כְּלֵי הַקּוֹדֶשׁ, וְיָמוּתוּ.
